28@component 

29class LLMMetadataExtractor: 

30 """ 

31 Extracts metadata from documents using a Large Language Model (LLM). 

32 

33 The metadata is extracted by providing a prompt to an LLM that generates the metadata. 

34 

35 This component expects as input a list of documents and a prompt. The prompt should have a variable called 

36 `document` that will point to a single document in the list of documents. So to access the content of the document, 

37 you can use `{{ document.content }}` in the prompt. 

38 

39 The component will run the LLM on each document in the list and extract metadata from the document. The metadata 

40 will be added to the document's metadata field. If the LLM fails to extract metadata from a document, the document 

41 will be added to the `failed_documents` list. The failed documents will have the keys `metadata_extraction_error` and 

42 `metadata_extraction_response` in their metadata. These documents can be re-run with another extractor to 

43 extract metadata by using the `metadata_extraction_response` and `metadata_extraction_error` in the prompt.

153 def __init__( 

154 self, 

155 prompt: str, 

156 chat_generator: ChatGenerator, 

157 expected_keys: list[str] | None = None, 

158 page_range: list[str | int] | None = None, 

159 raise_on_failure: bool = False, 

160 max_workers: int = 3, 

161 ) -> None: 

162 """ 

163 Initializes the LLMMetadataExtractor. 

164 

165 :param prompt: The prompt to be used for the LLM. 

166 :param chat_generator: a ChatGenerator instance which represents the LLM. In order for the component to work, 

167 the LLM should be configured to return a JSON object. For example, when using the OpenAIChatGenerator, you 

168 should pass `{"response_format": {"type": "json_object"}}` in the `generation_kwargs`. 

169 :param expected_keys: The keys expected in the JSON output from the LLM. 

170 :param page_range: A range of pages to extract metadata from. For example, page_range=['1', '3'] will extract 

171 metadata from the first and third pages of each document. It also accepts printable range strings, e.g.: 

172 ['1-3', '5', '8', '10-12'] will extract metadata from pages 1, 2, 3, 5, 8, 10,11, 12. 

173 If None, metadata will be extracted from the entire document for each document in the documents list. 

174 This parameter is optional and can be overridden in the `run` method. 

175 :param raise_on_failure: Whether to raise an error on failure during the execution of the Generator or 

176 validation of the JSON output. 

177 :param max_workers: The maximum number of workers to use in the thread pool executor. 

178 This parameter is used limit the maximum number of requests that should be allowed to run concurrently 

179 when using the `run_async` method. 

180 """ 

181 self.prompt = prompt 

182 ast = SandboxedEnvironment().parse(prompt) 

183 template_variables = meta.find_undeclared_variables(ast) 

184 variables = list(template_variables) 

185 if len(variables) > 1 or variables[0] != "document": 

186 raise ValueError( 

187 f"Prompt must have exactly one variable called 'document'. Found {','.join(variables)} in the prompt." 

188 ) 

189 self.builder = PromptBuilder(prompt, required_variables=variables) 

190 self.raise_on_failure = raise_on_failure 

191 self.expected_keys = expected_keys or [] 

192 self.splitter = DocumentSplitter(split_by="page", split_length=1) 

193 self.expanded_range = expand_page_range(page_range) if page_range else None 

194 self.max_workers = max_workers 

195 self._chat_generator = chat_generator 

196 self._is_warmed_up = False

239 def _extract_metadata(self, llm_answer: str) -> dict[str, Any]: 

240 try: 

241 parsed_metadata = _parse_dict_from_json(llm_answer, expected_keys=self.expected_keys, raise_on_failure=True) 

242 except (ValueError, json.JSONDecodeError) as e: 

243 logger.warning( 

244 "Response from the LLM is not valid JSON or missing expected keys. Received output: {response}", 

245 response=llm_answer, 

246 ) 

247 if self.raise_on_failure: 

248 raise e 

249 return {"error": "Response is not valid JSON or missing keys. Error: " + str(e)} 

250 

251 return parsed_metadata

253 def _prepare_prompts( 

254 self, documents: list[Document], expanded_range: list[int] | None = None 

255 ) -> list[ChatMessage | None]: 

256 all_prompts: list[ChatMessage | None] = [] 

257 for document in documents: 

258 if not document.content: 

259 logger.warning("Document {doc_id} has no content. Skipping metadata extraction.", doc_id=document.id) 

260 all_prompts.append(None) 

261 continue 

262 

263 if expanded_range: 

264 doc_copy = copy.deepcopy(document) 

265 pages = self.splitter.run(documents=[doc_copy]) 

266 content = "" 

267 for idx, page in enumerate(pages["documents"]): 

268 if idx + 1 in expanded_range and page.content is not None: 

269 content += page.content 

270 doc_copy = replace(doc_copy, content=content) 

271 else: 

272 doc_copy = document 

273 

274 prompt_with_doc = self.builder.run(template=self.prompt, template_variables={"document": doc_copy}) 

275 

276 # build a ChatMessage with the prompt 

277 message = ChatMessage.from_user(prompt_with_doc["prompt"]) 

278 all_prompts.append(message) 

279 

280 return all_prompts

282 def _run_on_thread(self, prompt: ChatMessage | None) -> dict[str, Any]: 

283 # If prompt is None, return an error dictionary 

284 if prompt is None: 

285 return {"error": "Document has no content, skipping LLM call."} 

286 

287 try: 

288 result = self._chat_generator.run(messages=[prompt]) 

289 except Exception as e: 

290 if self.raise_on_failure: 

291 raise e 

292 logger.exception( 

293 "LLM {class_name} execution failed. Skipping metadata extraction. Failed with exception '{error}'.", 

294 class_name=self._chat_generator.__class__.__name__, 

295 error=e, 

296 ) 

297 result = {"error": "LLM failed with exception: " + str(e)} 

298 return result

318 def _process_results( 

319 self, documents: list[Document], results: Iterable[dict[str, Any]] 

320 ) -> tuple[list[Document], list[Document]]: 

321 successful_documents = [] 

322 failed_documents = [] 

323 for document, result in zip(documents, results, strict=True): 

324 new_meta = {**document.meta} 

325 if "error" in result: 

326 new_meta["metadata_extraction_error"] = result["error"] 

327 new_meta["metadata_extraction_response"] = None 

328 failed_documents.append(replace(document, meta=new_meta)) 

329 continue 

330 

331 parsed_metadata = self._extract_metadata(result["replies"][0].text) 

332 if "error" in parsed_metadata: 

333 new_meta["metadata_extraction_error"] = parsed_metadata["error"] 

334 new_meta["metadata_extraction_response"] = result["replies"][0] 

335 failed_documents.append(replace(document, meta=new_meta)) 

336 continue 

337 

338 for key in parsed_metadata: 

339 new_meta[key] = parsed_metadata[key] 

340 # Remove metadata_extraction_error and metadata_extraction_response if present from previous runs 

341 new_meta.pop("metadata_extraction_error", None) 

342 new_meta.pop("metadata_extraction_response", None) 

343 successful_documents.append(replace(document, meta=new_meta)) 

344 return successful_documents, failed_documents

346 @component.output_types(documents=list[Document], failed_documents=list[Document]) 

347 def run(self, documents: list[Document], page_range: list[str | int] | None = None) -> dict[str, Any]: 

348 """ 

349 Extract metadata from documents using a Large Language Model. 

350 

351 If `page_range` is provided, the metadata will be extracted from the specified range of pages. This component 

352 will split the documents into pages and extract metadata from the specified range of pages. The metadata will be 

353 extracted from the entire document if `page_range` is not provided. 

354 

355 The original documents will be returned updated with the extracted metadata. 

356 

357 :param documents: List of documents to extract metadata from. 

358 :param page_range: A range of pages to extract metadata from. For example, page_range=['1', '3'] will extract 

359 metadata from the first and third pages of each document. It also accepts printable range 

360 strings, e.g.: ['1-3', '5', '8', '10-12'] will extract metadata from pages 1, 2, 3, 5, 8, 10, 

361 11, 12. 

362 If None, metadata will be extracted from the entire document for each document in the 

363 documents list. 

364 :returns: 

365 A dictionary with the keys: 

366 - "documents": A list of documents that were successfully updated with the extracted metadata. 

367 - "failed_documents": A list of documents that failed to extract metadata. These documents will have 

368 "metadata_extraction_error" and "metadata_extraction_response" in their metadata. These documents can be 

369 re-run with the extractor to extract metadata. 

370 """ 

371 if len(documents) == 0: 

372 logger.warning("No documents provided. Skipping metadata extraction.") 

373 return {"documents": [], "failed_documents": []} 

374 

375 if not self._is_warmed_up: 

376 self.warm_up() 

377 

378 expanded_range = self.expanded_range 

379 if page_range: 

380 expanded_range = expand_page_range(page_range) 

381 

382 # Create ChatMessage prompts for each document 

383 all_prompts = self._prepare_prompts(documents=documents, expanded_range=expanded_range) 

384 

385 # Run the LLM on each prompt 

386 with ThreadPoolExecutor(max_workers=self.max_workers) as executor: 

387 results = executor.map(self._run_on_thread, all_prompts) 

388 

389 successful_documents, failed_documents = self._process_results(documents, results) 

390 

391 return {"documents": successful_documents, "failed_documents": failed_documents}

393 @component.output_types(documents=list[Document], failed_documents=list[Document]) 

394 async def run_async(self, documents: list[Document], page_range: list[str | int] | None = None) -> dict[str, Any]: 

395 """ 

396 Asynchronously extract metadata from documents using a Large Language Model. 

397 

398 If `page_range` is provided, the metadata will be extracted from the specified range of pages. This component 

399 will split the documents into pages and extract metadata from the specified range of pages. The metadata will be 

400 extracted from the entire document if `page_range` is not provided. 

401 

402 The original documents will be returned updated with the extracted metadata. 

403 

404 This is the asynchronous version of the `run` method. It has the same parameters 

405 and return values but can be used with `await` in an async code. 

406 

407 :param documents: List of documents to extract metadata from. 

408 :param page_range: A range of pages to extract metadata from. For example, page_range=['1', '3'] will extract 

409 metadata from the first and third pages of each document. It also accepts printable range 

410 strings, e.g.: ['1-3', '5', '8', '10-12'] will extract metadata from pages 1, 2, 3, 5, 8, 10, 

411 11, 12. 

412 If None, metadata will be extracted from the entire document for each document in the 

413 documents list. 

414 :returns: 

415 A dictionary with the keys: 

416 - "documents": A list of documents that were successfully updated with the extracted metadata. 

417 - "failed_documents": A list of documents that failed to extract metadata. These documents will have 

418 "metadata_extraction_error" and "metadata_extraction_response" in their metadata. These documents can be 

419 re-run with the extractor to extract metadata. 

420 """ 

421 if not hasattr(self._chat_generator, "run_async"): 

422 logger.warning( 

423 "{chat_generator_type} does not implement method 'run_async'. Falling back to 'run'.", 

424 chat_generator_type=type(self._chat_generator).__name__, 

425 ) 

426 return self.run(documents, page_range) 

427 

428 if len(documents) == 0: 

429 logger.warning("No documents provided. Skipping metadata extraction.") 

430 return {"documents": [], "failed_documents": []} 

431 

432 if not self._is_warmed_up: 

433 self.warm_up() 

434 

435 expanded_range = self.expanded_range 

436 if page_range: 

437 expanded_range = expand_page_range(page_range) 

438 

439 # Create ChatMessage prompts for each document 

440 all_prompts = self._prepare_prompts(documents=documents, expanded_range=expanded_range) 

441 

442 # Run the LLM on each prompt 

443 sem = Semaphore(max(1, self.max_workers)) 

444 async with sem: 

445 results = await gather(*[self._run_async(prompt) for prompt in all_prompts]) 

446 

447 successful_documents, failed_documents = self._process_results(documents, results) 

448 

449 return {"documents": successful_documents, "failed_documents": failed_documents}
