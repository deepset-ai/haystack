63def _merge_tool_outputs_into_state(tool: Tool, result: Any, state: State) -> None: 

64 """ 

65 Write tool outputs into State according to the tool's `outputs_to_state` mapping. 

66 

67 :raises RuntimeError: If writing an output value into the state fails. 

68 """ 

69 if not isinstance(result, dict): 

70 return 

71 

72 for state_key, config in (tool.outputs_to_state or {}).items(): 

73 source_key = config.get("source", None) 

74 if source_key and source_key not in result: 

75 continue

83def _result_to_string(result: Any) -> str: 

84 """ 

85 Convert a tool result to a string. 

86 

87 Strings are returned as-is; all other types are passed through a JSON serialization step to produce more readable 

88 output, with a fallback to plain str() conversion if serialization fails. 

89 

90 :param result: The tool result to convert. 

91 :returns: A string representation of the tool result. 

92 """ 

93 if isinstance(result, str): 

94 return result 

95 serializable = _serializable_value(value=result, use_placeholders=False)

97 return json.dumps(serializable, ensure_ascii=False) 

98 except Exception as error: 

99 logger.warning( 

100 "Tool result is not JSON serializable. Falling back to str conversion. Result: {result}\nError: {err}", 

101 result=result, 

102 err=error, 

103 ) 

104 return str(result)

107def _process_tool_output(config: dict[str, Any], result: Any, tool_call: ToolCall, *, raise_on_failure: bool) -> Any: 

108 """ 

109 Extract and convert a single tool output according to `config`. 

110 

111 `config` may contain `source` (key to extract from result dict), `handler` (conversion callable), and 

112 `raw_result` (return the value without string conversion). 

113 

114 If a configured `handler` raises, the exception is re-raised when `raise_on_failure` is True; otherwise 

115 a warning is logged and the value is converted via `_result_to_string`. 

116 """ 

117 source_key = config.get("source") 

118 value = result.get(source_key) if source_key is not None and isinstance(result, dict) else result 

119 

120 handler = config.get("handler") 

121 raw_result = config.get("raw_result", False) 

122 

123 if handler is None: 

124 # raw result is mostly used to allow ImageContent or TextContent blocks to be directly returned and consumed 

125 # by ChatMessage.from_tool without string conversion. 

126 if raw_result: 

127 return value 

128 return _result_to_string(value) 

129 

130 try: 

131 return handler(value) 

132 except Exception as e: 

133 if raise_on_failure: 

134 raise 

135 logger.warning( 

136 "Output handler '{handler}' for tool '{tool}' failed, falling back to string conversion. Error: {err}", 

137 handler=handler.__name__, 

138 tool=tool_call.tool_name, 

139 err=e, 

140 ) 

141 return _result_to_string(value)

144def _build_tool_result_message(result: Any, tool_call: ToolCall, tool: Tool, *, raise_on_failure: bool) -> ChatMessage: 

145 """Convert a raw tool result into a ChatMessage, applying `outputs_to_string` config if present.""" 

146 outputs_config = tool.outputs_to_string or {} 

147 

148 # Single-output config (or no config): keys are at the root level 

149 if not outputs_config or any(k in outputs_config for k in ("source", "handler", "raw_result")): 

150 tool_result = _process_tool_output(outputs_config, result, tool_call, raise_on_failure=raise_on_failure) 

151 return ChatMessage.from_tool(tool_result=tool_result, origin=tool_call) 

152 

153 # Multi-output config: each key maps to its own sub-config — stringify each value, then stringify the whole dict 

154 tool_result_dict = { 

155 output_key: _process_tool_output( 

156 {**cfg, "raw_result": False}, result, tool_call, raise_on_failure=raise_on_failure 

157 ) 

158 for output_key, cfg in outputs_config.items() 

159 } 

160 return ChatMessage.from_tool(tool_result=_result_to_string(tool_result_dict), origin=tool_call)

181def _create_tool_span(tool: Tool, tool_call: ToolCall) -> Any: 

182 """ 

183 Create one tracing span for a single tool call, nested under the currently active span. 

184 

185 The established standard for tracing agents is one span per tool call, so each call gets its own span rather than 

186 grouping all of a step's calls together. The OpenTelemetry GenAI semantic conventions codify this with a dedicated 

187 "execute tool" span per invocation 

188 (https://github.com/open-telemetry/semantic-conventions-genai/blob/main/docs/gen-ai/gen-ai-agent-spans.md#execute-tool-span), 

189 and tracing backends such as Langfuse follow the same model. The span is tagged with the tool's identity; the caller 

190 adds the call arguments and result as content tags. 

191 """ 

192 return tracing.tracer.trace( 

193 "haystack.agent.step.tool", 

194 tags={"haystack.tool.name": tool_call.tool_name, "haystack.tool.description": tool.description}, 

195 parent_span=tracing.tracer.current_span(), 

196 )

199def _make_context_bound_invoke(tool: Tool, args: dict[str, Any], tool_call: ToolCall) -> Callable[[], Any]: 

200 """ 

201 Return a zero-arg callable that runs `tool.invoke(**args)` under the current contextvars snapshot. 

202 

203 This preserves tracing spans and other context-local state across thread-pool boundaries, so the per-call span 

204 created inside the worker nests correctly under the step span. The callable returns a ToolInvocationError instead 

205 of raising so that parallel executions can collect failures without aborting the whole batch. 

206 """ 

207 ctx = contextvars.copy_context() 

208 

209 def _invoke() -> Any: 

210 with _create_tool_span(tool, tool_call) as span: 

211 span.set_content_tag("haystack.agent.step.tool.input", tool_call.arguments) 

212 try: 

213 result = tool.invoke(**args) 

214 except ToolInvocationError as e: 

215 span.set_content_tag("haystack.agent.step.tool.output", {"error": str(e)}) 

216 return e 

217 span.set_content_tag("haystack.agent.step.tool.output", result) 

218 return result 

219 

220 def _runner() -> Any: 

221 return ctx.run(_invoke) 

222 

223 return _runner

226def _make_bounded_invoke_async( 

227 tool: Tool, args: dict[str, Any], semaphore: asyncio.Semaphore, tool_call: ToolCall 

228) -> Callable[[], Any]: 

229 """ 

230 Return a zero-arg async callable that awaits `tool.invoke_async(**args)` while holding `semaphore`. 

231 

232 Concurrency is bounded uniformly across native-async tools and sync-fallback tools (which dispatch 

233 to a worker thread inside `Tool.invoke_async`). ContextVars naturally inherit into child tasks for 

234 the native-async branch, and `asyncio.to_thread` propagates them for the fallback branch, so the per-call 

235 span nests correctly under the step span. 

236 

237 Returns a `ToolInvocationError` instead of raising so that gathered executions can collect failures 

238 without aborting the whole batch. 

239 """ 

240 

241 async def _runner() -> Any: 

242 async with semaphore: 

243 with _create_tool_span(tool, tool_call) as span: 

244 span.set_content_tag("haystack.agent.step.tool.input", tool_call.arguments) 

245 try: 

246 result = await tool.invoke_async(**args) 

247 except ToolInvocationError as e: 

248 span.set_content_tag("haystack.agent.step.tool.output", {"error": str(e)}) 

249 return e 

250 span.set_content_tag("haystack.agent.step.tool.output", result) 

251 return result 

252 

253 return _runner

277def _inject_state_args(tool: Tool, llm_args: dict[str, Any], state: State) -> dict[str, Any]: 

278 """ 

279 Merge LLM-provided arguments with state-sourced arguments. 

280 

281 LLM args take precedence. State values are pulled in only for the keys a tool explicitly declares via its 

282 `inputs_from_state` mapping, then the live State object is injected for any param annotated as State. 

283 

284 :param tool: The tool being invoked, used to determine parameter mappings and State injection. 

285 :param llm_args: The arguments provided by the LLM, which take precedence over state values. 

286 :param state: The current runtime state, used to source additional arguments as needed. 

287 :returns: A dict of arguments to invoke the tool with, combining LLM and state values according to the rules 

288 described above. 

289 """ 

290 final_args = dict(llm_args) 

291 func_params = _get_func_params(tool) 

292 

293 # A tool reads from State by name only via an explicit `inputs_from_state` mapping 

294 for state_key, param_name in (tool.inputs_from_state or {}).items(): 

295 if param_name not in final_args and state.has(state_key): 

296 final_args[param_name] = state.get(state_key) 

297 

298 # We also inject the full State object for any parameter annotated as State 

299 for param_name, param_type in func_params.items(): 

300 if _unwrap_optional(param_type) is State: 

301 final_args[param_name] = state 

302 

303 return final_args

344def _resolve_tool_calls( 

345 messages_with_tool_calls: list[ChatMessage], tools_with_names: dict[str, Tool], *, raise_on_failure: bool 

346) -> tuple[list[ToolCall], list[Tool], list[ChatMessage]]: 

347 """ 

348 Walk all tool calls in `messages_with_tool_calls` and resolve each to its Tool. 

349 

350 Argument preparation is deliberately *not* done here: args are prepared per execution batch (see 

351 `_schedule_tool_calls`) so that a tool reading from State observes writes made by tools that ran earlier in the same 

352 step. 

353 

354 :returns: (tool_calls, resolved_tools, error_messages) 

355 - tool_calls: ToolCall objects for each valid call, in call order 

356 - resolved_tools: the resolved Tool for each entry in `tool_calls` (parallel list) 

357 - error_messages: ChatMessages for tool-not-found errors (when raise_on_failure is False) 

358 """ 

359 tool_calls: list[ToolCall] = [] 

360 resolved_tools: list[Tool] = [] 

361 error_messages: list[ChatMessage] = [] 

362 

363 for message in messages_with_tool_calls: 

364 for tool_call in message.tool_calls: 

365 tool_name = tool_call.tool_name 

366 

367 if tool_name not in tools_with_names: 

368 error = ToolNotFoundException(tool_name, list(tools_with_names.keys())) 

369 if raise_on_failure: 

370 raise error 

371 logger.error("{error_exception}", error_exception=error) 

372 error_messages.append(ChatMessage.from_tool(tool_result=str(error), origin=tool_call, error=True)) 

373 continue 

374 

375 tool_calls.append(tool_call) 

376 resolved_tools.append(tools_with_names[tool_name]) 

377 

378 return tool_calls, resolved_tools, error_messages

381def _keys_intersect(a: _StateKeys, b: _StateKeys) -> bool: 

382 """ 

383 Return whether two State-key sets share at least one key, treating `_ALL_STATE_KEYS` as a wildcard. 

384 

385 Used to detect read-after-write dependencies between tool calls: the reader's read set is tested against the 

386 writer's write set. 

387 

388 :param a: A set of state keys, or the `_ALL_STATE_KEYS` wildcard meaning "every key". 

389 :param b: A set of state keys, or the `_ALL_STATE_KEYS` wildcard meaning "every key". 

390 :returns: True if the sets overlap (a wildcard overlaps any non-empty set, and two wildcards always overlap). 

391 """ 

392 if a is _ALL_STATE_KEYS: 

393 # `a` covers every key, so it overlaps `b` as long as `b` touches any key. Two wildcards always overlap; 

394 # otherwise `bool(b)` is True iff the concrete set `b` is non-empty. 

395 return b is _ALL_STATE_KEYS or bool(b) 

396 if b is _ALL_STATE_KEYS: 

397 # Symmetric case: wildcard `b` overlaps `a` iff the concrete set `a` is non-empty (`bool(set)` == non-empty). 

398 return bool(a) 

399 # Both are concrete sets: they overlap iff their set intersection is non-empty. 

400 return bool(a & b) # type: ignore[operator]

403def _state_io_for_call(tool: Tool, llm_args: dict[str, Any]) -> tuple[_StateKeys, _StateKeys]: 

404 """ 

405 Compute the State keys a tool call reads from and writes to. 

406 

407 Mirrors the resolution logic in `_inject_state_args`: 

408 - A tool with a `State`-annotated parameter can read/write any key, so both sets are the `_ALL_STATE_KEYS` wildcard. 

409 - Otherwise reads come from the tool's explicit `inputs_from_state` mapping, excluding any parameter the LLM already 

410 supplied (LLM args take precedence and short-circuit the state lookup). 

411 - Writes are the keys in `outputs_to_state`. 

412 

413 :returns: A `(reads, writes)` tuple of state-key sets (or the `_ALL_STATE_KEYS` wildcard). 

414 """ 

415 func_params = _get_func_params(tool) 

416 # Check if State is in func_params 

417 if any(_unwrap_optional(param_type) is State for param_type in func_params.values()): 

418 return _ALL_STATE_KEYS, _ALL_STATE_KEYS 

419 

420 # Calculate reads 

421 param_mappings = tool.inputs_from_state or {} 

422 reads = {state_key for state_key, param_name in param_mappings.items() if param_name not in llm_args} 

423 # Calculate writes 

424 writes = set((tool.outputs_to_state or {}).keys()) 

425 

426 return reads, writes

429def _schedule_tool_calls(tool_calls: list[ToolCall], tools: list[Tool]) -> list[list[int]]: 

430 """ 

431 Group tool calls into ordered execution batches based on their State read/write sets. 

432 

433 Calls within a batch are mutually independent and run in parallel; batches run sequentially. The schedule guarantees 

434 that a call reading a State key always runs in a later batch than any call (in the same step) that writes that 

435 key — so read-after-write dependencies are honored regardless of the order the LLM requested the calls in. 

436 

437 This is a layered topological sort: each round, every call whose dependencies have all been scheduled forms the 

438 next parallel batch. Dependency cycles — e.g. a tool that both reads and writes the same key, requested more than 

439 once — cannot be ordered by the read-after-write rule alone, so they are broken deterministically by call order 

440 (the lowest-index remaining call runs next, on its own). 

441 

442 Pure write-write overlaps create no dependency: nobody reads the contended key, and outputs are merged into State 

443 sequentially in call order afterward, so the result stays deterministic without serializing execution. 

444 

445 :param tool_calls: The tool calls to schedule, in call order. 

446 :param tools: The resolved Tool for each entry in `tool_calls` (parallel list). 

447 :returns: A list of batches, each a list of indices into `tool_calls`. 

448 """ 

449 # Per-call (reads, writes) State-key sets, in call order. 

450 io_list = [_state_io_for_call(tool, tc.arguments) for tc, tool in zip(tool_calls, tools, strict=True)] 

451 n = len(io_list) 

452 

453 # deps[j] = indices that must run before j because j reads a key they write (read-after-write). 

454 deps: list[set[int]] = [set() for _ in range(n)] 

455 for j in range(n): 

456 reads_j, _ = io_list[j] 

457 for i in range(n): 

458 if i == j: 

459 continue 

460 _, writes_i = io_list[i] 

461 if _keys_intersect(reads_j, writes_i): 

462 deps[j].add(i) 

463 

464 scheduled = [False] * n 

465 done: set[int] = set() 

466 batches: list[list[int]] = [] 

467 

468 while len(done) < n: 

469 # A call is ready once every writer it depends on has already been scheduled (`deps[k] <= done`, i.e. its 

470 # dependency set is a subset of the already-done set). All ready calls have no dependency on each other — 

471 # if one read a key another writes, it would still be waiting — so the whole `ready` list runs in parallel. 

472 ready = [k for k in range(n) if not scheduled[k] and deps[k] <= done] 

473 if not ready: 

474 # A dependency cycle remains: break it deterministically by running the lowest-index call next. 

475 ready = [next(k for k in range(n) if not scheduled[k])] 

476 for k in ready: 

477 scheduled[k] = True 

478 done.update(ready) 

479 batches.append(ready) 

480 

481 return batches

504def _run_tool( 

505 *, 

506 messages: list[ChatMessage], 

507 state: State, 

508 tools: ToolsType, 

509 streaming_callback: StreamingCallbackT | None = None, 

510 raise_on_failure: bool = True, 

511 enable_streaming_callback_passthrough: bool = False, 

512 max_workers: int = 4, 

513) -> tuple[list[ChatMessage], State]: 

514 """ 

515 Invoke all tools referenced by tool calls in `messages`. 

516 

517 :param messages: ChatMessage objects that may contain tool calls. 

518 :param state: Runtime state passed to and updated by tools. 

519 :param tools: The tools available for invocation. 

520 :param streaming_callback: Called once per tool result as it becomes available. 

521 :param raise_on_failure: If True, raise on tool invocation failure; otherwise return an error message. 

522 :param enable_streaming_callback_passthrough: If True, pass the streaming callback to tools that accept it. 

523 :param max_workers: Maximum number of parallel tool invocations. 

524 :returns: (tool_messages, updated_state) 

525 """ 

526 tools_with_names = _validate_and_prepare_tools(tools) 

527 

528 messages_with_tool_calls = [m for m in messages if m.tool_calls] 

529 if not messages_with_tool_calls: 

530 return [], state 

531 

532 tool_calls, resolved_tools, error_messages = _resolve_tool_calls( 

533 messages_with_tool_calls, tools_with_names, raise_on_failure=raise_on_failure 

534 ) 

535 if not tool_calls: 

536 return error_messages, state 

537 

538 # Group the calls into batches that honor read-after-write dependencies on State (see `_schedule_tool_calls`). 

539 batches = _schedule_tool_calls(tool_calls, resolved_tools) 

540 

541 # Results are indexed by call position so the returned messages stay in call order, even though batches may 

542 # execute the calls in a different order. 

543 results: list[ChatMessage | None] = [None] * len(tool_calls) 

544 stream_index = 0 

545 

546 with ThreadPoolExecutor(max_workers=max_workers) as executor: 

547 for batch in batches: 

548 # Prepare args at the start of each batch so tools that read from State observe writes merged by earlier 

549 # batches. 

550 futures = {} 

551 for idx in batch: 

552 args = _prepare_tool_args( 

553 tool=resolved_tools[idx], 

554 tool_call_arguments=tool_calls[idx].arguments, 

555 state=state, 

556 streaming_callback=streaming_callback, 

557 enable_streaming_passthrough=enable_streaming_callback_passthrough, 

558 ) 

559 futures[idx] = executor.submit(_make_context_bound_invoke(resolved_tools[idx], args, tool_calls[idx])) 

560 

561 # Merge results in call order within the batch so write-write merges stay deterministic. 

562 for idx in batch: 

563 message = _finalize_tool_result( 

564 futures[idx].result(), 

565 tool_calls[idx], 

566 resolved_tools[idx], 

567 state, 

568 raise_on_failure=raise_on_failure, 

569 ) 

570 results[idx] = message 

571 if streaming_callback is not None: 

572 streaming_callback(_create_tool_result_streaming_chunk(message, tool_calls[idx], stream_index)) 

573 stream_index += 1 

574 

575 tool_messages = error_messages + [m for m in results if m is not None] 

576 

577 # We emit a final empty chunk with finish_reason "tool_call_results" to signal the end of the tool results stream. 

578 if tool_messages and streaming_callback is not None: 

579 streaming_callback( 

580 StreamingChunk(content="", finish_reason="tool_call_results", meta={"finish_reason": "tool_call_results"}) 

581 ) 

582 

583 return tool_messages, state

586async def _run_tool_async( 

587 *, 

588 messages: list[ChatMessage], 

589 state: State, 

590 tools: ToolsType, 

591 streaming_callback: StreamingCallbackT | None = None, 

592 raise_on_failure: bool = True, 

593 enable_streaming_callback_passthrough: bool = False, 

594 max_workers: int = 4, 

595) -> tuple[list[ChatMessage], State]: 

596 """ 

597 Asynchronous variant of `run_tool`. Tool calls execute concurrently via a thread pool. 

598 

599 :param messages: ChatMessage objects that may contain tool calls. 

600 :param state: Runtime state passed to and updated by tools. 

601 :param tools: The tools available for invocation. 

602 :param streaming_callback: Async callback called once per tool result. 

603 :param raise_on_failure: If True, raise on tool invocation failure; otherwise return an error message. 

604 :param enable_streaming_callback_passthrough: If True, pass the streaming callback to tools that accept it. 

605 :param max_workers: Maximum number of parallel tool invocations. 

606 :returns: (tool_messages, updated_state) 

607 """ 

608 tools_with_names = _validate_and_prepare_tools(tools) 

609 

610 messages_with_tool_calls = [m for m in messages if m.tool_calls] 

611 if not messages_with_tool_calls: 

612 return [], state 

613 

614 tool_calls, resolved_tools, error_messages = _resolve_tool_calls( 

615 messages_with_tool_calls, tools_with_names, raise_on_failure=raise_on_failure 

616 ) 

617 if not tool_calls: 

618 return error_messages, state 

619 

620 # Group the calls into batches that honor read-after-write dependencies on State (see `_schedule_tool_calls`). 

621 batches = _schedule_tool_calls(tool_calls, resolved_tools) 

622 

623 # Results are indexed by call position so the returned messages stay in call order, even though batches may 

624 # execute the calls in a different order. 

625 results: list[ChatMessage | None] = [None] * len(tool_calls) 

626 stream_index = 0 

627 

628 # `max_workers` + Semaphore bounds concurrency for both sync and async tool calls async tools are awaited directly, 

629 # and sync tools are dispatched to a worker thread inside `Tool.invoke_async`. 

630 semaphore = asyncio.Semaphore(max_workers) 

631 

632 for batch in batches: 

633 # Prepare args at the start of each batch so readers observe writes merged by earlier batches. 

634 tasks = {} 

635 for idx in batch: 

636 args = _prepare_tool_args( 

637 tool=resolved_tools[idx], 

638 tool_call_arguments=tool_calls[idx].arguments, 

639 state=state, 

640 streaming_callback=streaming_callback, 

641 enable_streaming_passthrough=enable_streaming_callback_passthrough, 

642 ) 

643 tasks[idx] = _make_bounded_invoke_async(resolved_tools[idx], args, semaphore, tool_calls[idx])() 

644 batch_results = await asyncio.gather(*tasks.values()) 

645 

646 # Merge results in call order within the batch so write-write merges stay deterministic. 

647 for idx, result in zip(tasks.keys(), batch_results, strict=True): 

648 message = _finalize_tool_result( 

649 result, tool_calls[idx], resolved_tools[idx], state, raise_on_failure=raise_on_failure 

650 ) 

651 results[idx] = message 

652 if streaming_callback is not None: 

653 await _invoke_streaming_callback( 

654 streaming_callback, _create_tool_result_streaming_chunk(message, tool_calls[idx], stream_index) 

655 ) 

656 stream_index += 1 

657 

658 tool_messages = error_messages + [m for m in results if m is not None] 

659 

660 # We emit a final empty chunk with finish_reason "tool_call_results" to signal the end of the tool results stream. 

661 if tool_messages and streaming_callback is not None: 

662 await _invoke_streaming_callback( 

663 streaming_callback, 

664 StreamingChunk(content="", finish_reason="tool_call_results", meta={"finish_reason": "tool_call_results"}), 

665 ) 

666 

667 return tool_messages, state
