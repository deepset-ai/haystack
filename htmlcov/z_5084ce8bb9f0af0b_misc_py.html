33def expand_page_range(page_range: list[str | int]) -> list[int]: 

34 """ 

35 Takes a list of page numbers and ranges and expands them into a list of page numbers. 

36 

37 For example, given a page_range=['1-3', '5', '8', '10-12'] the function will return [1, 2, 3, 5, 8, 10, 11, 12] 

38 

39 :param page_range: List of page numbers and ranges 

40 :returns: 

41 An expanded list of page integers 

42 

43 """ 

44 expanded_page_range = [] 

45 

46 for page in page_range: 

47 if isinstance(page, int): 

48 # check if it's a range wrongly passed as an integer expression 

49 if "-" in str(page): 

50 msg = "range must be a string in the format 'start-end'" 

51 raise ValueError(f"Invalid page range: {page} - {msg}") 

52 expanded_page_range.append(page) 

53 

54 elif isinstance(page, str) and page.isdigit(): 

55 expanded_page_range.append(int(page)) 

56 

57 elif isinstance(page, str) and "-" in page: 

58 start, end = page.split("-") 

59 expanded_page_range.extend(range(int(start), int(end) + 1)) 

60 

61 else: 

62 msg = "range must be a string in the format 'start-end' or an integer" 

63 raise ValueError(f"Invalid page range: {page} - {msg}") 

64 

65 if not expanded_page_range: 

66 raise ValueError("No valid page numbers or ranges found in the input list") 

67 

68 return expanded_page_range

98def _get_output_dir(out_dir: str) -> str: 

99 """ 

100 Find or create a writable directory for saving status files. 

101 

102 Tries in the following order: 

103 

104 1. ~/.haystack/{out_dir} 

105 2. {tempdir}/haystack/{out_dir} 

106 3. ./.haystack/{out_dir} 

107 

108 :raises RuntimeError: If no directory could be created. 

109 :returns: 

110 The path to the created directory. 

111 """ 

112 

113 candidates = [ 

114 Path.home() / ".haystack" / out_dir, 

115 Path(tempfile.gettempdir()) / "haystack" / out_dir, 

116 Path.cwd() / ".haystack" / out_dir, 

117 ] 

118 

119 for candidate in candidates: 

120 try: 

121 candidate.mkdir(parents=True, exist_ok=True) 

122 return str(candidate) 

123 except Exception: 

124 continue 

125 

126 raise RuntimeError( 

127 f"Could not create a writable directory for output files in any of the following locations: {candidates}" 

128 )

131def _deduplicate_documents(documents: list["Document"]) -> list["Document"]: 

132 """ 

133 Deduplicate a list of documents by their id keeping the duplicate with the highest score if a score is present. 

134 

135 :param documents: List of documents to deduplicate. 

136 :returns: List of deduplicated documents. 

137 """ 

138 # Keep for each Document id the one with the highest score 

139 highest_scoring_docs: dict[str, "Document"] = {} 

140 for doc in documents: 

141 score = doc.score if doc.score is not None else -inf 

142 best = highest_scoring_docs.get(doc.id) 

143 

144 if best is None or score > (best.score if best.score is not None else -inf): 

145 highest_scoring_docs[doc.id] = doc 

146 

147 return list(highest_scoring_docs.values())

150def _reciprocal_rank_fusion( 

151 document_lists: list[list["Document"]], weights: list[float] | None = None 

152) -> list["Document"]: 

153 """ 

154 Merge multiple ranked lists of Documents using Reciprocal Rank Fusion, deduplicating across lists. 

155 

156 See the original paper: https://plg.uwaterloo.ca/~gvcormac/cormacksigir09-rrf.pdf 

157 

158 The constant k is set to 61 (60 was suggested by the original paper, plus 1 as python lists are 0-based and the 

159 paper used 1-based ranking). 

160 

161 :param document_lists: A list of ranked document lists to fuse. 

162 :param weights: Optional per-list weights. Defaults to equal weights. 

163 :returns: 

164 Deduplicated list of documents with updated RRF scores. 

165 """ 

166 if not document_lists: 

167 return [] 

168 

169 k = 61 

170 scores_map: dict = defaultdict(int) 

171 documents_map: dict = {} 

172 resolved_weights = weights if weights else [1 / len(document_lists)] * len(document_lists) 

173 

174 for documents, weight in zip(document_lists, resolved_weights, strict=True): 

175 for rank, doc in enumerate(documents): 

176 scores_map[doc.id] += (weight * len(document_lists)) / (k + rank) 

177 documents_map[doc.id] = doc 

178 

179 for _id in scores_map: 

180 scores_map[_id] /= len(document_lists) / k 

181 

182 return [replace(doc, score=scores_map[doc.id]) for doc in documents_map.values()]

197def _parse_dict_from_json( 

198 text: str, expected_keys: list[str] | None = None, raise_on_failure: bool = True 

199) -> dict[str, Any] | None: 

200 """ 

201 Parses a JSON string containing a dictionary. 

202 

203 :param text: The string to parse. 

204 :param expected_keys: A list of keys that must be present in the parsed dictionary. 

205 :param raise_on_failure: If True, raises an exception on failure. If False, logs a warning and returns None. 

206 

207 :return: The parsed dictionary, or None if parsing fails and raise_on_failure is False. 

208 :raises json.JSONDecodeError: If the text is not valid JSON and raise_on_failure is True. 

209 :raises ValueError: If the parsed object is not a dictionary or has missing expected keys, 

210 and `raise_on_failure` is True. 

211 """ 

212 cleaned_text = text.strip() 

213 

214 try: 

215 parsed_json = json.loads(cleaned_text) 

216 except json.JSONDecodeError as e: 

217 if raise_on_failure: 

218 raise e 

219 logger.warning("Failed to parse JSON from text: {text}. Error: {error}", text=text, error=e) 

220 return None 

221 

222 if not isinstance(parsed_json, dict): 

223 if raise_on_failure: 

224 raise ValueError(f"Expected a JSON object containing a dictionary but got {type(parsed_json).__name__}") 

225 logger.warning( 

226 "Expected a JSON object containing a dictionary but got {type}. Returning None", 

227 type=type(parsed_json).__name__, 

228 ) 

229 return None 

230 

231 if not expected_keys: 

232 return parsed_json 

233 

234 missing_keys = [key for key in expected_keys if key not in parsed_json] 

235 if missing_keys: 

236 if raise_on_failure: 

237 raise ValueError(f"Missing expected keys in JSON: {missing_keys}. Got keys: {list(parsed_json.keys())}") 

238 logger.warning( 

239 "Missing expected keys in JSON: {missing_keys}. Got keys: {keys}", 

240 missing_keys=missing_keys, 

241 keys=list(parsed_json.keys()), 

242 ) 

243 return None 

244 

245 return parsed_json
