18@dataclass 

19class Tool: 

20 """ 

21 Data class representing a Tool that Language Models can prepare a call for. 

22 

23 Accurate definitions of the textual attributes such as `name` and `description` 

24 are important for the Language Model to correctly prepare the call. 

25 

26 For resource-intensive operations like establishing connections to remote services or 

27 loading models, override the `warm_up()` method. This method is called before the Tool 

28 is used and should be idempotent, as it may be called multiple times during 

29 pipeline/agent setup. 

30 

31 :param name: 

32 Name of the Tool. 

33 :param description: 

34 Description of the Tool. 

35 :param parameters: 

36 A JSON schema defining the parameters expected by the Tool. 

37 :param function: 

38 The function that will be invoked when the Tool is called. 

39 Must be a synchronous function; async functions are not supported. 

40 :param outputs_to_string: 

41 Optional dictionary defining how tool outputs should be converted into string(s) or results. 

42 If not provided, the tool result is converted to a string using a default handler. 

43 

44 `outputs_to_string` supports two formats: 

45 

46 1. Single output format - use "source", "handler", and/or "raw_result" at the root level: 

47 ```python 

48 { 

49 "source": "docs", "handler": format_documents, "raw_result": False 

50 } 

51 ``` 

52 - `source`: If provided, only the specified output key is sent to the handler. If not provided, the whole 

53 tool result is sent to the handler. 

54 - `handler`: A function that takes the tool output (or the extracted source value) and returns the 

55 final result. 

56 - `raw_result`: If `True`, the result is returned raw without string conversion, but applying the `handler` 

57 if provided. This is intended for tools that return images. In this mode, the Tool function or the 

58 `handler` must return a list of `TextContent`/`ImageContent` objects to ensure compatibility with Chat 

59 Generators. 

60 

61 2. Multiple output format - map keys to individual configurations: 

62 ```python 

63 { 

64 "formatted_docs": {"source": "docs", "handler": format_documents}, 

65 "summary": {"source": "summary_text", "handler": str.upper} 

66 } 

67 ``` 

68 Each key maps to a dictionary that can contain "source" and/or "handler". 

69 Note that `raw_result` is not supported in the multiple output format. 

70 :param inputs_from_state: 

71 Optional dictionary mapping state keys to tool parameter names. 

72 Example: `{"repository": "repo"}` maps state's "repository" to tool's "repo" parameter. 

73 :param outputs_to_state: 

74 Optional dictionary defining how tool outputs map to keys within state as well as optional handlers. 

75 If the source is provided only the specified output key is sent to the handler. 

76 Example: 

77 ```python 

78 { 

79 "documents": {"source": "docs", "handler": custom_handler} 

80 } 

81 ``` 

82 If the source is omitted the whole tool result is sent to the handler. 

83 Example: 

84 ```python 

85 { 

86 "documents": {"handler": custom_handler} 

87 } 

88 ``` 

89 :raises ValueError: If `function` is async, if `parameters` is not a valid JSON schema, or if the 

90 `outputs_to_state`, `outputs_to_string`, or `inputs_from_state` configurations are invalid. 

91 :raises TypeError: If any configuration value in `outputs_to_state`, `outputs_to_string`, or 

92 `inputs_from_state` has the wrong type. 

93 """ 

94 

95 name: str 

96 description: str 

97 parameters: dict[str, Any] 

98 function: Callable 

99 outputs_to_string: dict[str, Any] | None = None 

100 inputs_from_state: dict[str, str] | None = None 

101 outputs_to_state: dict[str, dict[str, Any]] | None = None

103 def __post_init__(self) -> None: # noqa: C901, PLR0912 

104 # Check that the function is not a coroutine (async function) 

105 if inspect.iscoroutinefunction(self.function): 

106 raise ValueError( 

107 f"Async functions are not supported as tools. " 

108 f"The function '{self.function.__name__}' is a coroutine function. " 

109 f"Please use a synchronous function instead." 

110 ) 

111 

112 # Check that the parameters define a valid JSON schema 

113 try: 

114 Draft202012Validator.check_schema(self.parameters) 

115 except SchemaError as e: 

116 raise ValueError("The provided parameters do not define a valid JSON schema") from e 

117 

118 # Validate outputs structure if provided 

119 if self.outputs_to_state is not None: 

120 for key, config in self.outputs_to_state.items(): 

121 if not isinstance(config, dict): 

122 raise TypeError(f"outputs_to_state configuration for key '{key}' must be a dictionary") 

123 if "source" in config and not isinstance(config["source"], str): 

124 raise ValueError(f"outputs_to_state source for key '{key}' must be a string.") 

125 if "handler" in config and not callable(config["handler"]): 

126 raise ValueError(f"outputs_to_state handler for key '{key}' must be callable") 

127 

128 # Validate that outputs_to_state source keys exist as valid tool outputs 

129 valid_outputs: set[str] | None = self._get_valid_outputs() 

130 if valid_outputs is not None: 

131 for state_key, config in self.outputs_to_state.items(): 

132 source = config.get("source") 

133 if source is not None and source not in valid_outputs: 

134 raise ValueError( 

135 f"outputs_to_state: '{self.name}' maps state key '{state_key}' to unknown output '{source}'" 

136 f"Valid outputs are: {valid_outputs}." 

137 ) 

138 

139 if self.outputs_to_string is not None: 

140 if "source" in self.outputs_to_string and not isinstance(self.outputs_to_string["source"], str): 

141 raise ValueError("outputs_to_string source must be a string.") 

142 if "handler" in self.outputs_to_string and not callable(self.outputs_to_string["handler"]): 

143 raise ValueError("outputs_to_string handler must be callable") 

144 if "raw_result" in self.outputs_to_string and not isinstance(self.outputs_to_string["raw_result"], bool): 

145 raise ValueError("outputs_to_string raw_result must be a boolean.") 

146 

147 if ( 

148 "source" in self.outputs_to_string 

149 or "handler" in self.outputs_to_string 

150 or "raw_result" in self.outputs_to_string 

151 ): 

152 # Single output configuration 

153 for key in self.outputs_to_string: 

154 if key not in {"source", "handler", "raw_result"}: 

155 raise ValueError( 

156 "Invalid outputs_to_string config. " 

157 "When using 'source', 'handler' or 'raw_result' at the root level, no other keys are " 

158 " allowed. Use individual output configs instead." 

159 ) 

160 else: 

161 # Multiple outputs configuration 

162 for key, config in self.outputs_to_string.items(): 

163 if not isinstance(config, dict): 

164 raise TypeError(f"outputs_to_string configuration for key '{key}' must be a dictionary") 

165 if "raw_result" in config: 

166 raise ValueError( 

167 f"Invalid outputs_to_string configuration for key '{key}': " 

168 f"'raw_result' is not supported in the multiple output format." 

169 ) 

170 if "source" not in config: 

171 raise ValueError( 

172 f"Invalid outputs_to_string configuration for key '{key}': " 

173 f"each output must have a 'source' defined." 

174 ) 

175 if "source" in config and not isinstance(config["source"], str): 

176 raise ValueError(f"outputs_to_string source for key '{key}' must be a string.") 

177 if "handler" in config and not callable(config["handler"]): 

178 raise ValueError(f"outputs_to_string handler for key '{key}' must be callable") 

179 

180 # Validate that inputs_from_state parameter names exist as valid tool parameters 

181 if self.inputs_from_state is not None: 

182 valid_inputs = self._get_valid_inputs() 

183 for state_key, param_name in self.inputs_from_state.items(): 

184 if not isinstance(param_name, str): 

185 raise TypeError( 

186 f"inputs_from_state values must be str, not {type(param_name).__name__}. " 

187 f"Got {param_name!r} for key '{state_key}'." 

188 ) 

189 if valid_inputs and param_name not in valid_inputs: 

190 raise ValueError( 

191 f"inputs_from_state maps '{state_key}' to unknown parameter '{param_name}'. " 

192 f"Valid parameters are: {valid_inputs}." 

193 )

195 def _get_valid_inputs(self) -> set[str]: 

196 """ 

197 Return the set of valid input parameter names that this tool accepts. 

198 

199 Used to validate that `inputs_from_state` only references parameters that actually exist. 

200 This prevents typos and catches configuration errors at tool construction time. 

201 

202 By default, introspects the function signature to get ALL parameters, including those 

203 that may be excluded from the JSON schema (e.g., parameters mapped from state). 

204 Falls back to schema properties if introspection fails. 

205 

206 Subclasses like ComponentTool override this to return component input socket names. 

207 

208 :returns: Set of valid input parameter names for validation. 

209 """ 

210 # Combine parameters from both function signature and schema for robustness 

211 # Function signature includes all parameters (even those excluded from schema) 

212 # Schema properties provide the validated parameter set 

213 valid_params: set[str] = set() 

214 

215 # Try to get parameters from function introspection 

216 try: 

217 sig = inspect.signature(self.function) 

218 valid_params.update(sig.parameters.keys()) 

219 except (ValueError, TypeError): 

220 pass # Introspection failed, will rely on schema 

221 

222 # Add parameters from schema (union with function params) 

223 valid_params.update(self.parameters.get("properties", {}).keys()) 

224 

225 return valid_params

227 def _get_valid_outputs(self) -> set[str] | None: 

228 """ 

229 Return the set of valid output names that this tool produces. 

230 

231 Used to validate that `outputs_to_state` only references outputs that actually exist. 

232 This prevents typos and catches configuration errors at tool construction time. 

233 

234 By default, returns None because regular function-based tools don't have a formal 

235 output schema. When None is returned, output validation is skipped. 

236 

237 Subclasses like ComponentTool override this to return component output socket names, 

238 enabling validation for tools where outputs are known. 

239 

240 :returns: Set of valid output names for validation, or None to skip validation. 

241 """ 

242 return None

261 def invoke(self, **kwargs: Any) -> Any: 

262 """ 

263 Invoke the Tool with the provided keyword arguments. 

264 """ 

265 try: 

266 result = self.function(**kwargs) 

267 except Exception as e: 

268 raise ToolInvocationError( 

269 f"Failed to invoke Tool `{self.name}` with parameters {kwargs}. Error: {e}", tool_name=self.name

271 return result

273 def to_dict(self) -> dict[str, Any]: 

274 """ 

275 Serializes the Tool to a dictionary. 

276 

277 :returns: 

278 Dictionary with serialized data. 

279 """ 

280 data = asdict(self) 

281 data["function"] = serialize_callable(self.function) 

282 

283 if self.outputs_to_state is not None: 

284 data["outputs_to_state"] = _serialize_outputs_to_state(self.outputs_to_state) 

285 

286 if self.outputs_to_string is not None: 

287 data["outputs_to_string"] = _serialize_outputs_to_string(self.outputs_to_string) 

288 

289 return {"type": generate_qualified_class_name(type(self)), "data": data}

292 def from_dict(cls, data: dict[str, Any]) -> "Tool": 

293 """ 

294 Deserializes the Tool from a dictionary. 

295 

296 :param data: 

297 Dictionary to deserialize from. 

298 :returns: 

299 Deserialized Tool. 

300 """ 

301 init_parameters = data["data"] 

302 init_parameters["function"] = deserialize_callable(init_parameters["function"]) 

303 if "outputs_to_state" in init_parameters and init_parameters["outputs_to_state"]: 

304 init_parameters["outputs_to_state"] = _deserialize_outputs_to_state(init_parameters["outputs_to_state"]) 

305 

306 if init_parameters.get("outputs_to_string") is not None: 

307 init_parameters["outputs_to_string"] = _deserialize_outputs_to_string(init_parameters["outputs_to_string"]) 

308 

309 return cls(**init_parameters)

312def _check_duplicate_tool_names(tools: list[Tool] | None) -> None: 

313 """ 

314 Checks for duplicate tool names and raises a ValueError if they are found. 

315 

316 :param tools: The list of tools to check. 

317 :raises ValueError: If duplicate tool names are found. 

318 """ 

319 if tools is None: 

320 return 

321 tool_names = [tool.name for tool in tools] 

322 duplicate_tool_names = {name for name in tool_names if tool_names.count(name) > 1} 

323 if duplicate_tool_names: 

324 raise ValueError(f"Duplicate tool names found: {duplicate_tool_names}")

327def _serialize_outputs_to_state(outputs_to_state: dict[str, dict[str, Any]]) -> dict[str, dict[str, Any]]: 

328 """ 

329 Serializes the outputs_to_state dictionary, converting any callable handlers to their string representation. 

330 

331 :param outputs_to_state: The outputs_to_state dictionary to serialize. 

332 :returns: The serialized outputs_to_state dictionary. 

333 """ 

334 serialized_outputs = {} 

335 for key, config in outputs_to_state.items(): 

336 serialized_config = config.copy() 

337 if "handler" in config: 

338 serialized_config["handler"] = serialize_callable(config["handler"]) 

339 serialized_outputs[key] = serialized_config 

340 return serialized_outputs
