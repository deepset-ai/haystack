22@component 

23class DocumentToImageContent: 

24 """ 

25 Converts documents sourced from PDF and image files into ImageContents. 

26 

27 This component processes a list of documents and extracts visual content from supported file formats, converting 

28 them into ImageContents that can be used for multimodal AI tasks. It handles both direct image files and PDF 

29 documents by extracting specific pages as images. 

30 

31 Documents are expected to have metadata containing: 

32 - The `file_path_meta_field` key with a valid file path that exists when combined with `root_path` 

33 - A supported image format (MIME type must be one of the supported image types) 

34 - For PDF files, a `page_number` key specifying which page to extract 

35 

36 ### Usage example 

37 <!-- test-ignore --> 

38 ```python 

39 from haystack import Document 

40 from haystack.components.converters.image.document_to_image import DocumentToImageContent 

41 

42 converter = DocumentToImageContent( 

43 file_path_meta_field="file_path", 

44 root_path="/data/files", 

45 detail="high", 

46 size=(800, 600) 

47 ) 

48 

49 documents = [ 

50 Document(content="Optional description of image.jpg", meta={"file_path": "image.jpg"}), 

51 Document(content="Text content of page 1 of doc.pdf", meta={"file_path": "doc.pdf", "page_number": 1}) 

52 ] 

53 

54 result = converter.run(documents) 

55 image_contents = result["image_contents"] 

56 # [ImageContent( 

57 # base64_image='/9j/4A...', mime_type='image/jpeg', detail='high', meta={'file_path': 'image.jpg'} 

58 # ), 

59 # ImageContent( 

60 # base64_image='/9j/4A...', mime_type='image/jpeg', detail='high', 

61 # meta={'page_number': 1, 'file_path': 'doc.pdf'} 

62 # )] 

63 ``` 

64 """ 

65 

66 def __init__( 

67 self, 

68 *, 

69 file_path_meta_field: str = "file_path", 

70 root_path: str | None = None, 

71 detail: Literal["auto", "high", "low"] | None = None, 

72 size: tuple[int, int] | None = None, 

73 ) -> None: 

74 """ 

75 Initialize the DocumentToImageContent component. 

76 

77 :param file_path_meta_field: The metadata field in the Document that contains the file path to the image or PDF. 

78 :param root_path: The root directory path where document files are located. If provided, file paths in 

79 document metadata will be resolved relative to this path. If None, file paths are treated as absolute paths. 

80 :param detail: Optional detail level of the image (only supported by OpenAI). Can be "auto", "high", or "low". 

81 This will be passed to the created ImageContent objects. 

82 :param size: If provided, resizes the image to fit within the specified dimensions (width, height) while 

83 maintaining aspect ratio. This reduces file size, memory usage, and processing time, which is beneficial 

84 when working with models that have resolution constraints or when transmitting images to remote services. 

85 """ 

86 pillow_import.check() 

87 pypdfium2_import.check() 

88 

89 self.file_path_meta_field = file_path_meta_field 

90 self.root_path = root_path or "" 

91 self.detail = detail 

92 self.size = size 

93 

94 @component.output_types(image_contents=list[ImageContent | None]) 

95 def run(self, documents: list[Document]) -> dict[str, list[ImageContent | None]]: 

96 """ 

97 Convert documents with image or PDF sources into ImageContent objects. 

98 

99 This method processes the input documents, extracting images from supported file formats and converting them 

100 into ImageContent objects. 

101 

102 :param documents: A list of documents to process. Each document should have metadata containing at minimum 

103 a 'file_path_meta_field' key. PDF documents additionally require a 'page_number' key to specify which 

104 page to convert. 

105 

106 :returns: 

107 Dictionary containing one key: 

108 - "image_contents": ImageContents created from the processed documents. These contain base64-encoded image 

109 data and metadata. The order corresponds to order of input documents. 

110 :raises ValueError: 

111 If any document is missing the required metadata keys, has an invalid file path, or has an unsupported 

112 MIME type. The error message will specify which document and what information is missing or incorrect. 

113 """ 

114 if not documents: 

115 return {"image_contents": []} 

116 

117 images_source_info = _extract_image_sources_info( 

118 documents=documents, file_path_meta_field=self.file_path_meta_field, root_path=self.root_path 

119 ) 

120 

121 image_contents: list[ImageContent | None] = [None] * len(documents) 

122 

123 pdf_page_infos: list[_PDFPageInfo] = [] 

124 

125 for doc_idx, image_source_info in enumerate(images_source_info): 

126 mime_type = image_source_info["mime_type"] 

127 path = image_source_info["path"] 

128 if mime_type == "application/pdf": 

129 # Store PDF documents for later processing 

130 page_number = image_source_info.get("page_number") 

131 assert page_number is not None # checked in _extract_image_sources_info but mypy doesn't know that 

132 pdf_page_info: _PDFPageInfo = {"doc_idx": doc_idx, "path": path, "page_number": page_number} 

133 pdf_page_infos.append(pdf_page_info) 

134 else: 

135 # Process images directly 

136 bytestream = ByteStream.from_file_path(filepath=path, mime_type=mime_type) 

137 _, base64_image = _encode_image_to_base64(bytestream=bytestream, size=self.size) 

138 image_contents[doc_idx] = ImageContent( 

139 base64_image=base64_image, 

140 mime_type=mime_type, 

141 detail=self.detail, 

142 meta={"file_path": documents[doc_idx].meta[self.file_path_meta_field]}, 

143 ) 

144 

145 # efficiently convert PDF pages to images: each PDF is opened and processed only once 

146 pdf_page_infos_by_doc_idx: dict[int, _PDFPageInfo] = { 

147 pdf_page_info["doc_idx"]: pdf_page_info for pdf_page_info in pdf_page_infos 

148 } 

149 pdf_images_by_doc_idx = _batch_convert_pdf_pages_to_images( 

150 pdf_page_infos=pdf_page_infos, size=self.size, return_base64=True 

151 ) 

152 for doc_idx, base64_pdf_image in pdf_images_by_doc_idx.items(): 

153 meta = { 

154 "file_path": documents[doc_idx].meta[self.file_path_meta_field], 

155 "page_number": pdf_page_infos_by_doc_idx[doc_idx]["page_number"], 

156 } 

157 # we know that base64_pdf_image is a string because we set return_base64=True but mypy doesn't know that 

158 assert isinstance(base64_pdf_image, str) 

159 image_contents[doc_idx] = ImageContent( 

160 base64_image=base64_pdf_image, mime_type="image/jpeg", detail=self.detail, meta=meta 

161 ) 

162 

163 none_image_contents_doc_ids = [ 

164 documents[doc_idx].id for doc_idx, image_content in enumerate(image_contents) if image_content is None 

165 ] 

166 if none_image_contents_doc_ids: 

167 logger.warning( 

168 "Conversion failed for some documents. Their output will be None. Document IDs: {document_ids}", 

169 document_ids=none_image_contents_doc_ids, 

170 ) 

171 

172 return {"image_contents": image_contents}
