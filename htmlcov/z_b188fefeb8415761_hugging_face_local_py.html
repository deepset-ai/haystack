24@component 

25class HuggingFaceLocalGenerator: 

26 """ 

27 Generates text using models from Hugging Face that run locally. 

28 

29 LLMs running locally may need powerful hardware. 

30 

31 ### Usage example 

32 

33 ```python 

34 from haystack.components.generators import HuggingFaceLocalGenerator 

35 

36 generator = HuggingFaceLocalGenerator( 

37 model="Qwen/Qwen3-0.6B", 

38 task="text-generation", 

39 generation_kwargs={"max_new_tokens": 100, "temperature": 0.9} 

40 ) 

41 

42 print(generator.run("Who is the best American actor?")) 

43 # >> {'replies': ['John Cusack']} 

44 ``` 

45 """ 

46 

47 def __init__( 

48 self, 

49 model: str = "Qwen/Qwen3-0.6B", 

50 task: Literal["text-generation", "text2text-generation"] | None = None, 

51 device: ComponentDevice | None = None, 

52 token: Secret | None = Secret.from_env_var(["HF_API_TOKEN", "HF_TOKEN"], strict=False), 

53 generation_kwargs: dict[str, Any] | None = None, 

54 huggingface_pipeline_kwargs: dict[str, Any] | None = None, 

55 stop_words: list[str] | None = None, 

56 streaming_callback: StreamingCallbackT | None = None, 

57 ) -> None: 

58 """ 

59 Creates an instance of a HuggingFaceLocalGenerator. 

60 

61 :param model: The Hugging Face text generation model name or path. 

62 :param task: The task for the Hugging Face pipeline. Possible options: 

63 - `text-generation`: Supported by decoder models, like GPT. 

64 - `text2text-generation`: Deprecated as of Transformers v5; use `text-generation` instead. 

65 Previously supported by encoder–decoder models such as T5. 

66 If the task is specified in `huggingface_pipeline_kwargs`, this parameter is ignored. 

67 If not specified, the component calls the Hugging Face API to infer the task from the model name. 

68 :param device: The device for loading the model. If `None`, automatically selects the default device. 

69 If a device or device map is specified in `huggingface_pipeline_kwargs`, it overrides this parameter. 

70 :param token: The token to use as HTTP bearer authorization for remote files. 

71 If the token is specified in `huggingface_pipeline_kwargs`, this parameter is ignored. 

72 :param generation_kwargs: A dictionary with keyword arguments to customize text generation. 

73 Some examples: `max_length`, `max_new_tokens`, `temperature`, `top_k`, `top_p`. 

74 See Hugging Face's documentation for more information: 

75 - [customize-text-generation](https://huggingface.co/docs/transformers/main/en/generation_strategies#customize-text-generation) 

76 - [transformers.GenerationConfig](https://huggingface.co/docs/transformers/main/en/main_classes/text_generation#transformers.GenerationConfig) 

77 :param huggingface_pipeline_kwargs: Dictionary with keyword arguments to initialize the 

78 Hugging Face pipeline for text generation. 

79 These keyword arguments provide fine-grained control over the Hugging Face pipeline. 

80 In case of duplication, these kwargs override `model`, `task`, `device`, and `token` init parameters. 

81 For available kwargs, see [Hugging Face documentation](https://huggingface.co/docs/transformers/en/main_classes/pipelines#transformers.pipeline.task). 

82 In this dictionary, you can also include `model_kwargs` to specify the kwargs for model initialization: 

83 [transformers.PreTrainedModel.from_pretrained](https://huggingface.co/docs/transformers/en/main_classes/model#transformers.PreTrainedModel.from_pretrained) 

84 :param stop_words: If the model generates a stop word, the generation stops. 

85 If you provide this parameter, don't specify the `stopping_criteria` in `generation_kwargs`. 

86 For some chat models, the output includes both the new text and the original prompt. 

87 In these cases, make sure your prompt has no stop words. 

88 :param streaming_callback: An optional callable for handling streaming responses. 

89 """ 

90 transformers_import.check() 

91 

92 self.token = token 

93 generation_kwargs = generation_kwargs or {} 

94 

95 huggingface_pipeline_kwargs = resolve_hf_pipeline_kwargs( 

96 huggingface_pipeline_kwargs=huggingface_pipeline_kwargs or {}, 

97 model=model, 

98 task=task, 

99 supported_tasks=SUPPORTED_TASKS, 

100 device=device, 

101 token=token, 

102 ) 

103 

104 # if not specified, set return_full_text to False for text-generation 

105 # only generated text is returned (excluding prompt) 

106 task = huggingface_pipeline_kwargs["task"] 

107 if task == "text-generation": 

108 generation_kwargs.setdefault("return_full_text", False) 

109 

110 if stop_words and "stopping_criteria" in generation_kwargs: 

111 raise ValueError( 

112 "Found both the `stop_words` init parameter and the `stopping_criteria` key in `generation_kwargs`. " 

113 "Please specify only one of them." 

114 ) 

115 generation_kwargs.setdefault("max_new_tokens", 512) 

116 

117 self.huggingface_pipeline_kwargs = huggingface_pipeline_kwargs 

118 self.generation_kwargs = generation_kwargs 

119 self.stop_words = stop_words 

120 self.pipeline: HfPipeline | None = None 

121 self.stopping_criteria_list: StoppingCriteriaList | None = None 

122 self.streaming_callback = streaming_callback

124 def _get_telemetry_data(self) -> dict[str, Any]: 

125 """ 

126 Data that is sent to Posthog for usage analytics. 

127 """ 

128 if isinstance(self.huggingface_pipeline_kwargs["model"], str): 

129 return {"model": self.huggingface_pipeline_kwargs["model"]} 

130 return {"model": f"[object of type {type(self.huggingface_pipeline_kwargs['model'])}]"}

132 @property 

133 def _warmed_up(self) -> bool: 

134 if self.stop_words: 

135 return (self.pipeline is not None) and (self.stopping_criteria_list is not None) 

136 return self.pipeline is not None

138 def warm_up(self) -> None: 

139 """ 

140 Initializes the component. 

141 """ 

142 if self._warmed_up: 

143 return 

144 

145 if self.pipeline is None: 

146 self.pipeline = pipeline(**self.huggingface_pipeline_kwargs) 

147 

148 if self.stop_words: 

149 # text-generation and text2text-generation pipelines always have a non-None tokenizer 

150 assert self.pipeline.tokenizer is not None 

151 

152 stop_words_criteria = StopWordsCriteria( 

153 tokenizer=self.pipeline.tokenizer, stop_words=self.stop_words, device=self.pipeline.device 

154 ) 

155 self.stopping_criteria_list = StoppingCriteriaList([stop_words_criteria])

199 @component.output_types(replies=list[str]) 

200 def run( 

201 self, 

202 prompt: str, 

203 streaming_callback: StreamingCallbackT | None = None, 

204 generation_kwargs: dict[str, Any] | None = None, 

205 ) -> dict[str, Any]: 

206 """ 

207 Run the text generation model on the given prompt. 

208 

209 :param prompt: 

210 A string representing the prompt. 

211 :param streaming_callback: 

212 A callback function that is called when a new token is received from the stream. 

213 :param generation_kwargs: 

214 Additional keyword arguments for text generation. 

215 

216 :returns: 

217 A dictionary containing the generated replies. 

218 - replies: A list of strings representing the generated replies. 

219 """ 

220 if not self._warmed_up: 

221 self.warm_up() 

222 

223 # at this point, we know that the pipeline has been initialized 

224 assert self.pipeline is not None 

225 # text-generation and text2text-generation pipelines always have a non-None tokenizer 

226 assert self.pipeline.tokenizer is not None 

227 

228 if not prompt: 

229 return {"replies": []} 

230 

231 # merge generation kwargs from init method with those from run method 

232 updated_generation_kwargs = {**self.generation_kwargs, **(generation_kwargs or {})} 

233 

234 # check if streaming_callback is passed 

235 streaming_callback = select_streaming_callback( 

236 init_callback=self.streaming_callback, runtime_callback=streaming_callback, requires_async=False 

237 ) 

238 

239 if streaming_callback: 

240 num_responses = updated_generation_kwargs.get("num_return_sequences", 1) 

241 if num_responses > 1: 

242 msg = ( 

243 "Streaming is enabled, but the number of responses is set to {num_responses}. " 

244 "Streaming is only supported for single response generation. " 

245 "Setting the number of responses to 1." 

246 ) 

247 logger.warning(msg, num_responses=num_responses) 

248 updated_generation_kwargs["num_return_sequences"] = 1 

249 

250 # streamer parameter hooks into HF streaming, HFTokenStreamingHandler is an adapter to our streaming 

251 updated_generation_kwargs["streamer"] = HFTokenStreamingHandler( 

252 tokenizer=self.pipeline.tokenizer, 

253 stream_handler=streaming_callback, 

254 stop_words=self.stop_words, 

255 component_info=ComponentInfo.from_component(self), 

256 ) 

257 

258 output = self.pipeline(prompt, stopping_criteria=self.stopping_criteria_list, **updated_generation_kwargs) 

259 replies = [o["generated_text"] for o in output if "generated_text" in o] 

260 

261 if self.stop_words: 

262 # the output of the pipeline includes the stop word 

263 replies = [reply.replace(stop_word, "").rstrip() for reply in replies for stop_word in self.stop_words] 

264 

265 return {"replies": replies}
