44class PipelineStreamHandle: 

45 """ 

46 Handle returned by `Pipeline.stream()`. 

47 

48 Async-iterable over `StreamingChunk`s produced by streaming components in the pipeline. After iteration ends, 

49 `result` holds the final pipeline output dict. 

50 

51 By default, iteration cleans up automatically: if the consumer abandons iteration, the underlying pipeline task is 

52 cancelled. `aclose()` is also available for explicit cleanup. 

53 """ 

54 

55 _END_OF_STREAM: ClassVar[_EndOfStream] = _EndOfStream() 

56 _CLEANUP_TIMEOUT_SECONDS: ClassVar[float] = 1.0 

57 

58 def __init__( 

59 self, 

60 queue: asyncio.Queue["StreamingChunk | _EndOfStream"], 

61 task: asyncio.Task[dict[str, Any]], 

62 cancel_on_abandon: bool = True, 

63 ) -> None: 

64 self._queue = queue 

65 self._task = task 

66 self._cancel_on_abandon = cancel_on_abandon 

67 

68 async def __aiter__(self) -> AsyncIterator[StreamingChunk]: 

69 """ 

70 Drain the queue and cancel the pipeline task if iteration is abandoned. 

71 

72 `__aiter__` is an async generator function: each `async for` call gets a generator and `try/finally` runs on 

73 exit. When `cancel_on_abandon` is True (default), abandoned iteration cancels the pipeline task; when False, 

74 the task is left running to completion. 

75 """ 

76 try: 

77 while True: 

78 item = await self._queue.get() 

79 if item is self._END_OF_STREAM: 

80 await self._task # called to make exceptions surface 

81 return 

82 yield cast(StreamingChunk, item) # at this point, item is guaranteed to be a StreamingChunk 

83 

84 finally: 

85 if self._cancel_on_abandon: 

86 await self.aclose() 

87 

88 @property 

89 def result(self) -> dict[str, Any]: 

90 """ 

91 Final pipeline output dict, available only after a successful, complete run. 

92 

93 Raises a `RuntimeError` if the pipeline has not finished or was cancelled. If the pipeline failed, re-raises the 

94 original exception. 

95 """ 

96 if not self._task.done(): 

97 raise RuntimeError("Pipeline has not finished; iterate the handle first.") 

98 if self._task.cancelled(): 

99 raise RuntimeError("Pipeline was cancelled; no result available.") 

100 exc = self._task.exception() 

101 if exc is not None: 

102 raise exc 

103 return self._task.result() 

104 

105 async def aclose(self) -> None: 

106 """ 

107 Cancel the underlying pipeline task. 

108 

109 Bounded by `_CLEANUP_TIMEOUT_SECONDS` so that components cannot block cleanup indefinitely. 

110 """ 

111 if not self._task.done(): 

112 self._task.cancel() 

113 with contextlib.suppress(BaseException): 

114 await asyncio.wait_for(self._task, timeout=self._CLEANUP_TIMEOUT_SECONDS)

118 """ 

119 Orchestration engine that runs components according to the execution graph.

121 Supports both a synchronous run path (`run`) and an asynchronous run path 

122 (`run_async`, `run_async_generator`, `stream`).

125 @staticmethod 

126 def _run_component( 

127 component_name: str, 

128 component: dict[str, Any], 

129 inputs: dict[str, Any], 

130 component_visits: dict[str, int], 

131 parent_span: tracing.Span | None = None, 

132 *, 

133 break_point: Breakpoint | None = None, 

134 ) -> Mapping[str, Any]: 

135 """ 

136 Runs a Component with the given inputs. 

137 

138 :param component_name: Name of the Component. 

139 :param component: Component with component metadata. 

140 :param inputs: Inputs for the Component. 

141 :param component_visits: Current state of component visits. 

142 :param parent_span: The parent span to use for the newly created span. 

143 This is to allow tracing to be correctly linked to the pipeline run. 

144 :raises PipelineRuntimeError: If Component doesn't return a dictionary. 

145 :return: The output of the Component. 

146 """ 

147 if ( 

148 isinstance(break_point, Breakpoint) 

149 and break_point.component_name == component_name 

150 and break_point.visit_count == component_visits[component_name] 

151 ): 

152 raise BreakpointException.from_triggered_breakpoint(break_point=break_point) 

153 

154 instance: Component = component["instance"] 

155 

156 with PipelineBase._create_component_span( 

157 component_name=component_name, instance=instance, inputs=inputs, parent_span=parent_span 

158 ) as span: 

159 # deepcopy inputs before passing to the tracer so that even if a tracer mutates them 

160 # the component always receives the original unmodified values 

161 inputs_copy = _deepcopy_with_exceptions(inputs) 

162 span.set_content_tag(_COMPONENT_INPUT, inputs) 

163 logger.info("Running component {component_name}", component_name=component_name) 

164 

165 try: 

166 component_output = instance.run(**inputs_copy) 

167 except BreakpointException as error: 

168 # Re-raise BreakpointException to preserve the original exception context 

169 # This is important when Agent components internally use Pipeline._run_component 

170 # and trigger breakpoints that need to bubble up to the main pipeline 

171 raise error 

172 

173 # Any components that internally use Pipeline._run_component could raise a PipelineRuntimeError with 

174 # additional context (e.g. Agent raises an agent snapshot) so we re-raise here instead of wrapping it in 

175 # another PipelineRuntimeError 

176 

177 except PipelineRuntimeError as runtime_error: 

178 raise runtime_error 

179 

180 # Catch all other exceptions and wrap them in a PipelineRuntimeError 

181 except Exception as error: 

182 raise PipelineRuntimeError.from_exception(component_name, instance.__class__, error) from error 

183 

184 component_visits[component_name] += 1 

185 

186 if not isinstance(component_output, Mapping): 

187 raise PipelineRuntimeError.from_invalid_output(component_name, instance.__class__, component_output) 

188 

189 _validate_component_output_keys(component_name, component, component_output) 

190 

191 span.set_tag(_COMPONENT_VISITS, component_visits[component_name]) 

192 span.set_content_tag(_COMPONENT_OUTPUT, component_output) 

193 

194 return component_output

196 def run( # noqa: PLR0915, PLR0912, C901 

197 self, 

198 data: dict[str, Any], 

199 include_outputs_from: set[str] | None = None, 

200 *, 

201 break_point: Breakpoint | None = None, 

202 pipeline_snapshot: PipelineSnapshot | None = None, 

203 snapshot_callback: SnapshotCallback | None = None, 

204 ) -> dict[str, Any]: 

205 """ 

206 Runs the Pipeline with given input data. 

207 

208 `run` executes synchronously and blocks the calling thread until the run completes. In an async context, 

209 use `run_async` instead. 

210 

211 Usage: 

212 ```python 

213 from haystack import Pipeline, Document 

214 from haystack.components.builders.answer_builder import AnswerBuilder 

215 from haystack.components.builders.chat_prompt_builder import ChatPromptBuilder 

216 from haystack.components.generators.chat import OpenAIChatGenerator 

217 from haystack.components.retrievers.in_memory import InMemoryBM25Retriever 

218 from haystack.dataclasses import ChatMessage 

219 from haystack.document_stores.in_memory import InMemoryDocumentStore 

220 from haystack.utils import Secret 

221 

222 # Write documents to InMemoryDocumentStore 

223 document_store = InMemoryDocumentStore() 

224 document_store.write_documents([ 

225 Document(content="My name is Jean and I live in Paris."), 

226 Document(content="My name is Mark and I live in Berlin."), 

227 Document(content="My name is Giorgio and I live in Rome.") 

228 ]) 

229 

230 retriever = InMemoryBM25Retriever(document_store=document_store) 

231 

232 prompt_template = \"\"\" 

233 Given these documents, answer the question. 

234 Documents: 

235 {% for doc in documents %} 

236 {{ doc.content }} 

237 {% endfor %} 

238 Question: {{question}} 

239 Answer: 

240 \"\"\" 

241 

242 template = [ChatMessage.from_user(prompt_template)] 

243 prompt_builder = ChatPromptBuilder( 

244 template=template, 

245 required_variables=["question", "documents"], 

246 variables=["question", "documents"] 

247 ) 

248 

249 llm = OpenAIChatGenerator() 

250 rag_pipeline = Pipeline() 

251 rag_pipeline.add_component("retriever", retriever) 

252 rag_pipeline.add_component("prompt_builder", prompt_builder) 

253 rag_pipeline.add_component("llm", llm) 

254 rag_pipeline.connect("retriever", "prompt_builder.documents") 

255 rag_pipeline.connect("prompt_builder", "llm") 

256 

257 question = "Who lives in Paris?" 

258 results = rag_pipeline.run( 

259 { 

260 "retriever": {"query": question}, 

261 "prompt_builder": {"question": question}, 

262 } 

263 ) 

264 

265 print(results["llm"]["replies"][0].text) 

266 # Jean lives in Paris 

267 ``` 

268 

269 :param data: 

270 A dictionary of inputs for the pipeline's components. Each key is a component name 

271 and its value is a dictionary of that component's input parameters: 

272 ``` 

273 data = { 

274 "comp1": {"input1": 1, "input2": 2}, 

275 } 

276 ``` 

277 For convenience, this format is also supported when input names are unique: 

278 ``` 

279 data = { 

280 "input1": 1, "input2": 2, 

281 } 

282 ``` 

283 :param include_outputs_from: 

284 Set of component names whose individual outputs are to be 

285 included in the pipeline's output. For components that are 

286 invoked multiple times (in a loop), only the last-produced 

287 output is included. 

288 

289 :param break_point: 

290 A breakpoint that pauses execution before the specified component runs by raising a 

291 `BreakpointException` carrying a `PipelineSnapshot` of the current pipeline state. 

292 

293 :param pipeline_snapshot: 

294 A snapshot of a previously interrupted pipeline execution to resume from. Can be combined with 

295 `break_point` to step through a pipeline: resume from the snapshot and pause again at the next 

296 breakpoint. The `break_point` must target a different component or visit count than the one the 

297 snapshot was created at, otherwise it would trigger again before any progress is made. 

298 

299 :param snapshot_callback: 

300 Optional callback function that is invoked when a pipeline snapshot is created. 

301 The callback receives a `PipelineSnapshot` object and can return an optional string 

302 (e.g., a file path or identifier). 

303 If provided, the callback is used instead of the default file-saving behavior, 

304 allowing custom handling of snapshots (e.g., saving to a database, sending to a remote service). 

305 If not provided, the default behavior saves snapshots to a JSON file. 

306 

307 :returns: 

308 A dictionary where each entry corresponds to a component name 

309 and its output. If `include_outputs_from` is `None`, this dictionary 

310 will only contain the outputs of leaf components, i.e., components 

311 without outgoing connections. 

312 

313 :raises ValueError: 

314 If invalid inputs are provided to the pipeline. 

315 :raises PipelineRuntimeError: 

316 If the Pipeline contains cycles with unsupported connections that would cause 

317 it to get stuck and fail running. 

318 Or if a Component fails or returns output in an unsupported type. 

319 :raises PipelineMaxComponentRuns: 

320 If a Component reaches the maximum number of times it can be run in this Pipeline. 

321 :raises PipelineBreakpointException: 

322 When a pipeline_breakpoint is triggered. Contains the component name, state, and partial results. 

323 """ 

324 pipeline_running(self) # telemetry 

325 

326 if ( 

327 break_point 

328 and pipeline_snapshot 

329 and break_point.component_name == pipeline_snapshot.break_point.component_name 

330 and break_point.visit_count == pipeline_snapshot.break_point.visit_count 

331 ): 

332 msg = ( 

333 "The provided break_point targets the same component and visit count as the break_point of the " 

334 "pipeline_snapshot. It would trigger again before the resumed component runs, so the pipeline " 

335 "could not make any progress. Provide a break_point with a different component or visit count." 

336 ) 

337 raise PipelineInvalidPipelineSnapshotError(message=msg) 

338 

339 # make sure all breakpoints are valid, i.e. reference components in the pipeline 

340 if break_point: 

341 _validate_break_point_against_pipeline(break_point, self.graph) 

342 

343 # warm up the pipeline by running each component's warm_up method 

344 self.warm_up() 

345 

346 if include_outputs_from is None: 

347 include_outputs_from = set() 

348 

349 pipeline_outputs: dict[str, Any] = {} 

350 

351 if not pipeline_snapshot: 

352 # normalize `data` 

353 data = self._prepare_component_input_data(data) 

354 

355 # Raise ValueError if input is malformed in some way 

356 self.validate_input(data) 

357 

358 # We create a list of components in the pipeline sorted by name, so that the algorithm runs 

359 # deterministically and independent of insertion order into the pipeline. 

360 ordered_component_names = sorted(self.graph.nodes.keys()) 

361 

362 # We track component visits to decide if a component can run. 

363 component_visits = dict.fromkeys(ordered_component_names, 0) 

364 

365 else: 

366 # Validate the pipeline snapshot against the current pipeline graph 

367 _validate_pipeline_snapshot_against_pipeline(pipeline_snapshot, self.graph) 

368 

369 # Handle resuming the pipeline from a snapshot 

370 component_visits = pipeline_snapshot.pipeline_state.component_visits 

371 ordered_component_names = pipeline_snapshot.ordered_component_names 

372 data = _deserialize_value_with_schema(pipeline_snapshot.pipeline_state.inputs) 

373 

374 # include_outputs_from from the snapshot when resuming 

375 include_outputs_from = pipeline_snapshot.include_outputs_from 

376 

377 # also intermediate_outputs from the snapshot when resuming 

378 pipeline_outputs = _deserialize_value_with_schema(pipeline_snapshot.pipeline_state.pipeline_outputs) 

379 

380 cached_topological_sort = None 

381 # We need to access a component's receivers multiple times during a pipeline run. 

382 # We store them here for easy access. 

383 cached_receivers = {name: self._find_receivers_from(name) for name in ordered_component_names} 

384 

385 with tracing.tracer.trace( 

386 "haystack.pipeline.run", 

387 tags={ 

388 "haystack.pipeline.input_data": data, 

389 "haystack.pipeline.output_data": pipeline_outputs, 

390 "haystack.pipeline.metadata": self.metadata, 

391 "haystack.pipeline.max_runs_per_component": self._max_runs_per_component, 

392 "haystack.pipeline.execution_mode": "sync", 

393 }, 

394 ) as span: 

395 inputs = self._convert_to_internal_format(pipeline_inputs=data) 

396 priority_queue = self._fill_queue(ordered_component_names, inputs, component_visits) 

397 

398 # check if pipeline is blocked before execution 

399 self.validate_pipeline(priority_queue) 

400 

401 while True: 

402 candidate = self._get_next_runnable_component(priority_queue, component_visits) 

403 

404 # If there are no runnable components left, we can exit the loop. 

405 # In practice this rarely happens because the queue is constantly refilled even with components that 

406 # have already run. They just get a BLOCKED priority since their inputs have already been consumed. 

407 if candidate is None: 

408 break 

409 

410 priority, component_name, component = candidate 

411 

412 # If the next component is blocked, we do a check to see if the pipeline is possibly blocked and raise 

413 # a warning if it is. 

414 if priority == ComponentPriority.BLOCKED: 

415 if self._is_pipeline_possibly_blocked(current_pipeline_outputs=pipeline_outputs): 

416 # Pipeline is most likely blocked (most likely a configuration issue) so we raise a warning. 

417 self._find_components_blocking_pipeline( 

418 priority_queue=priority_queue, component_visits=component_visits, inputs=inputs 

419 ) 

420 # We always exit the loop since we cannot run the next component. 

421 break 

422 

423 if len(priority_queue) > 0 and priority == ComponentPriority.DEFER: 

424 component_name, topological_sort = self._tiebreak_waiting_components( 

425 component_name=component_name, 

426 priority=priority, 

427 priority_queue=priority_queue, 

428 topological_sort=cached_topological_sort, 

429 ) 

430 cached_topological_sort = topological_sort 

431 component = self._get_component_with_graph_metadata_and_visits( 

432 component_name, component_visits[component_name] 

433 ) 

434 

435 if pipeline_snapshot: 

436 is_resume = pipeline_snapshot.break_point.component_name == component_name 

437 else: 

438 is_resume = False 

439 component_inputs = self._consume_component_inputs( 

440 component_name=component_name, component=component, inputs=inputs, is_resume=is_resume 

441 ) 

442 

443 # We need to add missing defaults using default values from input sockets because the run signature 

444 # might not provide these defaults for components with inputs defined dynamically upon component 

445 # initialization 

446 component_inputs = self._add_missing_input_defaults(component_inputs, component["input_sockets"]) 

447 

448 # Scenario 1: Pipeline snapshot is provided to resume the pipeline at a specific component 

449 # Deserialize the component_inputs if they are passed in the pipeline_snapshot. 

450 # this check will prevent other component_inputs generated at runtime from being deserialized 

451 if pipeline_snapshot and component_name in pipeline_snapshot.pipeline_state.inputs.keys(): 

452 for key, value in component_inputs.items(): 

453 component_inputs[key] = _deserialize_value_with_schema(value) 

454 

455 try: 

456 component_outputs = self._run_component( 

457 component_name=component_name, 

458 component=component, 

459 inputs=component_inputs, # the inputs to the current component 

460 component_visits=component_visits, 

461 parent_span=span, 

462 break_point=break_point, 

463 ) 

464 except (BreakpointException, PipelineRuntimeError) as error: 

465 saved_break_point: Breakpoint 

466 if isinstance(error, PipelineRuntimeError): 

467 saved_break_point = Breakpoint( 

468 component_name=component_name, 

469 visit_count=component_visits[component_name], 

470 snapshot_file_path=_get_output_dir("pipeline_snapshot"), 

471 ) 

472 else: 

473 saved_break_point = error.break_point 

474 

475 # Create a snapshot of the state of the pipeline before the error occurred. 

476 pipeline_snapshot = _create_pipeline_snapshot( 

477 inputs=_deepcopy_with_exceptions(inputs), 

478 component_inputs=_deepcopy_with_exceptions(component_inputs), 

479 break_point=saved_break_point, 

480 component_visits=component_visits, 

481 original_input_data=data, 

482 ordered_component_names=ordered_component_names, 

483 include_outputs_from=include_outputs_from, 

484 pipeline_outputs=pipeline_outputs, 

485 ) 

486 

487 # Attach the pipeline snapshot to the error before re-raising 

488 error.pipeline_snapshot = pipeline_snapshot 

489 full_file_path = _save_pipeline_snapshot( 

490 pipeline_snapshot=pipeline_snapshot, 

491 raise_on_failure=isinstance(error, BreakpointException), 

492 snapshot_callback=snapshot_callback, 

493 ) 

494 error.pipeline_snapshot_file_path = full_file_path 

495 raise error 

496 

497 # Updates global input state with component outputs and returns outputs that should go to 

498 # pipeline outputs. 

499 component_pipeline_outputs = self._write_component_outputs( 

500 component_name=component_name, 

501 component_outputs=component_outputs, 

502 inputs=inputs, 

503 receivers=cached_receivers[component_name], 

504 include_outputs_from=include_outputs_from, 

505 ) 

506 

507 if component_pipeline_outputs or component_name in include_outputs_from: 

508 pipeline_outputs[component_name] = component_pipeline_outputs 

509 if self._is_queue_stale(priority_queue): 

510 priority_queue = self._fill_queue(ordered_component_names, inputs, component_visits) 

511 

512 if isinstance(break_point, Breakpoint): 

513 logger.warning( 

514 "The given breakpoint {break_point} was never triggered. This is because:\n" 

515 "1. The provided component is not a part of the pipeline execution path.\n" 

516 "2. The component did not reach the visit count specified in the pipeline_breakpoint", 

517 pipeline_breakpoint=break_point, 

518 ) 

519 

520 return pipeline_outputs

522 @staticmethod 

523 async def _run_component_async( 

524 component_name: str, 

525 component: dict[str, Any], 

526 component_inputs: dict[str, Any], 

527 component_visits: dict[str, int], 

528 parent_span: tracing.Span | None = None, 

529 *, 

530 break_point: Breakpoint | None = None, 

531 ) -> Mapping[str, Any]: 

532 """ 

533 Executes a single component asynchronously. 

534 

535 If the component supports async execution, it is awaited directly as it will run async; 

536 otherwise the component is offloaded to executor. 

537 

538 The method also updates the `visits` count of the component, writes outputs to `inputs_state`, 

539 and returns pruned outputs that get stored in `pipeline_outputs`. 

540 

541 :param component_name: The name of the component. 

542 :param component_inputs: Inputs for the component. 

543 :returns: Outputs from the component that can be yielded from run_async_generator. 

544 """ 

545 if ( 

546 isinstance(break_point, Breakpoint) 

547 and break_point.component_name == component_name 

548 and break_point.visit_count == component_visits[component_name] 

549 ): 

550 raise BreakpointException.from_triggered_breakpoint(break_point=break_point) 

551 

552 instance: Component = component["instance"] 

553 

554 with PipelineBase._create_component_span( 

555 component_name=component_name, instance=instance, inputs=component_inputs, parent_span=parent_span 

556 ) as span: 

557 # deepcopy inputs before passing to the tracer so that even if a tracer mutates them 

558 # the component always receives the original unmodified values 

559 component_inputs_copy = _deepcopy_with_exceptions(component_inputs) 

560 span.set_content_tag(_COMPONENT_INPUT, component_inputs) 

561 logger.info("Running component {component_name}", component_name=component_name) 

562 

563 try: 

564 # For sync-only components, _run_component_async dispatches to a thread via asyncio.to_thread, 

565 # which copies the current contextvars context — preserving e.g. the active tracing span. 

566 outputs = await _execute_component_async(instance, **component_inputs_copy) 

567 except Exception as error:

569

574

576

578 span.set_content_tag(_COMPONENT_OUTPUT, outputs) 

579 

580 return outputs

582 @staticmethod 

583 async def _wait_for_tasks( 

584 running_tasks: dict[asyncio.Task, str], scheduled_components: set[str], *, return_when: str 

585 ) -> AsyncIterator[dict[str, Any]]: 

586 """ 

587 Waits for running tasks to finish and yields their partial outputs. 

588 

589 :param running_tasks: Mapping of in-flight tasks to the name of the component they run. Finished tasks are 

590 removed in place. 

591 :param scheduled_components: Set of component names that are scheduled but not yet finished. Finished 

592 components are discarded in place. 

593 :param return_when: Either `asyncio.FIRST_COMPLETED` to wait for a single task or `asyncio.ALL_COMPLETED` to 

594 wait for every running task. 

595 :returns: An async iterator of partial outputs, one per finished component that produced an output. 

596 """ 

597 if not running_tasks: 

598 return 

599 

600 done, _pending = await asyncio.wait(running_tasks.keys(), return_when=return_when) 

601 for finished in done: 

602 finished_component_name = running_tasks.pop(finished) 

603 try: 

604 partial_result = finished.result() 

605 except Exception: 

606 # A component failed. Cancel and drain the remaining in-flight tasks so they don't keep running in 

607 # the background (and leak) after the run is aborted, then re-raise the original error. 

608 await Pipeline._cancel_in_flight_tasks(running_tasks, scheduled_components) 

609 raise 

610 scheduled_components.discard(finished_component_name) 

611 if partial_result: 

612 yield {finished_component_name: _deepcopy_with_exceptions(partial_result)}

614 @staticmethod 

615 async def _cancel_in_flight_tasks(running_tasks: dict[asyncio.Task, str], scheduled_components: set[str]) -> None: 

616 """ 

617 Cancels all in-flight tasks and waits for the cancellations to settle. 

618 

619 Called when a component fails or when the run is abandoned early so that sibling tasks don't keep running in 

620 the background after the pipeline run is aborted. Exceptions from the cancelled tasks are suppressed since we 

621 are already unwinding. 

622 

623 Note: cancellation is only effective for components that run natively async. Sync components are offloaded to 

624 a thread via `asyncio.to_thread` and a running thread cannot be interrupted: cancelling its task abandons 

625 the await, but the thread keeps running until the component's `run` returns. Its outputs are then discarded 

626 (the task never writes them to the pipeline state), so state stays consistent, but side effects (e.g. API 

627 calls) still complete and the thread can outlive this cleanup. 

628 

629 :param running_tasks: Mapping of in-flight tasks to component names. Cleared in place. 

630 :param scheduled_components: Set of scheduled-but-unfinished component names. Cleared in place. 

631 """ 

632 for task in running_tasks: 

633 task.cancel() 

634 # return_exceptions=True so a failing or cancelled sibling doesn't mask the original error we re-raise. 

635 await asyncio.gather(*running_tasks.keys(), return_exceptions=True) 

636 for component_name in running_tasks.values(): 

637 scheduled_components.discard(component_name) 

638 running_tasks.clear()

640 async def _run_component_in_isolation( 

641 self, 

642 *, 

643 component_name: str, 

644 inputs: dict[str, dict[str, list[dict[str, Any]]]], 

645 pipeline_outputs: dict[str, Any], 

646 component_visits: dict[str, int], 

647 running_tasks: dict[asyncio.Task, str], 

648 scheduled_components: set[str], 

649 cached_receivers: dict[str, Any], 

650 include_outputs_from: set[str], 

651 parent_span: tracing.Span | None, 

652 ) -> AsyncIterator[dict[str, Any]]: 

653 """ 

654 Runs a component with HIGHEST priority in isolation. 

655 

656 We need to run components with HIGHEST priority (i.e. components with a GreedyVariadic input socket) by 

657 themselves, without any other components running concurrently. Otherwise, downstream components could produce 

658 additional inputs for the GreedyVariadic socket. 

659 

660 :param component_name: The name of the component to run. 

661 :param inputs: The global input state shared by all components. Mutated in place. 

662 :param pipeline_outputs: The accumulated pipeline outputs. Mutated in place. 

663 :param component_visits: Current state of component visits. Mutated in place. 

664 :param running_tasks: Mapping of in-flight tasks to component names. Drained in place before running. 

665 :param scheduled_components: Set of scheduled-but-unfinished component names. Mutated in place. 

666 :param cached_receivers: Precomputed mapping of component name to its downstream receivers. 

667 :param include_outputs_from: Set of component names whose outputs should always be included in the output. 

668 :param parent_span: The parent tracing span for the pipeline run. 

669 :returns: An async iterator of partial outputs. 

670 """ 

671 # 1) Wait for all in-flight tasks to finish so the HIGHEST component runs alone. 

672 async for partial_outputs in self._wait_for_tasks( 

673 running_tasks, scheduled_components, return_when=asyncio.ALL_COMPLETED 

674 ): 

675 yield partial_outputs 

676 

677 if component_name in scheduled_components: 

678 # If it's already scheduled for some reason, skip. 

679 return 

680 

681 # 2) Run the HIGHEST component by itself. 

682 scheduled_components.add(component_name) 

683 component = self._get_component_with_graph_metadata_and_visits(component_name, component_visits[component_name]) 

684 component_inputs = self._consume_component_inputs(component_name, component, inputs) 

685 component_inputs = self._add_missing_input_defaults(component_inputs, component["input_sockets"]) 

686 

687 component_outputs = await self._run_component_async( 

688 component_name=component_name, 

689 component=component, 

690 component_inputs=component_inputs, 

691 component_visits=component_visits, 

692 parent_span=parent_span, 

693 ) 

694 

695 pruned = self._write_component_outputs( 

696 component_name=component_name, 

697 component_outputs=component_outputs, 

698 inputs=inputs, 

699 receivers=cached_receivers[component_name], 

700 include_outputs_from=include_outputs_from, 

701 ) 

702 if pruned or component_name in include_outputs_from: 

703 pipeline_outputs[component_name] = pruned 

704 

705 scheduled_components.remove(component_name) 

706 if pruned or component_name in include_outputs_from: 

707 yield {component_name: _deepcopy_with_exceptions(pruned)}

709 def _schedule_component( 

710 self, 

711 *, 

712 component_name: str, 

713 inputs: dict[str, dict[str, list[dict[str, Any]]]], 

714 pipeline_outputs: dict[str, Any], 

715 component_visits: dict[str, int], 

716 running_tasks: dict[asyncio.Task, str], 

717 scheduled_components: set[str], 

718 ready_sem: asyncio.Semaphore, 

719 cached_receivers: dict[str, Any], 

720 include_outputs_from: set[str], 

721 parent_span: tracing.Span | None, 

722 ) -> None: 

723 """ 

724 Schedules a component to run as a background task without waiting for it to finish. 

725 

726 Inputs are consumed synchronously here (before the task is created) so that other components scheduled in the 

727 same iteration of the scheduling loop observe the updated input state and don't race for the same inputs. 

728 

729 :param component_name: The name of the component to schedule. 

730 :param inputs: The global input state shared by all components. Mutated in place. 

731 :param pipeline_outputs: The accumulated pipeline outputs. Mutated in place by the task once it finishes. 

732 :param component_visits: Current state of component visits. Mutated in place by the task once it finishes. 

733 :param running_tasks: Mapping of in-flight tasks to component names. The new task is registered here. 

734 :param scheduled_components: Set of scheduled-but-unfinished component names. Mutated in place. 

735 :param ready_sem: Semaphore bounding how many components run concurrently. 

736 :param cached_receivers: Precomputed mapping of component name to its downstream receivers. 

737 :param include_outputs_from: Set of component names whose outputs should always be included in the output. 

738 :param parent_span: The parent tracing span for the pipeline run. 

739 """ 

740 if component_name in scheduled_components: 

741 return # already scheduled, do nothing 

742 

743 scheduled_components.add(component_name) 

744 

745 component = self._get_component_with_graph_metadata_and_visits(component_name, component_visits[component_name]) 

746 component_inputs = self._consume_component_inputs(component_name, component, inputs) 

747 component_inputs = self._add_missing_input_defaults(component_inputs, component["input_sockets"]) 

748 

749 async def _runner() -> Mapping[str, Any]: 

750 async with ready_sem: 

751 component_outputs = await self._run_component_async( 

752 component_name=component_name, 

753 component=component, 

754 component_inputs=component_inputs, 

755 component_visits=component_visits, 

756 parent_span=parent_span, 

757 ) 

758 

759 pruned = self._write_component_outputs( 

760 component_name=component_name, 

761 component_outputs=component_outputs, 

762 inputs=inputs, 

763 receivers=cached_receivers[component_name], 

764 include_outputs_from=include_outputs_from, 

765 ) 

766 if pruned or component_name in include_outputs_from: 

767 pipeline_outputs[component_name] = pruned 

768 

769 scheduled_components.remove(component_name) 

770 return pruned 

771 

772 task = asyncio.create_task(_runner()) 

773 running_tasks[task] = component_name

775 async def run_async_generator( # noqa: PLR0915,C901 

776 self, data: dict[str, Any], include_outputs_from: set[str] | None = None, concurrency_limit: int = 4 

777 ) -> AsyncGenerator[dict[str, Any], None]: 

778 """ 

779 Executes the pipeline step by step asynchronously, yielding partial outputs when any component finishes. 

780 

781 Usage: 

782 ```python 

783 from haystack import Document 

784 from haystack.components.builders import ChatPromptBuilder 

785 from haystack.dataclasses import ChatMessage 

786 from haystack.utils import Secret 

787 from haystack.document_stores.in_memory import InMemoryDocumentStore 

788 from haystack.components.retrievers.in_memory import InMemoryBM25Retriever 

789 from haystack.components.generators.chat import OpenAIChatGenerator 

790 from haystack.components.builders.prompt_builder import PromptBuilder 

791 from haystack import Pipeline 

792 import asyncio 

793 

794 # Write documents to InMemoryDocumentStore 

795 document_store = InMemoryDocumentStore() 

796 document_store.write_documents([ 

797 Document(content="My name is Jean and I live in Paris."), 

798 Document(content="My name is Mark and I live in Berlin."), 

799 Document(content="My name is Giorgio and I live in Rome.") 

800 ]) 

801 

802 prompt_template = [ 

803 ChatMessage.from_user( 

804 ''' 

805 Given these documents, answer the question. 

806 Documents: 

807 {% for doc in documents %} 

808 {{ doc.content }} 

809 {% endfor %} 

810 Question: {{question}} 

811 Answer: 

812 ''') 

813 ] 

814 

815 # Create and connect pipeline components 

816 retriever = InMemoryBM25Retriever(document_store=document_store) 

817 prompt_builder = ChatPromptBuilder(template=prompt_template) 

818 llm = OpenAIChatGenerator() 

819 

820 rag_pipeline = Pipeline() 

821 rag_pipeline.add_component("retriever", retriever) 

822 rag_pipeline.add_component("prompt_builder", prompt_builder) 

823 rag_pipeline.add_component("llm", llm) 

824 rag_pipeline.connect("retriever", "prompt_builder.documents") 

825 rag_pipeline.connect("prompt_builder", "llm") 

826 

827 # Prepare input data 

828 question = "Who lives in Paris?" 

829 data = { 

830 "retriever": {"query": question}, 

831 "prompt_builder": {"question": question}, 

832 } 

833 

834 

835 # Process results as they become available 

836 async def process_results(): 

837 async for partial_output in rag_pipeline.run_async_generator( 

838 data=data, 

839 include_outputs_from={"retriever", "llm"} 

840 ): 

841 # Each partial_output contains the results from a completed component 

842 if "retriever" in partial_output: 

843 print("Retrieved documents:", len(partial_output["retriever"]["documents"])) 

844 if "llm" in partial_output: 

845 print("Generated answer:", partial_output["llm"]["replies"][0]) 

846 

847 

848 asyncio.run(process_results()) 

849 ``` 

850 

851 :param data: Initial input data to the pipeline. 

852 :param concurrency_limit: The maximum number of components that are allowed to run concurrently. 

853 :param include_outputs_from: 

854 Set of component names whose individual outputs are to be 

855 included in the pipeline's output. For components that are 

856 invoked multiple times (in a loop), only the last-produced 

857 output is included. 

858 :return: An async iterator containing partial (and final) outputs. 

859 

860 :raises ValueError: 

861 If invalid inputs are provided to the pipeline, or if `concurrency_limit` is less than 1. 

862 :raises PipelineMaxComponentRuns: 

863 If a component exceeds the maximum number of allowed executions within the pipeline. 

864 :raises PipelineRuntimeError: 

865 If the Pipeline contains cycles with unsupported connections that would cause 

866 it to get stuck and fail running. 

867 Or if a Component fails or returns output in an unsupported type. 

868 """ 

869 if concurrency_limit < 1: 

870 raise ValueError("concurrency_limit must be greater than or equal to 1.") 

871 

872 pipeline_running(self) # telemetry 

873 

874 # warm up the pipeline by running each component's warm_up_async (or warm_up) method 

875 await self.warm_up_async() 

876 

877 if include_outputs_from is None: 

878 include_outputs_from = set() 

879 

880 pipeline_outputs: dict[str, Any] = {} 

881 

882 # Normalize `data` and raise ValueError if the input is malformed in some way. 

883 data = self._prepare_component_input_data(data) 

884 

885 # Raise ValueError if input is malformed in some way 

886 self.validate_input(data) 

887 

888 # We create a list of components in the pipeline sorted by name, so that the algorithm runs 

889 # deterministically and independent of insertion order into the pipeline. 

890 ordered_component_names = sorted(self.graph.nodes.keys()) 

891 

892 # We track component visits to decide if a component can run. 

893 component_visits = dict.fromkeys(ordered_component_names, 0) 

894 

895 cached_topological_sort = None 

896 # We need to access a component's receivers multiple times during a pipeline run. 

897 # We store them here for easy access. 

898 cached_receivers = {name: self._find_receivers_from(name) for name in ordered_component_names} 

899 

900 # Ephemeral concurrency state shared (and mutated in place) by the scheduling helpers below. 

901 ready_sem = asyncio.Semaphore(concurrency_limit) 

902 running_tasks: dict[asyncio.Task, str] = {} 

903 # A set of component names that have been scheduled but not finished. 

904 scheduled_components: set[str] = set() 

905 

906 with tracing.tracer.trace( 

907 "haystack.pipeline.run", 

908 tags={ 

909 "haystack.pipeline.input_data": data, 

910 "haystack.pipeline.output_data": pipeline_outputs, 

911 "haystack.pipeline.metadata": self.metadata, 

912 "haystack.pipeline.max_runs_per_component": self._max_runs_per_component, 

913 "haystack.pipeline.execution_mode": "async", 

914 }, 

915 ) as parent_span: 

916 inputs = self._convert_to_internal_format(pipeline_inputs=data) 

917 

918 # check if pipeline is blocked before execution 

919 self.validate_pipeline(self._fill_queue(ordered_component_names, inputs, component_visits)) 

920 

921 try: 

922 while True: 

923 # We rebuild the priority queue every iteration: each iteration waits for one or more concurrent 

924 # tasks to finish, which mutates `inputs` and can change many components' priorities at once, so 

925 # we rebuild to give every scheduling decision an up-to-date view. 

926 priority_queue = self._fill_queue(ordered_component_names, inputs, component_visits) 

927 candidate = self._get_next_runnable_component(priority_queue, component_visits) 

928 

929 # If we can't make progress with the queue but tasks are running, we wait for one to finish and 

930 # retry to potentially unblock the priority queue. 

931 if (candidate is None or candidate[0] == ComponentPriority.BLOCKED) and running_tasks: 

932 async for partial_outputs in self._wait_for_tasks( 

933 running_tasks, scheduled_components, return_when=asyncio.FIRST_COMPLETED 

934 ): 

935 yield partial_outputs 

936 continue 

937 

938 # If there are no runnable components left and nothing is running, we can exit the loop. 

939 if candidate is None and not running_tasks: 

940 break 

941 

942 priority, component_name, component = candidate # type: ignore 

943 

944 # If the next component is blocked, we do a check to see if the pipeline is possibly blocked and 

945 # raise a warning if it is. 

946 if priority == ComponentPriority.BLOCKED and not running_tasks: 

947 if self._is_pipeline_possibly_blocked(current_pipeline_outputs=pipeline_outputs): 

948 # Pipeline is most likely blocked (most likely a configuration issue) so we raise a warning. 

949 self._find_components_blocking_pipeline( 

950 priority_queue=priority_queue, component_visits=component_visits, inputs=inputs 

951 ) 

952 # We always exit the loop since we cannot run the next component. 

953 break 

954 

955 # If the next component is already scheduled, we wait for a task to finish to make progress. 

956 if component_name in scheduled_components: 

957 async for partial_outputs in self._wait_for_tasks( 

958 running_tasks, scheduled_components, return_when=asyncio.FIRST_COMPLETED 

959 ): 

960 yield partial_outputs 

961 continue 

962 

963 if priority == ComponentPriority.HIGHEST: 

964 # A HIGHEST priority component must run alone, so we hand off to the isolation helper. 

965 async for partial_outputs in self._run_component_in_isolation( 

966 component_name=component_name, 

967 inputs=inputs, 

968 pipeline_outputs=pipeline_outputs, 

969 component_visits=component_visits, 

970 running_tasks=running_tasks, 

971 scheduled_components=scheduled_components, 

972 cached_receivers=cached_receivers, 

973 include_outputs_from=include_outputs_from, 

974 parent_span=parent_span, 

975 ): 

976 yield partial_outputs 

977 continue 

978 

979 if priority == ComponentPriority.READY: 

980 # Schedule this component, then schedule as many additional READY components as concurrency 

981 # allows. 

982 self._schedule_component( 

983 component_name=component_name, 

984 inputs=inputs, 

985 pipeline_outputs=pipeline_outputs, 

986 component_visits=component_visits, 

987 running_tasks=running_tasks, 

988 scheduled_components=scheduled_components, 

989 ready_sem=ready_sem, 

990 cached_receivers=cached_receivers, 

991 include_outputs_from=include_outputs_from, 

992 parent_span=parent_span, 

993 ) 

994 

995 # Possibly schedule more READY tasks if concurrency not fully used 

996 while len(priority_queue) > 0 and not ready_sem.locked(): 

997 peek_priority, peek_name = priority_queue.peek() 

998 if peek_priority != ComponentPriority.READY: 

999 # We stop scheduling: the next component is BLOCKED (can't run), HIGHEST (must run 

1000 # alone), or DEFER (waiting for more inputs - we only schedule it once it becomes 

1001 # READY). 

1002 break 

1003 priority_queue.pop() 

1004 self._schedule_component( 

1005 component_name=peek_name, 

1006 inputs=inputs, 

1007 pipeline_outputs=pipeline_outputs, 

1008 component_visits=component_visits, 

1009 running_tasks=running_tasks, 

1010 scheduled_components=scheduled_components, 

1011 ready_sem=ready_sem, 

1012 cached_receivers=cached_receivers, 

1013 include_outputs_from=include_outputs_from, 

1014 parent_span=parent_span, 

1015 ) 

1016 

1017 # We only schedule components with priority DEFER when no other tasks are running. 

1018 elif priority == ComponentPriority.DEFER and not running_tasks: 

1019 if len(priority_queue) > 0: 

1020 component_name, cached_topological_sort = self._tiebreak_waiting_components( 

1021 component_name=component_name, 

1022 priority=priority, 

1023 priority_queue=priority_queue, 

1024 topological_sort=cached_topological_sort, 

1025 ) 

1026 

1027 self._schedule_component( 

1028 component_name=component_name, 

1029 inputs=inputs, 

1030 pipeline_outputs=pipeline_outputs, 

1031 component_visits=component_visits, 

1032 running_tasks=running_tasks, 

1033 scheduled_components=scheduled_components, 

1034 ready_sem=ready_sem, 

1035 cached_receivers=cached_receivers, 

1036 include_outputs_from=include_outputs_from, 

1037 parent_span=parent_span, 

1038 ) 

1039 

1040 # To make progress, we wait for one task to complete before restarting the loop. 

1041 async for partial_outputs in self._wait_for_tasks( 

1042 running_tasks, scheduled_components, return_when=asyncio.FIRST_COMPLETED 

1043 ): 

1044 yield partial_outputs 

1045 

1046 # Safety net: drain any leftover tasks once the scheduling loop has finished. With the current loop 

1047 # both `break` paths require `running_tasks` to be empty, so this is a no-op. We keep it so that a 

1048 # future change adding a `break` that leaves tasks in flight doesn't lose outputs. 

1049 async for partial_outputs in self._wait_for_tasks( 

1050 running_tasks, scheduled_components, return_when=asyncio.ALL_COMPLETED 

1051 ): 

1052 yield partial_outputs 

1053 

1054 # Yield the final pipeline outputs. 

1055 yield pipeline_outputs 

1056 finally: 

1057 # If iteration is abandoned early (e.g. the consumer stops iterating the generator and closes it) or 

1058 # the run is cancelled, cancel any tasks still in flight so they don't leak. 

1059 # This is a no-op on normal completion and on a component error, since no tasks are left running by then 

1060 await self._cancel_in_flight_tasks(running_tasks, scheduled_components)

1179 def stream( 

1180 self, 

1181 data: dict[str, Any], 

1182 *, 

1183 streaming_components: list[str] | None = None, 

1184 include_outputs_from: set[str] | None = None, 

1185 concurrency_limit: int = 4, 

1186 cancel_on_abandon: bool = True, 

1187 ) -> PipelineStreamHandle: 

1188 """ 

1189 Run the pipeline and return a handle that streams `StreamingChunk`s as they arrive. 

1190 

1191 Iterate the handle with `async for` to consume chunks; after iteration ends, `handle.result` holds the final 

1192 pipeline output dict (same as `run_async`). By default, if iteration is abandoned, the underlying pipeline task 

1193 is cancelled automatically. Pass `cancel_on_abandon=False` to instead let the pipeline run to completion. 

1194 

1195 For every async-capable component that exposes a `streaming_callback` input socket, a forwarder is injected at 

1196 runtime that pushes chunks onto the handle's queue. If a `streaming_callback` is provided at component init or 

1197 at runtime (inside `data`, e.g. `data={"llm": {"streaming_callback": cb}}`), it is also invoked for each chunk. 

1198 Async callbacks are preferred; a sync callback is accepted but will run synchronously on the event loop and 

1199 may block it. 

1200 

1201 Usage: 

1202 ```python 

1203 import asyncio 

1204 

1205 from haystack.components.builders import ChatPromptBuilder 

1206 from haystack.components.generators.chat import OpenAIChatGenerator 

1207 from haystack import Pipeline 

1208 from haystack.dataclasses import ChatMessage 

1209 

1210 pipe = Pipeline() 

1211 pipe.add_component( 

1212 "prompt_builder", 

1213 ChatPromptBuilder(template=[ChatMessage.from_user("Tell me about {{topic}}")]), 

1214 ) 

1215 pipe.add_component("llm", OpenAIChatGenerator()) 

1216 pipe.connect("prompt_builder.prompt", "llm.messages") 

1217 

1218 async def main(): 

1219 handle = pipe.stream(data={"prompt_builder": {"topic": "Italy"}}) 

1220 async for chunk in handle: 

1221 print(chunk.content, end="", flush=True) 

1222 return handle.result 

1223 

1224 result = asyncio.run(main()) 

1225 print(result["llm"]["replies"]) 

1226 ``` 

1227 

1228 :param data: 

1229 A dictionary of inputs for the pipeline's components. Each key is a component name 

1230 and its value is a dictionary of that component's input parameters: 

1231 ``` 

1232 data = { 

1233 "comp1": {"input1": 1, "input2": 2}, 

1234 } 

1235 ``` 

1236 For convenience, this format is also supported when input names are unique: 

1237 ``` 

1238 data = { 

1239 "input1": 1, "input2": 2, 

1240 } 

1241 ``` 

1242 :param streaming_components: Names of components to stream from. If `None` (default), every streaming-capable 

1243 component is forwarded. If a list, only the listed components are forwarded; unknown names or names of 

1244 components that do not support streaming raise `ValueError`. 

1245 :param include_outputs_from: 

1246 Set of component names whose individual outputs are to be 

1247 included in the pipeline's output. For components that are 

1248 invoked multiple times (in a loop), only the last-produced 

1249 output is included. 

1250 :param concurrency_limit: The maximum number of components that should be allowed to run concurrently. 

1251 :param cancel_on_abandon: If `True` (default), the underlying pipeline task is cancelled when iteration is 

1252 abandoned. If `False`, the pipeline runs to completion even when the consumer stops reading. 

1253 :returns: 

1254 A `PipelineStreamHandle` that is async-iterable over `StreamingChunk`s. After iteration ends, 

1255 `handle.result` holds the final pipeline output dict (same shape as `run_async`). 

1256 

1257 :raises ValueError: 

1258 If `streaming_components` contains unknown component names or components that do not support streaming, 

1259 or if invalid inputs are provided to the pipeline, or if `concurrency_limit` is less than 1. 

1260 :raises PipelineRuntimeError: 

1261 Surfaced during iteration. If the Pipeline contains cycles with unsupported connections that would cause 

1262 it to get stuck and fail running, or if a Component fails or returns output in an unsupported type. 

1263 :raises PipelineMaxComponentRuns: 

1264 Surfaced during iteration. If a Component reaches the maximum number of times it can be run in this 

1265 Pipeline. 

1266 """ 

1267 streaming_capable = { 

1268 name 

1269 for name in self.graph.nodes 

1270 if getattr(self.graph.nodes[name]["instance"], "__haystack_supports_async__", False) 

1271 and "streaming_callback" in self.graph.nodes[name]["instance"].__haystack_input__ 

1272 } 

1273 if streaming_components is not None: 

1274 requested = set(streaming_components) 

1275 unknown = requested - set(self.graph.nodes) 

1276 non_streaming = requested - unknown - streaming_capable 

1277 if unknown: 

1278 raise ValueError(f"Unknown components in streaming_components: {sorted(unknown)}") 

1279 if non_streaming: 

1280 raise ValueError(f"These components do not support streaming: {sorted(non_streaming)}") 

1281 

1282 queue: asyncio.Queue[StreamingChunk | _EndOfStream] = asyncio.Queue() 

1283 

1284 def make_forwarder(user_callback: StreamingCallbackT | None) -> AsyncStreamingCallbackT: 

1285 async def forwarder(chunk: StreamingChunk) -> None: 

1286 await queue.put(chunk) 

1287 if user_callback is not None: 

1288 await _invoke_streaming_callback(user_callback, chunk) 

1289 

1290 return forwarder 

1291 

1292 new_data: dict[str, Any] = self._prepare_component_input_data(data) 

1293 for name in streaming_capable: 

1294 if streaming_components is not None and name not in streaming_components: 

1295 continue 

1296 instance = self.graph.nodes[name]["instance"] 

1297 comp_inputs = new_data.setdefault(name, {}) 

1298 

1299 user_callback = select_streaming_callback( 

1300 init_callback=getattr(instance, "streaming_callback", None), 

1301 runtime_callback=comp_inputs.get("streaming_callback"), 

1302 requires_async=True, 

1303 ) 

1304 comp_inputs["streaming_callback"] = make_forwarder(user_callback) 

1305 

1306 async def runner() -> dict[str, Any]: 

1307 try: 

1308 return await self.run_async( 

1309 new_data, include_outputs_from=include_outputs_from, concurrency_limit=concurrency_limit 

1310 ) 

1311 finally: 

1312 await queue.put(PipelineStreamHandle._END_OF_STREAM) 

1313 

1314 task = asyncio.create_task(runner()) 

1315 return PipelineStreamHandle(queue=queue, task=task, cancel_on_abandon=cancel_on_abandon)
