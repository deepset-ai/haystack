36class Pipeline(PipelineBase): 

37 """ 

38 Synchronous version of the orchestration engine. 

39 

40 Orchestrates component execution according to the execution graph, one after the other. 

41 """ 

42 

43 @staticmethod 

44 def _run_component( 

45 component_name: str, 

46 component: dict[str, Any], 

47 inputs: dict[str, Any], 

48 component_visits: dict[str, int], 

49 parent_span: tracing.Span | None = None, 

50 *, 

51 break_point: Breakpoint | None = None, 

52 ) -> Mapping[str, Any]: 

53 """ 

54 Runs a Component with the given inputs. 

55 

56 :param component_name: Name of the Component. 

57 :param component: Component with component metadata. 

58 :param inputs: Inputs for the Component. 

59 :param component_visits: Current state of component visits. 

60 :param parent_span: The parent span to use for the newly created span. 

61 This is to allow tracing to be correctly linked to the pipeline run. 

62 :raises PipelineRuntimeError: If Component doesn't return a dictionary. 

63 :return: The output of the Component. 

64 """ 

65 if ( 

66 isinstance(break_point, Breakpoint) 

67 and break_point.component_name == component_name 

68 and break_point.visit_count == component_visits[component_name] 

69 ): 

70 raise BreakpointException.from_triggered_breakpoint(break_point=break_point) 

71 

72 instance: Component = component["instance"] 

73 

74 with PipelineBase._create_component_span( 

75 component_name=component_name, instance=instance, inputs=inputs, parent_span=parent_span 

76 ) as span: 

77 # deepcopy inputs before passing to the tracer so that even if a tracer mutates them 

78 # the component always receives the original unmodified values 

79 inputs_copy = _deepcopy_with_exceptions(inputs) 

80 span.set_content_tag(_COMPONENT_INPUT, inputs) 

81 logger.info("Running component {component_name}", component_name=component_name) 

82 

83 try: 

84 component_output = instance.run(**inputs_copy) 

85 except BreakpointException as error: 

86 # Re-raise BreakpointException to preserve the original exception context 

87 # This is important when Agent components internally use Pipeline._run_component 

88 # and trigger breakpoints that need to bubble up to the main pipeline 

89 raise error 

90 

91 # Any components that internally use Pipeline._run_component could raise a PipelineRuntimeError with 

92 # additional context (e.g. Agent raises an agent snapshot) so we re-raise here instead of wrapping it in 

93 # another PipelineRuntimeError 

94 

95 except PipelineRuntimeError as runtime_error: 

96 raise runtime_error 

97 

98 # Catch all other exceptions and wrap them in a PipelineRuntimeError 

99 except Exception as error: 

100 raise PipelineRuntimeError.from_exception(component_name, instance.__class__, error) from error 

101 

102 component_visits[component_name] += 1 

103 

104 if not isinstance(component_output, Mapping): 

105 raise PipelineRuntimeError.from_invalid_output(component_name, instance.__class__, component_output) 

106 

107 _validate_component_output_keys(component_name, component, component_output) 

108 

109 span.set_tag(_COMPONENT_VISITS, component_visits[component_name]) 

110 span.set_content_tag(_COMPONENT_OUTPUT, component_output) 

111 

112 return component_output

114 def run( # noqa: PLR0915, PLR0912, C901 

115 self, 

116 data: dict[str, Any], 

117 include_outputs_from: set[str] | None = None, 

118 *, 

119 break_point: Breakpoint | AgentBreakpoint | None = None, 

120 pipeline_snapshot: PipelineSnapshot | None = None, 

121 snapshot_callback: SnapshotCallback | None = None, 

122 ) -> dict[str, Any]: 

123 """ 

124 Runs the Pipeline with given input data. 

125 

126 Usage: 

127 ```python 

128 from haystack import Pipeline, Document 

129 from haystack.components.builders.answer_builder import AnswerBuilder 

130 from haystack.components.builders.chat_prompt_builder import ChatPromptBuilder 

131 from haystack.components.generators.chat import OpenAIChatGenerator 

132 from haystack.components.retrievers.in_memory import InMemoryBM25Retriever 

133 from haystack.dataclasses import ChatMessage 

134 from haystack.document_stores.in_memory import InMemoryDocumentStore 

135 from haystack.utils import Secret 

136 

137 # Write documents to InMemoryDocumentStore 

138 document_store = InMemoryDocumentStore() 

139 document_store.write_documents([ 

140 Document(content="My name is Jean and I live in Paris."), 

141 Document(content="My name is Mark and I live in Berlin."), 

142 Document(content="My name is Giorgio and I live in Rome.") 

143 ]) 

144 

145 retriever = InMemoryBM25Retriever(document_store=document_store) 

146 

147 prompt_template = \"\"\" 

148 Given these documents, answer the question. 

149 Documents: 

150 {% for doc in documents %} 

151 {{ doc.content }} 

152 {% endfor %} 

153 Question: {{question}} 

154 Answer: 

155 \"\"\" 

156 

157 template = [ChatMessage.from_user(prompt_template)] 

158 prompt_builder = ChatPromptBuilder( 

159 template=template, 

160 required_variables=["question", "documents"], 

161 variables=["question", "documents"] 

162 ) 

163 

164 llm = OpenAIChatGenerator() 

165 rag_pipeline = Pipeline() 

166 rag_pipeline.add_component("retriever", retriever) 

167 rag_pipeline.add_component("prompt_builder", prompt_builder) 

168 rag_pipeline.add_component("llm", llm) 

169 rag_pipeline.connect("retriever", "prompt_builder.documents") 

170 rag_pipeline.connect("prompt_builder", "llm") 

171 

172 question = "Who lives in Paris?" 

173 results = rag_pipeline.run( 

174 { 

175 "retriever": {"query": question}, 

176 "prompt_builder": {"question": question}, 

177 } 

178 ) 

179 

180 print(results["llm"]["replies"][0].text) 

181 # Jean lives in Paris 

182 ``` 

183 

184 :param data: 

185 A dictionary of inputs for the pipeline's components. Each key is a component name 

186 and its value is a dictionary of that component's input parameters: 

187 ``` 

188 data = { 

189 "comp1": {"input1": 1, "input2": 2}, 

190 } 

191 ``` 

192 For convenience, this format is also supported when input names are unique: 

193 ``` 

194 data = { 

195 "input1": 1, "input2": 2, 

196 } 

197 ``` 

198 :param include_outputs_from: 

199 Set of component names whose individual outputs are to be 

200 included in the pipeline's output. For components that are 

201 invoked multiple times (in a loop), only the last-produced 

202 output is included. 

203 

204 :param break_point: 

205 A set of breakpoints that can be used to debug the pipeline execution. 

206 

207 :param pipeline_snapshot: 

208 A dictionary containing a snapshot of a previously saved pipeline execution. 

209 

210 :param snapshot_callback: 

211 Optional callback function that is invoked when a pipeline snapshot is created. 

212 The callback receives a `PipelineSnapshot` object and can return an optional string 

213 (e.g., a file path or identifier). 

214 If provided, the callback is used instead of the default file-saving behavior, 

215 allowing custom handling of snapshots (e.g., saving to a database, sending to a remote service). 

216 If not provided, the default behavior saves snapshots to a JSON file. 

217 

218 :returns: 

219 A dictionary where each entry corresponds to a component name 

220 and its output. If `include_outputs_from` is `None`, this dictionary 

221 will only contain the outputs of leaf components, i.e., components 

222 without outgoing connections. 

223 

224 :raises ValueError: 

225 If invalid inputs are provided to the pipeline. 

226 :raises PipelineRuntimeError: 

227 If the Pipeline contains cycles with unsupported connections that would cause 

228 it to get stuck and fail running. 

229 Or if a Component fails or returns output in an unsupported type. 

230 :raises PipelineMaxComponentRuns: 

231 If a Component reaches the maximum number of times it can be run in this Pipeline. 

232 :raises PipelineBreakpointException: 

233 When a pipeline_breakpoint is triggered. Contains the component name, state, and partial results. 

234 """ 

235 pipeline_running(self) 

236 

237 if break_point and pipeline_snapshot: 

238 msg = ( 

239 "pipeline_breakpoint and pipeline_snapshot cannot be provided at the same time. " 

240 "The pipeline run will be aborted." 

241 ) 

242 raise PipelineInvalidPipelineSnapshotError(message=msg) 

243 

244 # make sure all breakpoints are valid, i.e. reference components in the pipeline 

245 if break_point: 

246 _validate_break_point_against_pipeline(break_point, self.graph) 

247 

248 # TODO: Remove this warmup once we can check reliably whether a component has been warmed up or not 

249 # As of now it's here to make sure we don't have failing tests that assume warm_up() is called in run() 

250 self.warm_up() 

251 

252 if include_outputs_from is None: 

253 include_outputs_from = set() 

254 

255 pipeline_outputs: dict[str, Any] = {} 

256 

257 if not pipeline_snapshot: 

258 # normalize `data` 

259 data = self._prepare_component_input_data(data) 

260 

261 # Raise ValueError if input is malformed in some way 

262 self.validate_input(data) 

263 

264 # We create a list of components in the pipeline sorted by name, so that the algorithm runs 

265 # deterministically and independent of insertion order into the pipeline. 

266 ordered_component_names = sorted(self.graph.nodes.keys()) 

267 

268 # We track component visits to decide if a component can run. 

269 component_visits = dict.fromkeys(ordered_component_names, 0) 

270 

271 else: 

272 # Validate the pipeline snapshot against the current pipeline graph 

273 _validate_pipeline_snapshot_against_pipeline(pipeline_snapshot, self.graph) 

274 

275 # Handle resuming the pipeline from a snapshot 

276 component_visits = pipeline_snapshot.pipeline_state.component_visits 

277 ordered_component_names = pipeline_snapshot.ordered_component_names 

278 data = _deserialize_value_with_schema(pipeline_snapshot.pipeline_state.inputs) 

279 

280 # include_outputs_from from the snapshot when resuming 

281 include_outputs_from = pipeline_snapshot.include_outputs_from 

282 

283 # also intermediate_outputs from the snapshot when resuming 

284 pipeline_outputs = _deserialize_value_with_schema(pipeline_snapshot.pipeline_state.pipeline_outputs) 

285 

286 cached_topological_sort = None 

287 # We need to access a component's receivers multiple times during a pipeline run. 

288 # We store them here for easy access. 

289 cached_receivers = {name: self._find_receivers_from(name) for name in ordered_component_names} 

290 

291 with tracing.tracer.trace( 

292 "haystack.pipeline.run", 

293 tags={ 

294 "haystack.pipeline.input_data": data, 

295 "haystack.pipeline.output_data": pipeline_outputs, 

296 "haystack.pipeline.metadata": self.metadata, 

297 "haystack.pipeline.max_runs_per_component": self._max_runs_per_component, 

298 }, 

299 ) as span: 

300 inputs = self._convert_to_internal_format(pipeline_inputs=data) 

301 priority_queue = self._fill_queue(ordered_component_names, inputs, component_visits) 

302 

303 # check if pipeline is blocked before execution 

304 self.validate_pipeline(priority_queue) 

305 

306 while True: 

307 candidate = self._get_next_runnable_component(priority_queue, component_visits) 

308 

309 # If there are no runnable components left, we can exit the loop. 

310 # In practice this rarely happens because the queue is constantly refilled even with components that 

311 # have already run. They just get a BLOCKED priority since their inputs have already been consumed. 

312 if candidate is None: 

313 break 

314 

315 priority, component_name, component = candidate 

316 

317 # If the next component is blocked, we do a check to see if the pipeline is possibly blocked and raise 

318 # a warning if it is. 

319 if priority == ComponentPriority.BLOCKED: 

320 if self._is_pipeline_possibly_blocked(current_pipeline_outputs=pipeline_outputs): 

321 # Pipeline is most likely blocked (most likely a configuration issue) so we raise a warning. 

322 self._find_components_blocking_pipeline( 

323 priority_queue=priority_queue, component_visits=component_visits, inputs=inputs 

324 ) 

325 # We always exit the loop since we cannot run the next component. 

326 break 

327 

328 if len(priority_queue) > 0 and priority in [ComponentPriority.DEFER, ComponentPriority.DEFER_LAST]: 

329 component_name, topological_sort = self._tiebreak_waiting_components( 

330 component_name=component_name, 

331 priority=priority, 

332 priority_queue=priority_queue, 

333 topological_sort=cached_topological_sort, 

334 ) 

335 cached_topological_sort = topological_sort 

336 component = self._get_component_with_graph_metadata_and_visits( 

337 component_name, component_visits[component_name] 

338 ) 

339 

340 if pipeline_snapshot: 

341 if isinstance(pipeline_snapshot.break_point, AgentBreakpoint): 

342 name_to_check = pipeline_snapshot.break_point.agent_name 

343 else: 

344 name_to_check = pipeline_snapshot.break_point.component_name 

345 is_resume = name_to_check == component_name 

346 else: 

347 is_resume = False 

348 component_inputs = self._consume_component_inputs( 

349 component_name=component_name, component=component, inputs=inputs, is_resume=is_resume 

350 ) 

351 

352 # We need to add missing defaults using default values from input sockets because the run signature 

353 # might not provide these defaults for components with inputs defined dynamically upon component 

354 # initialization 

355 component_inputs = self._add_missing_input_defaults(component_inputs, component["input_sockets"]) 

356 

357 # Scenario 1: Pipeline snapshot is provided to resume the pipeline at a specific component 

358 # Deserialize the component_inputs if they are passed in the pipeline_snapshot. 

359 # this check will prevent other component_inputs generated at runtime from being deserialized 

360 if pipeline_snapshot: 

361 if component_name in pipeline_snapshot.pipeline_state.inputs.keys(): 

362 for key, value in component_inputs.items(): 

363 component_inputs[key] = _deserialize_value_with_schema(value) 

364 

365 # If we are resuming from an AgentBreakpoint, we inject the agent_snapshot into the Agents inputs 

366 if ( 

367 isinstance(pipeline_snapshot.break_point, AgentBreakpoint) 

368 and component_name == pipeline_snapshot.break_point.agent_name 

369 ): 

370 component_inputs["snapshot"] = pipeline_snapshot.agent_snapshot 

371 component_inputs["break_point"] = None 

372 

373 # If AgentBreakpoint is provided pass onto Agent's inputs 

374 if isinstance(break_point, AgentBreakpoint) and component_name == break_point.agent_name: 

375 component_inputs["break_point"] = break_point 

376 component_inputs["snapshot_callback"] = snapshot_callback 

377 

378 try: 

379 component_outputs = self._run_component( 

380 component_name=component_name, 

381 component=component, 

382 inputs=component_inputs, # the inputs to the current component 

383 component_visits=component_visits, 

384 parent_span=span, 

385 # A break point is provided to stop the pipeline at a specific component 

386 break_point=break_point if isinstance(break_point, Breakpoint) else None, 

387 ) 

388 except (BreakpointException, PipelineRuntimeError) as error: 

389 saved_break_point: Breakpoint | AgentBreakpoint 

390 if isinstance(error, PipelineRuntimeError): 

391 saved_break_point = Breakpoint( 

392 component_name=component_name, 

393 visit_count=component_visits[component_name], 

394 snapshot_file_path=_get_output_dir("pipeline_snapshot"), 

395 ) 

396 else: 

397 saved_break_point = error.break_point 

398 

399 # Create a snapshot of the state of the pipeline before the error occurred. 

400 pipeline_snapshot = _create_pipeline_snapshot( 

401 inputs=_deepcopy_with_exceptions(inputs), 

402 component_inputs=_deepcopy_with_exceptions(component_inputs), 

403 break_point=saved_break_point, 

404 component_visits=component_visits, 

405 original_input_data=data, 

406 ordered_component_names=ordered_component_names, 

407 include_outputs_from=include_outputs_from, 

408 pipeline_outputs=pipeline_outputs, 

409 ) 

410 

411 # If the PipelineRuntimeError or BreakpointException came from an Agent component, we take the 

412 # agent snapshot and attach it to the pipeline snapshot we create here. 

413 # We also update the break_point to be an AgentBreakpoint. 

414 if error.pipeline_snapshot and error.pipeline_snapshot.agent_snapshot: 

415 pipeline_snapshot = replace( 

416 pipeline_snapshot, 

417 agent_snapshot=error.pipeline_snapshot.agent_snapshot, 

418 break_point=error.pipeline_snapshot.agent_snapshot.break_point, 

419 ) 

420 

421 # Attach the pipeline snapshot to the error before re-raising 

422 error.pipeline_snapshot = pipeline_snapshot 

423 full_file_path = _save_pipeline_snapshot( 

424 pipeline_snapshot=pipeline_snapshot, 

425 raise_on_failure=isinstance(error, BreakpointException), 

426 snapshot_callback=snapshot_callback, 

427 ) 

428 error.pipeline_snapshot_file_path = full_file_path 

429 raise error 

430 

431 # Updates global input state with component outputs and returns outputs that should go to 

432 # pipeline outputs. 

433 component_pipeline_outputs = self._write_component_outputs( 

434 component_name=component_name, 

435 component_outputs=component_outputs, 

436 inputs=inputs, 

437 receivers=cached_receivers[component_name], 

438 include_outputs_from=include_outputs_from, 

439 ) 

440 

441 if component_pipeline_outputs or component_name in include_outputs_from: 

442 pipeline_outputs[component_name] = component_pipeline_outputs 

443 if self._is_queue_stale(priority_queue): 

444 priority_queue = self._fill_queue(ordered_component_names, inputs, component_visits) 

445 

446 if isinstance(break_point, Breakpoint): 

447 logger.warning( 

448 "The given breakpoint {break_point} was never triggered. This is because:\n" 

449 "1. The provided component is not a part of the pipeline execution path.\n" 

450 "2. The component did not reach the visit count specified in the pipeline_breakpoint", 

451 pipeline_breakpoint=break_point, 

452 ) 

453 

454 return pipeline_outputs
