157 def _prepare_input(self, text: str) -> dict[str, Any]: 

158 if not isinstance(text, str): 

159 raise TypeError( 

160 "OpenAITextEmbedder expects a string as an input." 

161 "In case you want to embed a list of Documents, please use the OpenAIDocumentEmbedder." 

162 ) 

163 

164 text_to_embed = self.prefix + text + self.suffix 

165 

166 kwargs: dict[str, Any] = {"model": self.model, "input": text_to_embed, "encoding_format": "float"} 

167 if self.dimensions is not None: 

168 kwargs["dimensions"] = self.dimensions 

169 return kwargs

171 def _prepare_output(self, result: CreateEmbeddingResponse) -> dict[str, Any]: 

172 return {"embedding": result.data[0].embedding, "meta": {"model": result.model, "usage": dict(result.usage)}}

174 @component.output_types(embedding=list[float], meta=dict[str, Any]) 

175 def run(self, text: str) -> dict[str, Any]: 

176 """ 

177 Embeds a single string. 

178 

179 :param text: 

180 Text to embed. 

181 

182 :returns: 

183 A dictionary with the following keys: 

184 - `embedding`: The embedding of the input text. 

185 - `meta`: Information about the usage of the model. 

186 """ 

187 create_kwargs = self._prepare_input(text=text) 

188 response = self.client.embeddings.create(**create_kwargs) 

189 return self._prepare_output(result=response)
