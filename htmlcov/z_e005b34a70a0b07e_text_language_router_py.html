5from haystack import component, logging 

6from haystack.lazy_imports import LazyImport 

7 

8logger = logging.getLogger(__name__) 

9 

10with LazyImport("Run 'pip install langdetect'") as langdetect_import: 

11 import langdetect 

12 

13 

14@component 

15class TextLanguageRouter: 

16 """ 

17 Routes text strings to different output connections based on their language. 

18 

19 Provide a list of languages during initialization. If the document's text doesn't match any of the 

20 specified languages, the metadata value is set to "unmatched". 

21 For routing documents based on their language, use the DocumentLanguageClassifier component, 

22 followed by the MetaDataRouter. 

23 

24 ### Usage example 

25 

26 ```python 

27 from haystack import Pipeline, Document 

28 from haystack.components.routers import TextLanguageRouter 

29 from haystack.document_stores.in_memory import InMemoryDocumentStore 

30 from haystack.components.retrievers.in_memory import InMemoryBM25Retriever 

31 

32 document_store = InMemoryDocumentStore() 

33 document_store.write_documents([Document(content="Elvis Presley was an American singer and actor.")]) 

34 

35 p = Pipeline() 

36 p.add_component(instance=TextLanguageRouter(languages=["en"]), name="text_language_router") 

37 p.add_component(instance=InMemoryBM25Retriever(document_store=document_store), name="retriever") 

38 p.connect("text_language_router.en", "retriever.query") 

39 

40 result = p.run({"text_language_router": {"text": "Who was Elvis Presley?"}}) 

41 assert result["retriever"]["documents"][0].content == "Elvis Presley was an American singer and actor." 

42 

43 result = p.run({"text_language_router": {"text": "ένα ελληνικό κείμενο"}}) 

44 assert result["text_language_router"]["unmatched"] == "ένα ελληνικό κείμενο" 

45 ``` 

46 """ 

47 

48 def __init__(self, languages: list[str] | None = None) -> None: 

49 """ 

50 Initialize the TextLanguageRouter component. 

51 

52 :param languages: A list of ISO language codes. 

53 See the supported languages in [`langdetect` documentation](https://github.com/Mimino666/langdetect#languages). 

54 If not specified, defaults to ["en"]. 

55 """ 

56 langdetect_import.check() 

57 if not languages: 

58 languages = ["en"] 

59 self.languages = languages 

60 component.set_output_types(self, unmatched=str, **dict.fromkeys(languages, str)) 

61 

62 def run(self, text: str) -> dict[str, str]: 

63 """ 

64 Routes the text strings to different output connections based on their language. 

65 

66 If the document's text doesn't match any of the specified languages, the metadata value is set to "unmatched". 

67 

68 :param text: A text string to route. 

69 

70 :returns: A dictionary in which the key is the language (or `"unmatched"`), 

71 and the value is the text. 

72 

73 :raises TypeError: If the input is not a string. 

74 """ 

75 if not isinstance(text, str): 

76 msg = ( 

77 "TextLanguageRouter expects a string as input. In case you want to classify a document, please use " 

78 "the DocumentLanguageClassifier and MetaDataRouter." 

79 ) 

80 raise TypeError(msg) 

81 

82 output: dict[str, str] = {} 

83 

84 detected_language = self._detect_language(text) 

85 if detected_language in self.languages: 

86 output[detected_language] = text 

87 else: 

88 output["unmatched"] = text 

89 

90 return output 

91 

92 def _detect_language(self, text: str) -> str | None: 

93 language = None 

94 try: 

95 language = langdetect.detect(text) 

96 except langdetect.LangDetectException as exception: 

97 logger.warning("Langdetect cannot detect the language of text. Error: {error}", error=exception) 

98 # Only log the text in debug mode, as it might contain sensitive information 

99 logger.debug("Langdetect cannot detect the language of text: {text}", text=text) 

100 return language
