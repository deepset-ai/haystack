25class TransformersSimilarityRanker: 

26 """ 

27 Ranks documents based on their semantic similarity to the query. 

28 

29 It uses a pre-trained cross-encoder model from Hugging Face to embed the query and the documents. 

30 

31 Note: 

32 This component is considered legacy and will no longer receive updates. It may be deprecated in a future release, 

33 with removal following after a deprecation period. 

34 Consider using SentenceTransformersSimilarityRanker instead, which provides the same functionality along with 

35 additional features. 

36 

37 ### Usage example 

38 

39 ```python 

40 from haystack import Document 

41 from haystack.components.rankers import TransformersSimilarityRanker 

42 

43 ranker = TransformersSimilarityRanker() 

44 docs = [Document(content="Paris"), Document(content="Berlin")] 

45 query = "City in Germany" 

46 result = ranker.run(query=query, documents=docs) 

47 docs = result["documents"] 

48 print(docs[0].content) 

49 ``` 

50 """

221 @component.output_types(documents=list[Document]) 

222 def run( 

223 self, 

224 query: str, 

225 documents: list[Document], 

226 top_k: int | None = None, 

227 scale_score: bool | None = None, 

228 calibration_factor: float | None = None, 

229 score_threshold: float | None = None, 

230 ) -> dict[str, Any]: 

231 """ 

232 Returns a list of documents ranked by their similarity to the given query. 

233 

234 Before ranking, documents are deduplicated by their id, retaining only the document with the highest score 

235 if a score is present. 

236 

237 :param query: 

238 The input query to compare the documents to. 

239 :param documents: 

240 A list of documents to be ranked. 

241 :param top_k: 

242 The maximum number of documents to return. 

243 :param scale_score: 

244 If `True`, scales the raw logit predictions using a Sigmoid activation function. 

245 If `False`, disables scaling of the raw logit predictions. 

246 :param calibration_factor: 

247 Use this factor to calibrate probabilities with `sigmoid(logits * calibration_factor)`. 

248 Used only if `scale_score` is `True`. 

249 :param score_threshold: 

250 Use it to return documents only with a score above this threshold. 

251 :returns: 

252 A dictionary with the following keys: 

253 - `documents`: A list of documents closest to the query, sorted from most similar to least similar. 

254 

255 :raises ValueError: 

256 If `top_k` is not > 0. 

257 If `scale_score` is True and `calibration_factor` is not provided. 

258 """ 

259 # If a model path is provided but the model isn't loaded 

260 if self.model is None: 

261 self.warm_up() 

262 

263 if not documents: 

264 return {"documents": []} 

265 

266 top_k = top_k or self.top_k 

267 scale_score = scale_score or self.scale_score 

268 calibration_factor = calibration_factor or self.calibration_factor 

269 score_threshold = score_threshold or self.score_threshold 

270 

271 if top_k <= 0: 

272 raise ValueError(f"top_k must be > 0, but got {top_k}") 

273 

274 if scale_score and calibration_factor is None: 

275 raise ValueError( 

276 f"scale_score is True so calibration_factor must be provided, but got {calibration_factor}" 

277 ) 

278 

279 query_doc_pairs = [] 

280 deduplicated_documents = _deduplicate_documents(documents) 

281 for doc in deduplicated_documents: 

282 meta_values_to_embed = [ 

283 str(doc.meta[key]) for key in self.meta_fields_to_embed if key in doc.meta and doc.meta[key] 

284 ] 

285 text_to_embed = self.embedding_separator.join(meta_values_to_embed + [doc.content or ""]) 

286 query_doc_pairs.append([self.query_prefix + query, self.document_prefix + text_to_embed]) 

287 

288 class _Dataset(Dataset): 

289 def __init__(self, batch_encoding: Any) -> None: 

290 self.batch_encoding = batch_encoding 

291 

292 def __len__(self) -> int: 

293 return len(self.batch_encoding["input_ids"]) 

294 

295 def __getitem__(self, item: int) -> dict[str, Any]: 

296 return {key: self.batch_encoding.data[key][item] for key in self.batch_encoding.data.keys()} 

297 

298 # mypy doesn't know this is set in warm_up 

299 batch_enc = self.tokenizer(query_doc_pairs, padding=True, truncation=True, return_tensors="pt").to( 

300 self.device.first_device.to_torch() # type: ignore[union-attr] 

301 ) 

302 dataset = _Dataset(batch_enc) 

303 inp_dataloader = DataLoader(dataset, batch_size=self.batch_size, shuffle=False) 

304 

305 list_similarity_scores = [] 

306 with torch.inference_mode(): 

307 for features in inp_dataloader: 

308 model_preds = self.model(**features).logits.squeeze(dim=1) # type: ignore[misc] 

309 list_similarity_scores.extend(model_preds) 

310 similarity_scores = torch.stack(list_similarity_scores) 

311 

312 if scale_score and calibration_factor is not None: 

313 similarity_scores = torch.sigmoid(similarity_scores * calibration_factor) 

314 

315 _, sorted_indices = torch.sort(similarity_scores, descending=True) 

316 

317 list_sorted_indices = sorted_indices.cpu().tolist() 

318 list_similarity_scores = similarity_scores.cpu().tolist() 

319 ranked_docs = [] 

320 for sorted_index in list_sorted_indices: 

321 i = sorted_index 

322 ranked_docs.append(replace(deduplicated_documents[i], score=list_similarity_scores[i])) 

323 

324 if score_threshold is not None: 

325 ranked_docs = [doc for doc in ranked_docs if doc.score >= score_threshold] 

326 

327 return {"documents": ranked_docs[:top_k]}
