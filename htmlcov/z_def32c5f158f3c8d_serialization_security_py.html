5""" 

6Security primitives for pipeline deserialization. 

7 

8This module provides an allowlist mechanism that gates arbitrary imports. 

9 

10Three ways to extend the allowlist: 

11- Per-call kwarg: `Pipeline.load(..., allowed_modules=["mypkg.*"])` 

12- Process-wide programmatic API: :func:`allow_deserialization_module` 

13- Environment variable: `HAYSTACK_DESERIALIZATION_ALLOWLIST="mypkg.*,otherpkg.*"` 

14 

15The two-mode loading API (`unsafe=True`) bypasses the allowlist entirely. 

16""" 

17

28# The default allowlist covers Haystack's own packages plus a small set of standard-library type modules 

29# that are commonly referenced in serialized type annotations (e.g. `typing.List[str]`, 

30# `collections.deque`). Importing these modules has no meaningful side effects on its own. 

31DEFAULT_ALLOWED_MODULES: tuple[str, ...] = ( 

32 "haystack", 

33 "haystack_integrations", 

34 "haystack_experimental", 

35 "builtins", 

36 "typing", 

37 "collections", 

38) 

39DESERIALIZATION_ALLOWLIST_ENV_VAR = "HAYSTACK_DESERIALIZATION_ALLOWLIST" 

40 

41# `builtins` is on the default allowlist because deserialization legitimately needs builtin *types* 

42# (e.g. `builtins.str`, used in serialized type annotations and as nested `{"type": ...}` class 

43# references) and harmless builtin callables that Haystack's own serializer emits (e.g. 

44# `serialize_callable(print)` -> `"builtins.print"`). The module-granular allowlist is too coarse 

45# to separate those from dangerous members, so the two builtin-resolving contexts are gated 

46# differently: 

47# - Type / class contexts (`deserialize_type`, `import_class_by_name`) require the resolved 

48# builtin to be a `type` (see :func:`_check_builtin_is_type`). That lets every builtin type 

49# through while rejecting every builtin *function*, with no denylist to maintain. 

50# - The callable context (`deserialize_callable`) genuinely returns functions, so it instead 

51# rejects the dangerous builtin *callables* named below (see :func:`_check_not_denied_builtin`). 

52_DENIED_BUILTIN_NAMES: frozenset[str] = frozenset( 

53 { 

54 "eval", # arbitrary code execution 

55 "exec", # arbitrary code execution 

56 "compile", # arbitrary code compilation 

57 "__import__", # dynamic import of any module (gateway to os/subprocess/...) 

58 "open", # filesystem read/write 

59 "getattr", # attribute-traversal gadget (classic sandbox escape) 

60 "setattr", # arbitrary attribute mutation 

61 "delattr", # arbitrary attribute deletion 

62 "globals", # access to module namespaces 

63 "locals", # access to local namespaces 

64 "vars", # access to object/module namespaces 

65 "breakpoint", # runs the PYTHONBREAKPOINT hook 

66 "__build_class__", # dynamic class creation 

67 "type", # dynamic class creation via type(name, bases, dict) 

68 } 

69) 

70 

71# Resolve names to objects once so callers can match by identity, which also catches aliases that 

72# reach the same builtin via a different import path (e.g. `io.open is builtins.open`). 

73_DENIED_BUILTIN_OBJECTS: frozenset = frozenset([getattr(builtins, name) for name in _DENIED_BUILTIN_NAMES])

96def allow_deserialization_module(pattern: str) -> None: 

97 """ 

98 Add a module pattern to the process-wide deserialization allowlist. 

99 

100 Once added, classes from modules matching the pattern can be deserialized from YAML / dict 

101 representations until the process exits. 

102 

103 A pattern matches a module name if: 

104 - The pattern contains `*`, `?` or `[` — :mod:`fnmatch` semantics are used. 

105 - Otherwise the pattern is treated as a prefix: a module matches if it equals the pattern or 

106 is a submodule of it (i.e. starts with `pattern + "."`). A trailing `.*` is stripped 

107 before this comparison, so `"mypkg"` and `"mypkg.*"` behave identically. 

108 

109 :param pattern: 

110 The module pattern to allow. 

111 """ 

112 if pattern not in _extra_allowed_modules: 

113 _extra_allowed_modules.append(pattern)

116def _module_matches(module_name: str, pattern: str) -> bool: 

117 """Return whether `module_name` matches the given allowlist `pattern`.""" 

118 # `pkg.*` (where the part before `.*` has no other wildcards) is treated as a prefix match — 

119 # matches `pkg` and any submodule. This is the most common form, and we want it to match 

120 # the bare top-level package too (which true fnmatch wouldn't, since `pkg.*` requires a 

121 # literal `.` to follow). Patterns like `j*on.*` keep their wildcards and fall through to 

122 # fnmatch so the semantics stay consistent. 

123 if pattern.endswith(".*") and not any(c in pattern[:-2] for c in "*?["): 

124 prefix = pattern[:-2] 

125 return module_name == prefix or module_name.startswith(prefix + ".") 

126 if any(c in pattern for c in "*?["): 

127 return fnmatch.fnmatchcase(module_name, pattern) 

128 return module_name == pattern or module_name.startswith(pattern + ".")

131def _patterns_from_env() -> list[str]: 

132 raw = os.environ.get(DESERIALIZATION_ALLOWLIST_ENV_VAR, "") 

133 return [p.strip() for p in raw.split(",") if p.strip()]

136def _is_module_allowed(module_name: str) -> bool: 

137 """Return whether `module_name` is on the active deserialization allowlist.""" 

138 ctx = _get_context() 

139 if ctx.unsafe: 

140 return True 

141 patterns: list[str] = [] 

142 patterns.extend(DEFAULT_ALLOWED_MODULES) 

143 patterns.extend(_extra_allowed_modules) 

144 patterns.extend(_patterns_from_env()) 

145 patterns.extend(ctx.extra_allowed) 

146 return any(_module_matches(module_name, p) for p in patterns) 

147 

148 

149def _check_module_allowed(module_name: str) -> None: 

150 """Raise :class:`DeserializationError` if `module_name` is not on the allowlist.""" 

151 if _is_module_allowed(module_name): 

152 return 

153 raise DeserializationError( 

154 f"Refusing to deserialize a class from module '{module_name}': the module is not on the " 

155 f"trusted-module allowlist. If you trust the source of this serialized data, you can either:\n" 

156 f" - extend the allowlist for this call: " 

157 f"Pipeline.load(..., allowed_modules=['{module_name}']),\n" 

158 f" - extend it process-wide via haystack.core.serialization.allow_deserialization_module" 

159 f"('{module_name}') or the {DESERIALIZATION_ALLOWLIST_ENV_VAR} environment variable,\n" 

160 f" - or bypass the allowlist entirely: Pipeline.load(..., unsafe=True)." 

161 )

164def _is_denied_builtin(resolved: object) -> bool: 

165 """ 

166 Return whether `resolved` is one of the builtins denied for callable deserialization. 

167 

168 Matches by identity (not membership) so an unhashable resolved object never raises. 

169 """ 

170 return any(resolved is denied for denied in _DENIED_BUILTIN_OBJECTS)

173def _check_not_denied_builtin(resolved: object, handle: str) -> None: 

174 """ 

175 Reject `resolved` if it is a builtin callable that is unsafe to resolve from serialized data. 

176 

177 Used by the callable-resolution path (`deserialize_callable`). Raises 

178 :class:`DeserializationError` for the primitives in :data:`_DENIED_BUILTIN_NAMES`, which can 

179 execute code, import modules, touch the filesystem, or escape via attribute/namespace access. 

180 The block applies even though `builtins` is on the allowlist, because the allowlist is 

181 module-granular. It is intentionally bypassed in `unsafe=True` mode, which disables all 

182 deserialization safety checks by design. 

183 

184 :param resolved: 

185 The object resolved from the serialized handle. 

186 :param handle: 

187 The original serialized handle, used only for the error message. 

188 """ 

189 if _get_context().unsafe: 

190 return 

191 if _is_denied_builtin(resolved): 

192 name = getattr(resolved, "__name__", str(resolved)) 

193 raise DeserializationError( 

194 f"Refusing to deserialize '{handle}': it resolves to the builtin '{name}', which is " 

195 f"blocked because it can be used to execute code, import modules, access the " 

196 f"filesystem, or escape via attribute access. If you trust the source of this data, " 

197 f"load it with unsafe=True to bypass deserialization safety checks." 

198 )

201def _check_builtin_is_type(resolved: object, handle: str) -> None: 

202 """ 

203 Reject a `builtins` member resolved in a type/class context that is not a `type`. 

204 

205 Used by `deserialize_type` and `import_class_by_name`, which resolve type annotations and class 

206 references — always classes. Requiring the resolved `builtins` member to be a `type` lets every 

207 builtin type through (e.g. `str`, `memoryview`) while rejecting every builtin *function* (e.g. 

208 `eval`, `exec`, `getattr`), with no denylist to maintain. Bypassed in `unsafe=True` mode. 

209 

210 :param resolved: 

211 The object resolved from the serialized handle. 

212 :param handle: 

213 The original serialized handle, used only for the error message. 

214 """ 

215 if _get_context().unsafe: 

216 return 

217 if not isinstance(resolved, type): 

218 raise DeserializationError( 

219 f"Refusing to deserialize '{handle}': it resolves to a builtin that is not a type and " 

220 f"cannot be used as a type annotation or class reference. If you trust the source of " 

221 f"this data, load it with unsafe=True to bypass deserialization safety checks." 

222 )

225@contextmanager 

226def _deserialization_context(allowed_modules: Iterable[str] | None = None, unsafe: bool = False) -> Iterator[None]: 

227 """ 

228 Context manager that activates a per-call deserialization context. 

229 

230 Patterns from `allowed_modules` are appended to the parent context's patterns, and `unsafe` 

231 is OR-ed with the parent's `unsafe` flag — so this never narrows the active permissions. 

232 The previous context is restored on exit. 

233 """ 

234 parent = _get_context() 

235 extra = parent.extra_allowed + (tuple(allowed_modules) if allowed_modules else ()) 

236 merged_unsafe = parent.unsafe or unsafe 

237 token = _current_context.set(_DeserializationContext(extra_allowed=extra, unsafe=merged_unsafe)) 

238 try: 

239 yield 

240 finally: 

241 _current_context.reset(token)
