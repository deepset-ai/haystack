23def generate_color_variations(n: int, base_color: str | None = "#3498DB", variation_range: float = 0.4) -> list[str]: 

24 """ 

25 Generate n different variations of a base color. 

26 

27 :param n: Number of variations to generate 

28 :param base_color: Hex color code, default is a shade of blue (#3498DB) 

29 :param variation_range: Range for varying brightness and saturation (0-1) 

30 

31 :returns: 

32 list: List of hex color codes representing variations of the base color 

33 """ 

34 # convert hex to RGB 

35 base_color = base_color.lstrip("#") # type:ignore 

36 r = int(base_color[0:2], 16) / 255.0 

37 g = int(base_color[2:4], 16) / 255.0 

38 b = int(base_color[4:6], 16) / 255.0 

39 

40 # convert RGB to HSV (Hue, Saturation, Value) 

41 h, s, v = colorsys.rgb_to_hsv(r, g, b) 

42 

43 variations = [] 

44 for _ in range(n): 

45 # vary saturation and brightness within the specified range 

46 new_s = max(0, min(1, s + random.uniform(-variation_range, variation_range))) 

47 new_v = max(0, min(1, v + random.uniform(-variation_range, variation_range))) 

48 

49 # keep hue the same for color consistency 

50 new_h = h 

51 

52 # Convert back to RGB and then to hex 

53 new_r, new_g, new_b = colorsys.hsv_to_rgb(new_h, new_s, new_v) 

54 hex_color = f"#{int(new_r * 255):02x}{int(new_g * 255):02x}{int(new_b * 255):02x}" 

55 

56 variations.append(hex_color) 

57 

58 return variations

61def _prepare_for_drawing(graph: networkx.MultiDiGraph) -> networkx.MultiDiGraph: 

62 """ 

63 Add some extra nodes to show the inputs and outputs of the pipeline. 

64 

65 Also adds labels to edges. 

66 """ 

67 # Label the edges 

68 for inp, outp, key, data in graph.edges(keys=True, data=True): 

69 data["label"] = ( 

70 f"{data['from_socket'].name} -> {data['to_socket'].name}{' (opt.)' if not data['mandatory'] else ''}" 

71 ) 

72 graph.add_edge(inp, outp, key=key, **data) 

73 

74 # Add inputs fake node 

75 graph.add_node("input") 

76 for node, in_sockets in find_pipeline_inputs(graph).items(): 

77 for in_socket in in_sockets: 

78 if not in_socket.senders and in_socket.is_mandatory: 

79 # If this socket has no sender it could be a socket that receives input 

80 # directly when running the Pipeline. We can't know that for sure, in doubt 

81 # we draw it as receiving input directly. 

82 graph.add_edge("input", node, label=in_socket.name, conn_type=_type_name(in_socket.type)) 

83 

84 # Add outputs fake node 

85 graph.add_node("output") 

86 for node, out_sockets in find_pipeline_outputs(graph).items(): 

87 for out_socket in out_sockets: 

88 graph.add_edge(node, "output", label=out_socket.name, conn_type=_type_name(out_socket.type)) 

89 

90 return graph

109def _validate_mermaid_params(params: dict[str, Any]) -> None: 

110 """ 

111 Validates and sets default values for Mermaid parameters. 

112 

113 :param params: 

114 Dictionary of customization parameters to modify the output. Refer to Mermaid documentation for more details. 

115 Supported keys: 

116 - format: Output format ('img', 'svg', or 'pdf'). Default: 'img'. 

117 - type: Image type for /img endpoint ('jpeg', 'png', 'webp'). Default: 'png'. 

118 - theme: Mermaid theme ('default', 'neutral', 'dark', 'forest'). Default: 'neutral'. 

119 - bgColor: Background color in hexadecimal (e.g., 'FFFFFF') or named format (e.g., '!white'). 

120 - width: Width of the output image (integer). 

121 - height: Height of the output image (integer). 

122 - scale: Scaling factor (1–3). Only applicable if 'width' or 'height' is specified. 

123 - fit: Whether to fit the diagram size to the page (PDF only, boolean). 

124 - paper: Paper size for PDFs (e.g., 'a4', 'a3'). Ignored if 'fit' is true. 

125 - landscape: Landscape orientation for PDFs (boolean). Ignored if 'fit' is true. 

126 

127 :raises ValueError: 

128 If any parameter is invalid or does not match the expected format. 

129 """ 

130 valid_img_types = {"jpeg", "png", "webp"} 

131 valid_themes = {"default", "neutral", "dark", "forest"} 

132 valid_formats = {"img", "svg", "pdf"} 

133 

134 params.setdefault("format", "img") 

135 params.setdefault("type", "png") 

136 params.setdefault("theme", "neutral") 

137 

138 if params["format"] not in valid_formats: 

139 raise ValueError(f"Invalid image format: {params['format']}. Valid options are: {valid_formats}.") 

140 

141 if params["format"] == "img" and params["type"] not in valid_img_types: 

142 raise ValueError(f"Invalid image type: {params['type']}. Valid options are: {valid_img_types}.") 

143 

144 if params["theme"] not in valid_themes: 

145 raise ValueError(f"Invalid theme: {params['theme']}. Valid options are: {valid_themes}.") 

146 

147 if "width" in params and not isinstance(params["width"], int): 

148 raise ValueError("Width must be an integer.") 

149 if "height" in params and not isinstance(params["height"], int): 

150 raise ValueError("Height must be an integer.") 

151 

152 if "scale" in params and not 1 <= params["scale"] <= 3: 

153 raise ValueError("Scale must be a number between 1 and 3.") 

154 if "scale" in params and not ("width" in params or "height" in params): 

155 raise ValueError("Scale is only allowed when width or height is set.") 

156 

157 if "bgColor" in params and not isinstance(params["bgColor"], str): 

158 raise ValueError("Background color must be a string.") 

159 

160 # PDF specific parameters 

161 if params["format"] == "pdf": 

162 if "fit" in params and not isinstance(params["fit"], bool): 

163 raise ValueError("Fit must be a boolean.") 

164 if "paper" in params and not isinstance(params["paper"], str): 

165 raise ValueError("Paper size must be a string (e.g., 'a4', 'a3').") 

166 if "landscape" in params and not isinstance(params["landscape"], bool): 

167 raise ValueError("Landscape must be a boolean.") 

168 if "fit" in params and ("paper" in params or "landscape" in params): 

169 logger.warning("`fit` overrides `paper` and `landscape` for PDFs. Ignoring `paper` and `landscape`.")

172def _to_mermaid_image( 

173 graph: networkx.MultiDiGraph, 

174 server_url: str = "https://mermaid.ink", 

175 params: dict | None = None, 

176 timeout: int = 30, 

177 super_component_mapping: dict[str, str] | None = None, 

178) -> bytes: 

179 """ 

180 Renders a pipeline using a Mermaid server. 

181 

182 :param graph: 

183 The graph to render as a Mermaid pipeline. 

184 :param server_url: 

185 Base URL of the Mermaid server (default: 'https://mermaid.ink'). 

186 :param params: 

187 Dictionary of customization parameters. See `validate_mermaid_params` for valid keys. 

188 :param timeout: 

189 Timeout in seconds for the request to the Mermaid server. 

190 :returns: 

191 The image, SVG, or PDF data returned by the Mermaid server as bytes. 

192 :raises ValueError: 

193 If any parameter is invalid or does not match the expected format. 

194 :raises PipelineDrawingError: 

195 If there is an issue connecting to the Mermaid server or the server returns an error. 

196 """ 

197 

198 if params is None: 

199 params = {} 

200 

201 _validate_mermaid_params(params) 

202 

203 theme = params.get("theme") 

204 init_params = json.dumps({"theme": theme}) 

205 

206 # Copy the graph to avoid modifying the original 

207 graph_styled = _to_mermaid_text(graph.copy(), init_params, super_component_mapping) 

208 json_string = json.dumps({"code": graph_styled}) 

209 

210 # Compress the JSON string with zlib (RFC 1950) 

211 compressor = zlib.compressobj(level=9, wbits=15) 

212 compressed_data = compressor.compress(json_string.encode("utf-8")) + compressor.flush() 

213 compressed_url_safe_base64 = base64.urlsafe_b64encode(compressed_data).decode("utf-8").strip() 

214 

215 # Determine the correct endpoint 

216 endpoint_format = params.get("format", "img") # Default to /img endpoint 

217 if endpoint_format not in {"img", "svg", "pdf"}: 

218 raise ValueError(f"Invalid format: {endpoint_format}. Valid options are 'img', 'svg', or 'pdf'.") 

219 

220 # Construct the URL without query parameters 

221 url = f"{server_url}/{endpoint_format}/pako:{compressed_url_safe_base64}" 

222 

223 # Add query parameters adhering to mermaid.ink documentation 

224 query_params = [] 

225 for key, value in params.items(): 

226 if key not in {"theme", "format"}: # Exclude theme (handled in init_params) and format (endpoint-specific) 

227 if value is True: 

228 query_params.append(f"{key}") 

229 else: 

230 query_params.append(f"{key}={value}") 

231 

232 if query_params: 

233 url += "?" + "&".join(query_params) 

234 

235 logger.debug("Rendering graph at {url}", url=url) 

236 try: 

237 resp = httpx.get(url, timeout=timeout) 

238 if resp.status_code >= 400: 

239 logger.warning( 

240 "Failed to draw the pipeline: {server_url} returned status {status_code}", 

241 server_url=server_url, 

242 status_code=resp.status_code, 

243 ) 

244 logger.info("Exact URL requested: {url}", url=url) 

245 logger.warning("No pipeline diagram will be saved.") 

246 resp.raise_for_status() 

247 

248 except Exception as exc: 

249 logger.warning( 

250 "Failed to draw the pipeline: could not connect to {server_url} ({error})", server_url=server_url, error=exc 

251 ) 

252 logger.info("Exact URL requested: {url}", url=url) 

253 logger.warning("No pipeline diagram will be saved.") 

254 raise PipelineDrawingError(f"There was an issue with {server_url}, see the stacktrace for details.") from exc 

255 

256 return resp.content

259def _to_mermaid_text( 

260 graph: networkx.MultiDiGraph, init_params: str | dict, super_component_mapping: dict[str, str] | None = None 

261) -> str: 

262 """ 

263 Converts a Networkx graph into Mermaid syntax. 

264 

265 The output of this function can be used in the documentation with `mermaid` codeblocks and will be 

266 automatically rendered. 

267 

268 :param graph: The graph to convert to Mermaid syntax 

269 :param init_params: Initialization parameters for Mermaid 

270 :param super_component_mapping: Mapping of component names to super component names 

271 """ 

272 # Copy the graph to avoid modifying the original 

273 graph = _prepare_for_drawing(graph.copy()) 

274 sockets = { 

275 comp: "".join( 

276 [ 

277 f"<li>{name} ({_type_name(socket.type)})</li>" 

278 for name, socket in data.get("input_sockets", {}).items() 

279 if (not socket.is_mandatory and not socket.senders) or socket.is_variadic 

280 ] 

281 ) 

282 for comp, data in graph.nodes(data=True) 

283 } 

284 optional_inputs = { 

285 comp: f"<br><br>Optional inputs:<ul style='text-align:left;'>{sockets}</ul>" if sockets else "" 

286 for comp, sockets in sockets.items() 

287 } 

288 

289 # Create node definitions 

290 states = {} 

291 super_component_components = super_component_mapping.keys() if super_component_mapping else {} 

292 

293 # color variations for super components 

294 super_component_colors = {} 

295 if super_component_components: 

296 unique_super_components = set(super_component_mapping.values()) # type:ignore 

297 color_variations = generate_color_variations(n=len(unique_super_components)) 

298 super_component_colors = dict(zip(unique_super_components, color_variations, strict=True)) 

299 

300 # Generate style definitions for each super component 

301 style_definitions = [] 

302 for super_comp, color in super_component_colors.items(): 

303 style_definitions.append(f"classDef {super_comp} fill:{color},color:white;") 

304 

305 for comp, data in graph.nodes(data=True): 

306 if comp in ["input", "output"]: 

307 continue 

308 

309 # styling based on whether the component is a SuperComponent 

310 if comp in super_component_components: 

311 super_component_name = super_component_mapping[comp] # type:ignore 

312 style = super_component_name 

313 else: 

314 style = "component" 

315 node_def = f'{comp}["<b>{comp}</b><br><small><i>{type(data["instance"]).__name__}{optional_inputs[comp]}</i></small>"]:::{style}' # noqa: E501 

316 states[comp] = node_def 

317 

318 connections_list = [] 

319 for from_comp, to_comp, conn_data in graph.edges(data=True): 

320 if from_comp != "input" and to_comp != "output": 

321 arrowtail = ARROWTAIL_MANDATORY if conn_data["mandatory"] else ARROWTAIL_OPTIONAL 

322 arrowhead = ARROWHEAD_MANDATORY if conn_data["mandatory"] else ARROWHEAD_OPTIONAL 

323 label = f'"{conn_data["label"]}<br><small><i>{conn_data["conn_type"]}</i></small>"' 

324 conn_string = f"{states[from_comp]} {arrowtail} {label} {arrowhead} {states[to_comp]}" 

325 connections_list.append(conn_string) 

326 

327 input_connections = [ 

328 f'i{{&ast;}}--"{conn_data["label"]}<br><small><i>{conn_data["conn_type"]}</i></small>"--> {states[to_comp]}' 

329 for _, to_comp, conn_data in graph.out_edges("input", data=True) 

330 ] 

331 output_connections = [ 

332 f'{states[from_comp]}--"{conn_data["label"]}<br><small><i>{conn_data["conn_type"]}</i></small>"--> o{{&ast;}}' 

333 for from_comp, _, conn_data in graph.in_edges("output", data=True) 

334 ] 

335 connections = "\n".join(connections_list + input_connections + output_connections) 

336 

337 # Create legend 

338 legend_nodes = [] 

339 if super_component_colors: 

340 legend_nodes.append("subgraph Legend") 

341 for super_comp in super_component_colors: 

342 legend_id = f"legend_{super_comp}" 

343 legend_nodes.append(f'{legend_id}["{super_comp}"]:::{super_comp}') 

344 legend_nodes.append("end") 

345 connections += "\n" + "\n".join(legend_nodes) 

346 

347 # Add style definitions to the template 

348 graph_styled = MERMAID_STYLED_TEMPLATE.format( 

349 params=init_params, connections=connections, style_definitions="\n".join(style_definitions) 

350 ) 

351 logger.debug("Mermaid diagram:\n{diagram}", diagram=graph_styled) 

352 

353 return graph_styled
