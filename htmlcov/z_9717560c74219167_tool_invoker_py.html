82class ToolInvoker: 

83 """ 

84 Invokes tools based on prepared tool calls and returns the results as a list of ChatMessage objects. 

85 

86 Also handles reading/writing from a shared `State`. 

87 At initialization, the ToolInvoker component is provided with a list of available tools. 

88 At runtime, the component processes a list of ChatMessage object containing tool calls 

89 and invokes the corresponding tools. 

90 The results of the tool invocations are returned as a list of ChatMessage objects with tool role. 

91 

92 Usage example: 

93 ```python 

94 from haystack.dataclasses import ChatMessage, ToolCall 

95 from haystack.tools import Tool 

96 from haystack.components.tools import ToolInvoker 

97 

98 # Tool definition 

99 def dummy_weather_function(city: str): 

100 return f"The weather in {city} is 20 degrees." 

101 

102 parameters = {"type": "object", 

103 "properties": {"city": {"type": "string"}}, 

104 "required": ["city"]} 

105 

106 tool = Tool(name="weather_tool", 

107 description="A tool to get the weather", 

108 function=dummy_weather_function, 

109 parameters=parameters) 

110 

111 # Usually, the ChatMessage with tool_calls is generated by a Language Model 

112 # Here, we create it manually for demonstration purposes 

113 tool_call = ToolCall( 

114 tool_name="weather_tool", 

115 arguments={"city": "Berlin"} 

116 ) 

117 message = ChatMessage.from_assistant(tool_calls=[tool_call]) 

118 

119 # ToolInvoker initialization and run 

120 invoker = ToolInvoker(tools=[tool]) 

121 result = invoker.run(messages=[message]) 

122 

123 print(result) 

124 ```

127 # >> { 

128 # >> 'tool_messages': [ 

129 # >> ChatMessage( 

130 # >> _role=<ChatRole.TOOL: 'tool'>, 

131 # >> _content=[ 

132 # >> ToolCallResult( 

133 # >> result='"The weather in Berlin is 20 degrees."', 

134 # >> origin=ToolCall( 

135 # >> tool_name='weather_tool', 

136 # >> arguments={'city': 'Berlin'}, 

137 # >> id=None

232 @staticmethod 

233 def _make_context_bound_invoke(tool_to_invoke: Tool, final_args: dict[str, Any]) -> Callable[[], Any]: 

234 """ 

235 Create a zero-arg callable that invokes the tool under the caller's contextvars Context. 

236 

237 We copy and use contextvars to preserve the caller’s ambient execution context (for example the active 

238 tracing Span) across thread boundaries. Python’s contextvars do not automatically propagate to worker 

239 threads (or to threadpool tasks spawned via run_in_executor), so without intervention nested tool calls 

240 would lose their parent trace/span and appear as separate roots. By capturing the current Context in the 

241 caller thread and invoking the tool under ctx.run(...) inside the executor, we ensure proper span parentage, 

242 consistent tagging, and reliable log/trace correlation in both sync and async paths. The callable returns 

243 ToolInvocationError instead of raising so parallel execution can collect failures. 

244 """ 

245 ctx = contextvars.copy_context() 

246 

247 def _runner() -> Any: 

248 try: 

249 return ctx.run(partial(tool_to_invoke.invoke, **final_args)) 

250 except ToolInvocationError as e: 

251 return e 

252 

253 return _runner

278 def _default_output_to_string_handler(self, result: Any) -> str: 

279 """ 

280 Default handler for converting a tool result to a string. 

281 

282 :param result: The tool result to convert to a string. 

283 :returns: The converted tool result as a string. 

284 """ 

285 # We iterate through all items in result and call to_dict() if present 

286 # Relevant for a few reasons: 

287 # - If using convert_result_to_json_string we'd rather convert Haystack objects to JSON serializable dicts 

288 # - If using default str() we prefer converting Haystack objects to dicts rather than relying on the 

289 # __repr__ method 

290 serializable = _serializable_value(value=result, use_placeholders=False) 

291 

292 if self.convert_result_to_json_string: 

293 try: 

294 # We disable ensure_ascii so special chars like emojis are not converted 

295 str_result = json.dumps(serializable, ensure_ascii=False) 

296 except Exception as error: 

297 # If the result is not JSON serializable, we fall back to str 

298 logger.warning( 

299 "Tool result is not JSON serializable. Falling back to str conversion. " 

300 "Result: {result}\nError: {err}", 

301 result=result, 

302 err=error, 

303 ) 

304 str_result = str(result) 

305 return str_result 

306 

307 return str(serializable)

309 def _process_output(self, config: dict[str, Any], result: Any, tool_call: ToolCall) -> Any: 

310 """ 

311 Processes a tool result based on the provided configuration. 

312 

313 :param config: Configuration dictionary that may contain "source", "handler", and "raw_result" keys. 

314 :param result: The tool result to process. 

315 :param tool_call: The ToolCall object for error reporting. 

316 :returns: The processed tool result. 

317 """ 

318 source_key = config.get("source") 

319 # If a source key is provided, we extract the result from the source key 

320 value = result.get(source_key) if source_key is not None and isinstance(result, dict) else result 

321 

322 handler = config.get("handler") 

323 raw_result = config.get("raw_result", False) 

324 

325 if handler is None: 

326 if raw_result: 

327 return value 

328 handler = self._default_output_to_string_handler 

329 

330 try: 

331 return handler(value) 

332 except Exception as e: 

333 if raw_result: 

334 raise ResultConversionError(tool_call.tool_name, handler.__name__, e) from e 

335 raise StringConversionError(tool_call.tool_name, handler.__name__, e) from e

337 def _prepare_tool_result_message(self, result: Any, tool_call: ToolCall, tool_to_invoke: Tool) -> ChatMessage: 

338 """ 

339 Prepares a ChatMessage with the result of a tool invocation. 

340 

341 :param result: 

342 The tool result. 

343 :param tool_call: 

344 The ToolCall object containing the tool name and arguments. 

345 :param tool_to_invoke: 

346 The Tool object that was invoked. 

347 :returns: 

348 A ChatMessage object containing the tool result. 

349 :raises 

350 StringConversionError: If the conversion to string of the tool output fails and `raise_on_failure` is True. 

351 ResultConversionError: If the conversion to result of the tool output fails and `raise_on_failure` is True. 

352 """ 

353 outputs_config = tool_to_invoke.outputs_to_string or {} 

354 try: 

355 # Root level single output configuration 

356 if ( 

357 not outputs_config 

358 or "source" in outputs_config 

359 or "handler" in outputs_config 

360 or "raw_result" in outputs_config 

361 ): 

362 tool_result = self._process_output(outputs_config, result, tool_call) 

363 return ChatMessage.from_tool(tool_result=tool_result, origin=tool_call) 

364 

365 # Multiple outputs configuration 

366 tool_result_dict = {} 

367 for output_key, config in outputs_config.items(): 

368 # For multiple outputs, we don't support raw_result and always convert to string 

369 tool_result_dict[output_key] = self._process_output({**config, "raw_result": False}, result, tool_call) 

370 

371 tool_result_str = self._default_output_to_string_handler(tool_result_dict) 

372 return ChatMessage.from_tool(tool_result=tool_result_str, origin=tool_call) 

373 except (StringConversionError, ResultConversionError) as e: 

374 if self.raise_on_failure: 

375 raise e 

376 logger.exception("{error_exception}", error_exception=e) 

377 return ChatMessage.from_tool(tool_result=str(e), origin=tool_call, error=True)

405 @staticmethod 

406 def _inject_state_args(tool: Tool, llm_args: dict[str, Any], state: State) -> dict[str, Any]: 

407 """ 

408 Combine LLM-provided arguments (llm_args) with state-based arguments. 

409 

410 Tool arguments take precedence in the following order: 

411 - LLM overrides state if the same param is present in both 

412 - local tool.inputs_from_state mappings (if any) 

413 - function signature name matching 

414 """ 

415 final_args = dict(llm_args) # start with LLM-provided 

416 func_params = ToolInvoker._get_func_params(tool).keys() 

417 

418 # Determine the source of parameter mappings (explicit tool inputs or direct function parameters) 

419 # Typically, a "Tool" might have .inputs_from_state = {"state_key": "tool_param_name"} 

420 if hasattr(tool, "inputs_from_state") and isinstance(tool.inputs_from_state, dict): 

421 param_mappings = tool.inputs_from_state 

422 else: 

423 param_mappings = {name: name for name in func_params} 

424 

425 # Populate final_args from state if not provided by LLM 

426 for state_key, param_name in param_mappings.items(): 

427 if param_name not in final_args and state.has(state_key): 

428 final_args[param_name] = state.get(state_key) 

429 

430 # Inject the live State object for any parameter annotated as State or Optional[State] 

431 for param_name, param_type in ToolInvoker._get_func_params(tool).items(): 

432 if _unwrap_optional(param_type) is State: 

433 final_args[param_name] = state 

434 

435 return final_args

437 @staticmethod 

438 def _merge_tool_outputs(tool: Tool, result: Any, state: State) -> None: 

439 """ 

440 Merges the tool result into the State. 

441 

442 This method processes the output of a tool execution and integrates it into the global state. 

443 It also determines what message, if any, should be returned for further processing in a conversation. 

444 

445 Processing Steps: 

446 1. If `result` is not a dictionary, nothing is stored into state and the full `result` is returned. 

447 2. If the `tool` does not define an `outputs_to_state` mapping nothing is stored into state. 

448 The return value in this case is simply the full `result` dictionary. 

449 3. If the tool defines an `outputs_to_state` mapping (a dictionary describing how the tool's output should be 

450 processed), the method delegates to `_handle_tool_outputs` to process the output accordingly. 

451 This allows certain fields in `result` to be mapped explicitly to state fields or formatted using custom 

452 handlers. 

453 

454 :param tool: Tool instance containing optional `outputs_to_state` mapping to guide result processing. 

455 :param result: The output from tool execution. Can be a dictionary, or any other type. 

456 :param state: The global State object to which results should be merged. 

457 :returns: Three possible values: 

458 - A string message for conversation 

459 - The merged result dictionary 

460 - Or the raw result if not a dictionary 

461 """ 

462 # If result is not a dictionary we exit 

463 if not isinstance(result, dict): 

464 return 

465 

466 # If there is no specific `outputs_to_state` mapping, we exit 

467 if not hasattr(tool, "outputs_to_state") or not isinstance(tool.outputs_to_state, dict): 

468 return 

469 

470 # Update the state with the tool outputs 

471 for state_key, config in tool.outputs_to_state.items(): 

472 # Get the source key from the output config, otherwise use the entire result 

473 source_key = config.get("source", None) 

474 

475 # If a source key is specified but absent from the result, skip this mapping to avoid passing a None value 

476 # to state. This can be a common scenario with PipelineTool wrapping a pipeline that has conditional 

477 # branches where not all outputs are always produced even if defined in outputs_to_state. 

478 if source_key and source_key not in result: 

479 continue 

480 

481 output_value = result.get(source_key) if source_key else result 

482 

483 # Merge other outputs into the state 

484 state.set(state_key, output_value, handler_override=config.get("handler"))

497 def _prepare_tool_call_params( 

498 self, 

499 *, 

500 messages_with_tool_calls: list[ChatMessage], 

501 state: State, 

502 streaming_callback: StreamingCallbackT | None, 

503 enable_streaming_passthrough: bool, 

504 tools_with_names: dict[str, Tool], 

505 ) -> tuple[list[ToolCall], list[dict[str, Any]], list[ChatMessage]]: 

506 """ 

507 Prepare tool call parameters for execution and collect any error messages. 

508 

509 :param messages_with_tool_calls: Messages containing tool calls to process 

510 :param state: The current state for argument injection 

511 :param streaming_callback: Optional streaming callback to inject 

512 :param enable_streaming_passthrough: Whether to pass streaming callback to tools 

513 :returns: Tuple of (tool_calls, tool_call_params, error_messages) 

514 """ 

515 tool_call_params = [] 

516 error_messages = [] 

517 tool_calls = [] 

518 

519 for message in messages_with_tool_calls: 

520 for tool_call in message.tool_calls: 

521 tool_name = tool_call.tool_name 

522 

523 # Check if the tool is available, otherwise return an error message 

524 if tool_name not in tools_with_names: 

525 error = ToolNotFoundException(tool_name, list(tools_with_names.keys())) 

526 if self.raise_on_failure: 

527 raise error 

528 logger.error("{error_exception}", error_exception=error) 

529 error_messages.append(ChatMessage.from_tool(tool_result=str(error), origin=tool_call, error=True)) 

530 continue 

531 

532 tool_to_invoke = tools_with_names[tool_name] 

533 

534 # Combine user + state inputs 

535 llm_args = tool_call.arguments.copy() 

536 final_args = self._inject_state_args(tool_to_invoke, llm_args, state) 

537 

538 # Check whether to inject streaming_callback 

539 if ( 

540 enable_streaming_passthrough 

541 and streaming_callback is not None 

542 and "streaming_callback" not in final_args 

543 and "streaming_callback" in self._get_func_params(tool_to_invoke).keys() 

544 ): 

545 final_args["streaming_callback"] = streaming_callback 

546 

547 tool_call_params.append({"tool_to_invoke": tool_to_invoke, "final_args": final_args}) 

548 tool_calls.append(tool_call) 

549 

550 return tool_calls, tool_call_params, error_messages

565 @component.output_types(tool_messages=list[ChatMessage], state=State) 

566 def run( 

567 self, 

568 messages: list[ChatMessage], 

569 state: State | None = None, 

570 streaming_callback: StreamingCallbackT | None = None, 

571 *, 

572 enable_streaming_callback_passthrough: bool | None = None, 

573 tools: ToolsType | None = None, 

574 ) -> dict[str, Any]: 

575 """ 

576 Processes ChatMessage objects containing tool calls and invokes the corresponding tools, if available. 

577 

578 :param messages: 

579 A list of ChatMessage objects. 

580 :param state: The runtime state that should be used by the tools. 

581 :param streaming_callback: A callback function that will be called to emit tool results. 

582 Note that the result is only emitted once it becomes available — it is not 

583 streamed incrementally in real time. 

584 :param enable_streaming_callback_passthrough: 

585 If True, the `streaming_callback` will be passed to the tool invocation if the tool supports it. 

586 This allows tools to stream their results back to the client. 

587 Note that this requires the tool to have a `streaming_callback` parameter in its `invoke` method signature. 

588 If False, the `streaming_callback` will not be passed to the tool invocation. 

589 If None, the value from the constructor will be used. 

590 :param tools: A list of Tool and/or Toolset objects, or a single Toolset for which the model can prepare calls. 

591 If set, it will override the `tools` parameter provided during initialization. 

592 :returns: 

593 A dictionary with the key `tool_messages` containing a list of ChatMessage objects with tool role. 

594 Each ChatMessage objects wraps the result of a tool invocation.
