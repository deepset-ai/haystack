13def serialize_callable(callable_handle: Callable) -> str: 

14 """ 

15 Serializes a callable to its full path. 

16 

17 :param callable_handle: The callable to serialize 

18 :return: The full path of the callable 

19 """ 

20 try: 

21 full_arg_spec = inspect.getfullargspec(callable_handle) 

22 is_instance_method = bool(full_arg_spec.args and full_arg_spec.args[0] == "self") 

23 except TypeError: 

24 is_instance_method = False 

25 if is_instance_method: 

26 raise SerializationError("Serialization of instance methods is not supported.") 

27 

28 # __qualname__ contains the fully qualified path we need for classmethods and staticmethods 

29 qualname = getattr(callable_handle, "__qualname__", "") 

30 if "<lambda>" in qualname: 

31 raise SerializationError("Serialization of lambdas is not supported.") 

32 if "<locals>" in qualname: 

33 raise SerializationError("Serialization of nested functions is not supported.") 

34 

35 name = qualname or callable_handle.__name__ 

36 

37 # Get the full package path of the function 

38 module = inspect.getmodule(callable_handle) 

39 if module is not None: 

40 full_path = f"{module.__name__}.{name}" 

41 else: 

42 full_path = name 

43 return full_path

46def deserialize_callable(callable_handle: str) -> Callable: 

47 """ 

48 Deserializes a callable given its full import path as a string. 

49 

50 :param callable_handle: The full path of the callable_handle 

51 :return: The callable 

52 :raises DeserializationError: If the callable cannot be found 

53 """ 

54 # Import here to avoid circular imports 

55 from haystack.tools.tool import Tool 

56 

57 parts = callable_handle.split(".") 

58 

59 for i in range(len(parts), 0, -1): 

60 module_name = ".".join(parts[:i]) 

61 try: 

62 mod: Any = thread_safe_import(module_name) 

63 except Exception: 

64 # keep reducing i until we find a valid module import 

65 continue 

66 

67 attr_value = mod 

68 for part in parts[i:]: 

69 try: 

70 attr_value = getattr(attr_value, part) 

71 except AttributeError as e: 

72 raise DeserializationError(f"Could not find attribute '{part}' in {attr_value.__name__}") from e 

73 

74 # when the attribute is a classmethod, we need the underlying function 

75 if isinstance(attr_value, (classmethod, staticmethod)): 

76 attr_value = attr_value.__func__ 

77 

78 # Handle the case where @tool decorator replaced the function with a Tool object 

79 if isinstance(attr_value, Tool): 

80 attr_value = attr_value.function 

81 

82 if not callable(attr_value): 

83 raise DeserializationError(f"The final attribute is not callable: {attr_value}") 

84 

85 return attr_value 

86 

87 # Fallback if we never find anything 

88 raise DeserializationError(f"Could not import '{callable_handle}' as a module or callable.")
