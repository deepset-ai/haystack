24def serialize_callable(callable_handle: Callable) -> str: 

25 """ 

26 Serializes a callable to its full path. 

27 

28 :param callable_handle: The callable to serialize 

29 :return: The full path of the callable 

30 """ 

31 try: 

32 full_arg_spec = inspect.getfullargspec(callable_handle) 

33 is_instance_method = bool(full_arg_spec.args and full_arg_spec.args[0] == "self") 

34 except TypeError: 

35 is_instance_method = False 

36 if is_instance_method: 

37 raise SerializationError("Serialization of instance methods is not supported.") 

38 

39 # __qualname__ contains the fully qualified path we need for classmethods and staticmethods 

40 qualname = getattr(callable_handle, "__qualname__", "") 

41 if "<lambda>" in qualname: 

42 raise SerializationError("Serialization of lambdas is not supported.") 

43 if "<locals>" in qualname: 

44 raise SerializationError("Serialization of nested functions is not supported.") 

45 

46 name = qualname or callable_handle.__name__ 

47 

48 # Get the full package path of the function 

49 module = inspect.getmodule(callable_handle) 

50 if module is not None: 

51 full_path = f"{module.__name__}.{name}" 

52 else: 

53 full_path = name 

54 

55 # Serialization succeeds, but a denied builtin (e.g. `eval`) won't reload without `unsafe=True`. 

56 if _is_denied_builtin(callable_handle): 

57 logger.warning( 

58 "Serialized callable '{full_path}' is a builtin that is blocked during deserialization; " 

59 "the resulting pipeline will only be loadable with unsafe=True.", 

60 full_path=full_path, 

61 ) 

62 

63 return full_path

66def deserialize_callable(callable_handle: str) -> Callable: 

67 """ 

68 Deserializes a callable given its full import path as a string. 

69 

70 Every module path tried during resolution is checked against the 

71 deserialization allowlist (see `haystack.core.serialization_security`). Callables in modules 

72 outside the allowlist are rejected with a `DeserializationError` before any import is 

73 attempted. To allow a third-party module, extend the allowlist via 

74 `Pipeline.load(..., allowed_modules=[...])`, `allow_deserialization_module(...)`, or the 

75 `HAYSTACK_DESERIALIZATION_ALLOWLIST` environment variable. 

76 

77 :param callable_handle: The full path of the callable_handle 

78 :return: The callable 

79 :raises DeserializationError: 

80 If the module path is not on the deserialization allowlist, or if the callable cannot 

81 be found. 

82 """ 

83 # Import here to avoid circular imports 

84 from haystack.hooks.from_function import FunctionHook 

85 from haystack.tools.tool import Tool 

86 

87 parts = callable_handle.split(".") 

88 

89 for i in range(len(parts), 0, -1): 

90 module_name = ".".join(parts[:i]) 

91 # Only import modules that are on the allowlist. Gating the import (rather than a mere 

92 # string prefix of the handle) means a disallowed module is never imported for its 

93 # side effects, and the resolver can only ever start from a trusted module. Shorter 

94 # prefixes are tried in turn, so `json.dumps` still resolves when `json` is allowed. 

95 if not _is_module_allowed(module_name): 

96 continue 

97 try: 

98 mod: Any = thread_safe_import(module_name) 

99 except Exception: 

100 # keep reducing i until we find a valid module import 

101 continue 

102 

103 attr_value = mod 

104 for part in parts[i:]: 

105 try: 

106 attr_value = getattr(attr_value, part) 

107 except AttributeError as e: 

108 container = getattr(attr_value, "__name__", type(attr_value).__name__) 

109 raise DeserializationError(f"Could not find attribute '{part}' in {container}") from e 

110 # A crafted handle can walk into a *module* re-exported as an attribute of an 

111 # allowlisted module (e.g. `haystack.utils.auth.os` -> the `os` module). The declared 

112 # path had an allowlisted prefix, but the module's real identity (`__name__`) is not 

113 # allowlisted. Re-check every module hop so the walk cannot escape the allowlist. 

114 if isinstance(attr_value, ModuleType): 

115 _check_module_allowed(attr_value.__name__) 

116 

117 # when the attribute is a classmethod, we need the underlying function 

118 if isinstance(attr_value, (classmethod, staticmethod)): 

119 attr_value = attr_value.__func__ 

120 

121 # Handle the case where @tool decorator replaced the function with a Tool object 

122 if isinstance(attr_value, Tool): 

123 attr_value = attr_value.function or attr_value.async_function 

124 

125 # Handle the case where @hook decorator replaced the function with a FunctionHook object 

126 if isinstance(attr_value, FunctionHook): 

127 attr_value = attr_value.function or attr_value.async_function 

128 

129 if not callable(attr_value): 

130 raise DeserializationError(f"The final attribute is not callable: {attr_value}") 

131 

132 # Final defense: gate on the module the resolved callable actually comes from, not on the 

133 # declared handle. This catches a dangerous callable bound as a plain (non-module) attribute 

134 # of an allowlisted object, which the module-walk check above would not see. `module_name` 

135 # is the allowlisted module we resolved from, so a private C accelerator backing it (e.g. 

136 # `operator.add` -> `_operator`) is still accepted. 

137 _check_resolved_module_allowed(attr_value, declared_module=module_name) 

138 

139 # `builtins` is on the allowlist (for `builtins.print` etc.), so the module check 

140 # above does not stop dangerous builtins like `eval`/`exec` from resolving here. Block them. 

141 _check_not_denied_builtin(attr_value, callable_handle) 

142 

143 return attr_value 

144 

145 # Nothing on the allowlist was importable. Surface the standard allowlist error when the 

146 # top-level module is untrusted; otherwise report a plain resolution failure. 

147 _check_module_allowed(callable_handle) 

148 raise DeserializationError(f"Could not import '{callable_handle}' as a module or callable.")
