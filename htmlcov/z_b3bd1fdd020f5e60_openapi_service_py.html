20 # Patch the request method to add support for the proper raw_response handling 

21 # If you see that https://github.com/Dorthu/openapi3/pull/124/ 

22 # is merged, we can remove this patch - notify authors of this code 

23 def patch_request( 

24 self: "Operation", 

25 base_url: str, 

26 *, 

27 data: Any | None = None, 

28 parameters: dict[str, Any] | None = None, 

29 raw_response: bool = False, 

30 security: dict[str, str] | None = None, 

31 session: Any | None = None, 

32 verify: bool | str = True, 

33 ) -> Any | None: 

34 """ 

35 Sends an HTTP request as described by this path. 

36 

37 :param base_url: The URL to append this operation's path to when making 

38 the call. 

39 :param data: The request body to send. 

40 :param parameters: The parameters used to create the path. 

41 :param raw_response: If true, return the raw response instead of validating 

42 and extrapolating it. 

43 :param security: The security scheme to use, and the values it needs to 

44 process successfully. 

45 :param session: A persistent request session. 

46 :param verify: If we should do an SSL verification on the request or not. 

47 In case str was provided, will use that as the CA. 

48 :return: The response data, either raw or processed depending on raw_response flag. 

49 """ 

50 # Set request method (e.g. 'GET') 

51 self._request = requests.Request(self.path[-1]) 

52 

53 # Set self._request.url to base_url w/ path 

54 self._request.url = base_url + self.path[-2] 

55 

56 parameters = parameters or {} 

57 security = security or {} 

58 

59 if security and self.security: 

60 security_requirement = None 

61 for scheme, value in security.items(): 

62 security_requirement = None 

63 for r in self.security: 

64 if r.name == scheme: 

65 security_requirement = r 

66 self._request_handle_secschemes(r, value) 

67 

68 if security_requirement is None: 

69 err_msg = """No security requirement satisfied (accepts {}) \ 

70 """.format(", ".join(self.security.keys())) 

71 raise ValueError(err_msg) 

72 

73 if self.requestBody: 

74 if self.requestBody.required and data is None: 

75 err_msg = "Request Body is required but none was provided." 

76 raise ValueError(err_msg) 

77 

78 self._request_handle_body(data) 

79 

80 self._request_handle_parameters(parameters) 

81 

82 if session is None: 

83 session = self._session 

84 

85 # send the prepared request 

86 result = session.send(self._request.prepare(), verify=verify) 

87 

88 # spec enforces these are strings 

89 status_code = str(result.status_code) 

90 

91 # find the response model in spec we received 

92 expected_response = None 

93 if status_code in self.responses: 

94 expected_response = self.responses[status_code] 

95 elif "default" in self.responses: 

96 expected_response = self.responses["default"] 

97 

98 if expected_response is None: 

99 raise UnexpectedResponseError(result, self) 

100 

101 # if we got back a valid response code (or there was a default) and no 

102 # response content was expected, return None 

103 if expected_response.content is None: 

104 return None 

105 

106 content_type = result.headers["Content-Type"] 

107 if ";" in content_type: 

108 # if the content type that came in included an encoding, we'll ignore 

109 # it for now (requests has already parsed it for us) and only look at 

110 # the MIME type when determining if an expected content type was returned. 

111 content_type = content_type.split(";")[0].strip() 

112 

113 expected_media = expected_response.content.get(content_type, None) 

114 

115 # If raw_response is True, return the raw text or json based on content type 

116 if raw_response: 

117 if "application/json" in content_type: 

118 return result.json() 

119 return result.text 

120 

121 if expected_media is None and "/" in content_type: 

122 # accept media type ranges in the spec. the most specific matching 

123 # type should always be chosen, but if we do not have a match here 

124 # a generic range should be accepted if one if provided 

125 # https://github.com/OAI/OpenAPI-Specification/blob/master/versions/3.0.1.md#response-object 

126 

127 generic_type = content_type.split("/")[0] + "/*" 

128 expected_media = expected_response.content.get(generic_type, None) 

129 

130 if expected_media is None: 

131 err_msg = """Unexpected Content-Type {} returned for operation {} \ 

132 (expected one of {})""" 

133 err_var = result.headers["Content-Type"], self.operationId, ",".join(expected_response.content.keys()) 

134 

135 raise RuntimeError(err_msg.format(*err_var)) 

136 

137 if content_type.lower() == "application/json": 

138 return expected_media.schema.model(result.json()) 

139 

140 raise NotImplementedError("Only application/json content type is supported")

146@component 

147class OpenAPIServiceConnector: 

148 """ 

149 A component which connects the Haystack framework to OpenAPI services. 

150 

151 The `OpenAPIServiceConnector` component connects the Haystack framework to OpenAPI services, enabling it to call 

152 operations as defined in the OpenAPI specification of the service. 

153 

154 It integrates with `ChatMessage` dataclass, where the `ToolCall` entries in messages are used to determine the 

155 method to be called and the parameters to be passed. The method name and parameters are then used to invoke the 

156 method on the OpenAPI service. The response from the service is returned as a `ChatMessage`. 

157 

158 Before using this component, users usually resolve service endpoint parameters with a help of 

159 `OpenAPIServiceToFunctions` component. 

160 

161 The example below demonstrates how to use the `OpenAPIServiceConnector` to invoke a method on a https://serper.dev/ 

162 service specified via OpenAPI specification. 

163 

164 Note, however, that `OpenAPIServiceConnector` is usually not meant to be used directly, but rather as part of a 

165 pipeline that includes the `OpenAPIServiceToFunctions` component and a Chat Generator component using an LLM 

166 with tool calling capabilities. In the example below we use the tool call payload directly, but in a 

167 real-world scenario, the tool calls would usually be generated by the Chat Generator component. 

168 

169 You need to define the `serper_token` variable with your Serper.dev API token for the example to work. 

170 Can be through the `SERPERDEV_API_KEY` environment variable or by directly assigning the token string to the 

171 variable in the code. 

172 

173 Usage example: 

174 <!-- test-ignore --> 

175 ```python 

176 import json 

177 import httpx 

178 

179 from haystack.components.connectors import OpenAPIServiceConnector 

180 from haystack.dataclasses import ChatMessage, ToolCall 

181 from haystack.utils import Secret 

182 

183 tool_call = ToolCall( 

184 tool_name="search", 

185 arguments={"q": "Why was Sam Altman ousted from OpenAI?"}, 

186 ) 

187 message = ChatMessage.from_assistant(tool_calls=[tool_call]) 

188 

189 serper_token = Secret.from_env_var("SERPERDEV_API_KEY").resolve_value() 

190 serperdev_openapi_spec = json.loads(httpx.get("https://bit.ly/serper_dev_spec", follow_redirects=True).text) 

191 service_connector = OpenAPIServiceConnector() 

192 result = service_connector.run( 

193 messages=[message], 

194 service_openapi_spec=serperdev_openapi_spec, 

195 service_credentials=serper_token, 

196 ) 

197 print(result) 

198 

199 # {'service_response': [ChatMessage(_role=<ChatRole.USER: 'user'>, _content=[TextContent(text= 

200 # '{"searchParameters": {"q": "Why was Sam Altman ousted from OpenAI?", 

201 # "type": "search", "engine": "google"}, "answerBox": {"snippet": "Concerns over AI safety and OpenAI\'s role 

202 # in protecting were at the center of Altman\'s brief ouster from the company."... 

203 ``` 

204 

205 """

217 @component.output_types(service_response=dict[str, Any]) 

218 def run( 

219 self, 

220 messages: list[ChatMessage], 

221 service_openapi_spec: dict[str, Any], 

222 service_credentials: dict | str | None = None, 

223 ) -> dict[str, list[ChatMessage]]: 

224 """ 

225 Processes a list of chat messages to invoke a method on an OpenAPI service. 

226 

227 It parses the last message in the list, expecting it to contain tool calls. 

228 

229 :param messages: A list of `ChatMessage` objects containing the messages to be processed. The last message 

230 should contain the tool calls. 

231 :param service_openapi_spec: The OpenAPI JSON specification object of the service to be invoked. All the refs 

232 should already be resolved. 

233 :param service_credentials: The credentials to be used for authentication with the service. 

234 Currently, only the http and apiKey OpenAPI security schemes are supported. 

235 

236 :return: A dictionary with the following keys: 

237 - `service_response`: a list of `ChatMessage` objects, each containing the response from the service. The 

238 response is in JSON format, and the `content` attribute of the `ChatMessage` contains 

239 the JSON string. 

240 

241 :raises ValueError: If the last message is not from the assistant or if it does not contain tool calls. 

242 """ 

243 

244 last_message = messages[-1] 

245 if not last_message.is_from(ChatRole.ASSISTANT): 

246 raise ValueError(f"{last_message} is not from the assistant.") 

247 

248 tool_calls = last_message.tool_calls 

249 if not tool_calls: 

250 raise ValueError(f"The provided ChatMessage has no tool calls.\nChatMessage: {last_message}") 

251 

252 function_payloads = [] 

253 for tool_call in tool_calls: 

254 function_payloads.append({"arguments": tool_call.arguments, "name": tool_call.tool_name}) 

255 

256 # instantiate the OpenAPI service for the given specification 

257 openapi_service = OpenAPI(service_openapi_spec, ssl_verify=self.ssl_verify) 

258 self._authenticate_service(openapi_service, service_credentials) 

259 

260 response_messages = [] 

261 for method_invocation_descriptor in function_payloads: 

262 service_response = self._invoke_method(openapi_service, method_invocation_descriptor) 

263 # openapi3 parses the JSON service response into a model object, which is not our focus at the moment. 

264 # Instead, we require direct access to the raw JSON data of the response, rather than the model objects 

265 # provided by the openapi3 library. This approach helps us avoid issues related to (de)serialization. 

266 # By accessing the raw JSON response through `service_response._raw_data`, we can serialize this data 

267 # into a string. Finally, we use this string to create a ChatMessage object. 

268 response_messages.append(ChatMessage.from_user(json.dumps(service_response))) 

269 

270 return {"service_response": response_messages}

293 def _authenticate_service(self, openapi_service: "OpenAPI", credentials: dict | str | None = None) -> None: 

294 """ 

295 Authentication with an OpenAPI service. 

296 

297 Authenticates with the OpenAPI service if required, supporting both single (str) and multiple 

298 authentication methods (dict). 

299 

300 OpenAPI spec v3 supports the following security schemes: 

301 http – for Basic, Bearer and other HTTP authentications schemes 

302 apiKey – for API keys and cookie authentication 

303 oauth2 – for OAuth 2 

304 openIdConnect – for OpenID Connect Discovery 

305 

306 Currently, only the http and apiKey schemes are supported. Multiple security schemes can be defined in the 

307 OpenAPI spec, and the credentials should be provided as a dictionary with keys matching the security scheme 

308 names. If only one security scheme is defined, the credentials can be provided as a simple string. 

309 

310 :param openapi_service: The OpenAPI service instance. 

311 :param credentials: Credentials for authentication, which can be either a string (e.g. token) or a dictionary 

312 with keys matching the authentication method names. 

313 :raises ValueError: If authentication fails, is not found, or if appropriate credentials are missing. 

314 """ 

315 if openapi_service.raw_element.get("components", {}).get("securitySchemes"): 

316 service_name = openapi_service.info.title 

317 if not credentials: 

318 raise ValueError(f"Service {service_name} requires authentication but no credentials were provided.") 

319 

320 # a dictionary of security schemes defined in the OpenAPI spec 

321 # each key is the name of the security scheme, and the value is the scheme definition 

322 security_schemes = openapi_service.components.securitySchemes.raw_element 

323 supported_schemes = ["http", "apiKey"] # todo: add support for oauth2 and openIdConnect 

324 

325 authenticated = False 

326 for scheme_name, scheme in security_schemes.items(): 

327 if scheme["type"] in supported_schemes: 

328 auth_credentials = None 

329 if isinstance(credentials, str): 

330 auth_credentials = credentials 

331 elif isinstance(credentials, dict) and scheme_name in credentials: 

332 auth_credentials = credentials[scheme_name] 

333 if auth_credentials: 

334 openapi_service.authenticate(scheme_name, auth_credentials) 

335 authenticated = True 

336 break 

337 

338 raise ValueError( 

339 f"Service {service_name} requires {scheme_name} security scheme but no " 

340 f"credentials were provided for it. Check the service configuration and credentials." 

341 ) 

342 if not authenticated: 

343 raise ValueError( 

344 f"Service {service_name} requires authentication but no credentials were provided " 

345 f"for it. Check the service configuration and credentials." 

346 )

348 def _invoke_method(self, openapi_service: "OpenAPI", method_invocation_descriptor: dict[str, Any]) -> Any: 

349 """ 

350 Invokes the specified method on the OpenAPI service. 

351 

352 The method name and arguments are passed in the method_invocation_descriptor. 

353 

354 :param openapi_service: The OpenAPI service instance. 

355 :param method_invocation_descriptor: The method name and arguments to be passed to the method. The payload 

356 should contain the method name (key: "name") and the arguments (key: "arguments"). The name is a string, and 

357 the arguments are a dictionary of key-value pairs. 

358 :return: A service JSON response. 

359 :raises ValueError: If the descriptor is malformed or a required parameter is missing. 

360 :raises TypeError: If the operation is not found in the OpenAPI specification. 

361 """ 

362 name = method_invocation_descriptor.get("name") 

363 invocation_arguments = copy(method_invocation_descriptor.get("arguments", {})) 

364 if not name or not invocation_arguments: 

365 raise ValueError( 

366 f"Invalid function calling descriptor: {method_invocation_descriptor} . It should contain " 

367 f"a method name and arguments." 

368 ) 

369 

370 # openapi3 specific method to call the operation, do we have it? 

371 method_to_call = getattr(openapi_service, f"call_{name}", None) 

372 if not callable(method_to_call): 

373 raise TypeError(f"Operation {name} not found in OpenAPI specification {openapi_service.info.title}") 

374 

375 # get the operation reference from the method_to_call 

376 operation = method_to_call.operation.__self__ 

377 operation_dict = operation.raw_element 

378 

379 # Pack URL/query parameters under "parameters" key 

380 method_call_params: dict[str, dict[str, Any]] = defaultdict(dict) 

381 parameters = operation_dict.get("parameters", []) 

382 request_body = operation_dict.get("requestBody", {}) 

383 

384 for param in parameters: 

385 param_name = param["name"] 

386 param_value = invocation_arguments.get(param_name) 

387 if param_value: 

388 method_call_params["parameters"][param_name] = param_value 

389 elif param.get("required", False): 

390 raise ValueError(f"Missing parameter: '{param_name}' required for the '{name}' operation.") 

391 

392 # Pack request body parameters under "data" key 

393 if request_body: 

394 schema = request_body.get("content", {}).get("application/json", {}).get("schema", {}) 

395 required_params = schema.get("required", []) 

396 for param_name in schema.get("properties", {}): 

397 param_value = invocation_arguments.get(param_name) 

398 if param_value: 

399 method_call_params["data"][param_name] = param_value 

400 elif param_name in required_params: 

401 raise ValueError( 

402 f"Missing requestBody parameter: '{param_name}' required for the '{name}' operation." 

403 ) 

404 # call the underlying service REST API with the parameters 

405 return method_to_call(**method_call_params, raw_response=True)
