103@component 

104class Agent: 

105 """ 

106 A tool-using Agent powered by a large language model. 

107 

108 The Agent processes messages and calls tools until it meets an exit condition. 

109 You can set one or more exit conditions to control when it stops. 

110 For example, it can stop after generating a response or after calling a tool. 

111 

112 Without tools, the Agent works like a standard LLM that generates text. It produces one response and then stops. 

113 

114 ### Usage examples 

115 

116 This is an example agent that: 

117 1. Searches for tipping customs in France. 

118 2. Uses a calculator to compute tips based on its findings. 

119 3. Returns the final answer with its context. 

120 

121 ```python 

122 from haystack.components.agents import Agent 

123 from haystack.components.generators.chat import OpenAIChatGenerator 

124 from haystack.dataclasses import ChatMessage 

125 from haystack.tools import Tool 

126 

127 # Tool functions - in practice, these would have real implementations 

128 def search(query: str) -> str: 

129 '''Search for information on the web.''' 

130 # Placeholder: would call actual search API 

131 return "In France, a 15% service charge is typically included, but leaving 5-10% extra is appreciated." 

132 

133 def calculator(operation: str, a: float, b: float) -> float: 

134 '''Perform mathematical calculations.''' 

135 if operation == "multiply": 

136 return a * b 

137 elif operation == "percentage": 

138 return (a / 100) * b 

139 return 0 

140 

141 # Define tools with JSON Schema 

142 tools = [ 

143 Tool( 

144 name="search", 

145 description="Searches for information on the web", 

146 parameters={ 

147 "type": "object", 

148 "properties": { 

149 "query": {"type": "string", "description": "The search query"} 

150 }, 

151 "required": ["query"] 

152 }, 

153 function=search 

154 ), 

155 Tool( 

156 name="calculator", 

157 description="Performs mathematical calculations", 

158 parameters={ 

159 "type": "object", 

160 "properties": { 

161 "operation": {"type": "string", "description": "Operation: multiply, percentage"}, 

162 "a": {"type": "number", "description": "First number"}, 

163 "b": {"type": "number", "description": "Second number"} 

164 }, 

165 "required": ["operation", "a", "b"] 

166 }, 

167 function=calculator 

168 ) 

169 ] 

170 

171 # Create and run the agent 

172 agent = Agent( 

173 chat_generator=OpenAIChatGenerator(), 

174 tools=tools 

175 ) 

176 

177 result = agent.run( 

178 messages=[ChatMessage.from_user("Calculate the appropriate tip for an €85 meal in France")] 

179 ) 

180 

181 print(result["messages"][-1].text) 

182 ```

184 #### Using a `user_prompt` template with variables 

185 

186 You can define a reusable `user_prompt` with Jinja2 template variables so the Agent can be invoked 

187 with different inputs without manually constructing `ChatMessage` objects each time. 

188 This is especially useful when embedding the Agent in a pipeline or calling it in a loop. 

189 

190 ```python 

191 from haystack.components.agents import Agent 

192 from haystack.components.generators.chat import OpenAIChatGenerator 

193 from haystack.tools import tool 

194 from typing import Annotated 

195 

196 

197 @tool 

198 def translate( 

199 text: Annotated[str, "The text to translate"], 

200 target_language: Annotated[str, "The language to translate to"], 

201 ) -> str: 

202 \"\"\"Translate text to a target language.\"\"\" 

203 # Placeholder: would call an actual translation API 

204 return f"[Translated '{text}' to {target_language}]" 

205 

206 agent = Agent( 

207 chat_generator=OpenAIChatGenerator(), 

208 tools=[translate], 

209 system_prompt="You are a helpful translation assistant.", 

210 user_prompt=\"\"\"{% message role="user"%} 

211 Translate the following document to {{ language }}: {{ document }} 

212 {% endmessage %}\"\"\", 

213 required_variables=["language", "document"], 

214 ) 

215 

216 # The template variables 'language' and 'document' become inputs to the run method 

217 result = agent.run( 

218 messages=[], 

219 language="French", 

220 document="The weather is lovely today and the sun is shining.", 

221 ) 

222 

223 print(result["last_message"].text) 

224 ```

228 def __init__( 

229 self, 

230 *, 

231 chat_generator: ChatGenerator, 

232 tools: ToolsType | None = None, 

233 system_prompt: str | None = None, 

234 user_prompt: str | None = None, 

235 required_variables: list[str] | Literal["*"] | None = None, 

236 exit_conditions: list[str] | None = None, 

237 state_schema: dict[str, Any] | None = None, 

238 max_agent_steps: int = 100, 

239 streaming_callback: StreamingCallbackT | None = None, 

240 raise_on_tool_invocation_failure: bool = False, 

241 tool_invoker_kwargs: dict[str, Any] | None = None, 

242 confirmation_strategies: dict[str | tuple[str, ...], ConfirmationStrategy] | None = None, 

243 ) -> None: 

244 """ 

245 Initialize the agent component. 

246 

247 :param chat_generator: An instance of the chat generator that your agent should use. It must support tools. 

248 :param tools: A list of Tool and/or Toolset objects, or a single Toolset that the agent can use. 

249 :param system_prompt: System prompt for the agent. Can be a plain string or a Jinja2 string template. 

250 For details on the supported template syntax, refer to the 

251 [documentation](https://docs.haystack.deepset.ai/docs/chatpromptbuilder#string-templates). 

252 :param user_prompt: User prompt for the agent, defined as a Jinja2 string template. If provided, this is 

253 appended to the messages provided at runtime. 

254 For details on the supported template syntax, refer to the 

255 [documentation](https://docs.haystack.deepset.ai/docs/chatpromptbuilder#string-templates). 

256 :param required_variables: 

257 List variables that must be provided as input to user_prompt or system_prompt. 

258 If a variable listed as required is not provided, an exception is raised. 

259 If set to `"*"`, all variables found in the prompts are required. Optional. 

260 :param exit_conditions: List of conditions that will cause the agent to return. 

261 Can include "text" if the agent should return when it generates a message without tool calls, 

262 or tool names that will cause the agent to return once the tool was executed. Defaults to ["text"]. 

263 :param state_schema: The schema for the runtime state used by the tools. 

264 :param max_agent_steps: Maximum number of steps the agent will run before stopping. Defaults to 100. 

265 If the agent exceeds this number of steps, it will stop and return the current state. 

266 :param streaming_callback: A callback that will be invoked when a response is streamed from the LLM. 

267 The same callback can be configured to emit tool results when a tool is called. 

268 :param raise_on_tool_invocation_failure: Should the agent raise an exception when a tool invocation fails? 

269 If set to False, the exception will be turned into a chat message and passed to the LLM. 

270 :param tool_invoker_kwargs: Additional keyword arguments to pass to the ToolInvoker. 

271 :param confirmation_strategies: A dictionary mapping tool names to ConfirmationStrategy instances. 

272 :raises TypeError: If the chat_generator does not support tools parameter in its run method. 

273 :raises ValueError: If the exit_conditions are not valid. 

274 :raises ValueError: If any `user_prompt` variable overlaps with `state` schema or `run` parameters. 

275 """ 

276 # Check if chat_generator supports tools parameter 

277 chat_generator_run_method = inspect.signature(chat_generator.run) 

278 self._chat_generator_supports_tools: bool = "tools" in chat_generator_run_method.parameters 

279 if tools and not self._chat_generator_supports_tools: 

280 raise TypeError( 

281 f"{type(chat_generator).__name__} does not accept tools parameter in its run method. " 

282 "The Agent component requires a chat generator that supports tools when tools are provided." 

283 ) 

284 

285 valid_exits = ["text"] + [tool.name for tool in flatten_tools_or_toolsets(tools)] 

286 if exit_conditions is None: 

287 exit_conditions = ["text"] 

288 if not all(condition in valid_exits for condition in exit_conditions): 

289 raise ValueError( 

290 f"Invalid exit conditions provided: {exit_conditions}. " 

291 f"Valid exit conditions must be a subset of {valid_exits}. " 

292 "Ensure that each exit condition corresponds to either 'text' or a valid tool name." 

293 ) 

294 

295 # Validate state schema if provided 

296 if state_schema is not None: 

297 _validate_schema(state_schema) 

298 self._state_schema = state_schema or {} 

299 

300 # Initialize state schema 

301 # shallow copy is sufficient: we only add a top-level "messages" key, never mutate nested values 

302 resolved_state_schema = dict(self._state_schema) 

303 if resolved_state_schema.get("messages") is None: 

304 resolved_state_schema["messages"] = {"type": list[ChatMessage], "handler": merge_lists} 

305 self.state_schema = resolved_state_schema 

306 

307 self.chat_generator = chat_generator 

308 self.tools = tools or [] 

309 self.system_prompt = system_prompt 

310 self.user_prompt = user_prompt 

311 self.required_variables = required_variables 

312 self.exit_conditions = exit_conditions 

313 self.max_agent_steps = max_agent_steps 

314 self.raise_on_tool_invocation_failure = raise_on_tool_invocation_failure 

315 self.streaming_callback = streaming_callback 

316 

317 # Set input and output types for the component based on the State schema 

318 self._run_method_params = _get_run_method_params(self) 

319 output_types = {"last_message": ChatMessage} 

320 for param, config in self.state_schema.items(): 

321 output_types[param] = config["type"] 

322 # Skip setting input types for parameters that are already in the run method 

323 if param in self._run_method_params: 

324 continue 

325 component.set_input_type(self, name=param, type=config["type"], default=None) 

326 component.set_output_types(self, **output_types) 

327 

328 # required_variables is initially set to [] and populated later by _register_prompt_variables 

329 self._user_chat_prompt_builder = ( 

330 ChatPromptBuilder(template=user_prompt, required_variables=[]) if user_prompt is not None else None 

331 ) 

332 # Only create a system prompt builder when the prompt uses Jinja2 message syntax 

333 self._system_chat_prompt_builder: ChatPromptBuilder | None = None 

334 if system_prompt is not None and _JINJA2_CHAT_TEMPLATE_RE.search(system_prompt): 

335 self._system_chat_prompt_builder = ChatPromptBuilder(template=system_prompt, required_variables=[]) 

336 

337 self._register_prompt_variables() 

338 

339 self.tool_invoker_kwargs = tool_invoker_kwargs 

340 self._tool_invoker = None 

341 if self.tools: 

342 resolved_tool_invoker_kwargs = { 

343 "tools": self.tools, 

344 "raise_on_failure": self.raise_on_tool_invocation_failure, 

345 **(tool_invoker_kwargs or {}), 

346 } 

347 self._tool_invoker = ToolInvoker(**resolved_tool_invoker_kwargs) 

348 elif type(self).__name__ == "Agent": 

349 logger.warning( 

350 "No tools provided to the Agent. The Agent will behave like a ChatGenerator and only return text " 

351 "responses. To enable tool usage, pass tools directly to the Agent, not to the chat_generator." 

352 ) 

353 

354 self._confirmation_strategies = confirmation_strategies or {} 

355 

356 self._is_warmed_up = False

358 def _register_prompt_variables(self) -> None: 

359 """ 

360 Collect variables from both Chat Prompt Builders and register Agent inputs. 

361 

362 Sets `required_variables` for both Chat Prompt Builders, checks for conflicts with state schema and 

363 run parameters, and registers component inputs. 

364 """ 

365 required_variables = self.required_variables 

366 

367 if ( 

368 required_variables is not None 

369 and self._system_chat_prompt_builder is None 

370 and self._user_chat_prompt_builder is None 

371 ): 

372 logger.warning( 

373 "The parameter required_variables is provided but neither user_prompt nor system_prompt " 

374 "contains template variables. Either provide a prompt with Jinja2 template variables " 

375 "or remove required_variables, it has otherwise no effect." 

376 ) 

377 

378 all_variables: dict[str, list[str]] = {} 

379 for builder, label in [ 

380 (self._system_chat_prompt_builder, "system_prompt"), 

381 (self._user_chat_prompt_builder, "user_prompt"), 

382 ]: 

383 if builder is not None: 

384 # set required_variables on the builder, filtered to its own variables 

385 if required_variables == "*": 

386 builder.required_variables = "*" 

387 elif isinstance(self.required_variables, list): 

388 builder.required_variables = [v for v in self.required_variables if v in builder.variables] 

389 

390 for var_name in builder.variables: 

391 all_variables.setdefault(var_name, []).append(label) 

392 

393 for var_name, sources in all_variables.items(): 

394 prompt_source = " and ".join(sources) 

395 if var_name in self.state_schema: 

396 raise ValueError( 

397 f"Variable '{var_name}' from {prompt_source} is already defined in the state schema. " 

398 "Please rename the variable or remove it from the prompt to avoid conflicts." 

399 ) 

400 if var_name in self._run_method_params: 

401 raise ValueError( 

402 f"Variable '{var_name}' from {prompt_source} conflicts with input names in the run method. " 

403 "Please rename the variable or remove it from the prompt to avoid conflicts." 

404 ) 

405 if required_variables == "*" or (isinstance(required_variables, list) and var_name in required_variables): 

406 component.set_input_type(self, name=var_name, type=Any) 

407 else: 

408 component.set_input_type(self, name=var_name, type=Any, default=None)

451 @classmethod 

452 def from_dict(cls, data: dict[str, Any]) -> "Agent": 

453 """ 

454 Deserialize the agent from a dictionary. 

455 

456 :param data: Dictionary to deserialize from 

457 :return: Deserialized agent 

458 """ 

459 init_params = data.get("init_parameters", {}) 

460 

461 deserialize_component_inplace(init_params, key="chat_generator") 

462 

463 if init_params.get("state_schema") is not None: 

464 init_params["state_schema"] = _schema_from_dict(init_params["state_schema"]) 

465 

466 if init_params.get("streaming_callback") is not None: 

467 init_params["streaming_callback"] = deserialize_callable(init_params["streaming_callback"]) 

468 

469 deserialize_tools_or_toolset_inplace(init_params, key="tools") 

470 

471 if init_params.get("confirmation_strategies") is not None: 

472 restored: dict[str | tuple[str, ...], Any] = {} 

473 for raw_key in init_params["confirmation_strategies"].keys(): 

474 deserialize_component_inplace(init_params["confirmation_strategies"], key=raw_key) 

475 strategy = init_params["confirmation_strategies"][raw_key] 

476 if isinstance(raw_key, list): 

477 key = tuple(raw_key) 

478 else: 

479 key = raw_key 

480 restored[key] = strategy 

481 init_params["confirmation_strategies"] = restored 

482 

483 return default_from_dict(cls, data)

504 def _initialize_fresh_execution( 

505 self, 

506 messages: list[ChatMessage], 

507 streaming_callback: StreamingCallbackT | None, 

508 requires_async: bool, 

509 *, 

510 system_prompt: str | None = None, 

511 user_prompt: str | None = None, 

512 generation_kwargs: dict[str, Any] | None = None, 

513 tools: ToolsType | list[str] | None = None, 

514 confirmation_strategy_context: dict[str, Any] | None = None, 

515 **kwargs: Any, 

516 ) -> _ExecutionContext: 

517 """ 

518 Initialize execution context for a fresh run of the agent. 

519 

520 :param messages: List of ChatMessage objects to start the agent with. 

521 :param streaming_callback: Optional callback for streaming responses. 

522 :param requires_async: Whether the agent run requires asynchronous execution. 

523 :param system_prompt: System prompt for the agent. If provided, it overrides the default system prompt. 

524 :param user_prompt: User prompt for the agent. If provided, it overrides the default user prompt and is 

525 appended to the messages provided at runtime. 

526 :param generation_kwargs: Additional keyword arguments for chat generator. These parameters will 

527 override the parameters passed during component initialization. 

528 :param tools: Optional list of Tool objects, a Toolset, or list of tool names to use for this run. 

529 When passing tool names, tools are selected from the Agent's originally configured tools. 

530 :param confirmation_strategy_context: Optional dictionary for passing request-scoped resources 

531 to confirmation strategies. 

532 :param kwargs: Additional data to pass to the State used by the Agent. 

533 """ 

534 user_prompt = user_prompt or self.user_prompt 

535 system_prompt = system_prompt or self.system_prompt 

536 messages = messages or [] 

537 

538 if user_prompt is not None: 

539 if self._user_chat_prompt_builder is None: 

540 raise ValueError( 

541 "user_prompt is provided but the ChatPromptBuilder is not initialized. " 

542 "Please make sure a user_prompt is provided at initialization time." 

543 ) 

544 

545 # Only forward the prompt kwargs to the prompt builder 

546 prompt_kwargs = {var: kwargs[var] for var in self._user_chat_prompt_builder.variables if var in kwargs} 

547 user_messages = self._user_chat_prompt_builder.run(template=user_prompt, **prompt_kwargs)["prompt"] 

548 if len(user_messages) != 1: 

549 raise ValueError( 

550 f"user_prompt must render to exactly one user message. Got {len(user_messages)} messages." 

551 ) 

552 if not user_messages[0].is_from(ChatRole.USER): 

553 raise ValueError( 

554 f"user_prompt must render to a user message. Got a message with role {user_messages[0].role}." 

555 ) 

556 messages = messages + user_messages 

557 

558 if system_prompt is not None: 

559 if _JINJA2_CHAT_TEMPLATE_RE.search(system_prompt): 

560 if self._system_chat_prompt_builder is None: 

561 raise ValueError( 

562 "system_prompt contains Jinja2 template syntax but no system prompt builder is initialized. " 

563 "Please make sure a system_prompt with Jinja2 template syntax is provided at initialization " 

564 "time." 

565 ) 

566 

567 prompt_kwargs = { 

568 var: kwargs[var] for var in self._system_chat_prompt_builder.variables if var in kwargs 

569 } 

570 system_messages = self._system_chat_prompt_builder.run(template=system_prompt, **prompt_kwargs)[ 

571 "prompt" 

572 ] 

573 if len(system_messages) != 1: 

574 raise ValueError( 

575 f"system_prompt must render to exactly one system message. Got {len(system_messages)} messages." 

576 ) 

577 if not system_messages[0].is_from(ChatRole.SYSTEM): 

578 raise ValueError( 

579 "system_prompt must render to a system message. " 

580 f"Got a message with role {system_messages[0].role}." 

581 ) 

582 messages = system_messages + messages 

583 else: 

584 messages = [ChatMessage.from_system(system_prompt)] + messages 

585 

586 if all(m.is_from(ChatRole.SYSTEM) for m in messages): 

587 logger.warning("All messages provided to the Agent component are system messages. This is not recommended.") 

588 

589 state_kwargs: dict[str, Any] = {key: kwargs[key] for key in self.state_schema.keys() if key in kwargs} 

590 state = State(schema=self.state_schema, data=state_kwargs) 

591 state.set("messages", messages) 

592 

593 streaming_callback = select_streaming_callback( # type: ignore[call-overload] 

594 init_callback=self.streaming_callback, runtime_callback=streaming_callback, requires_async=requires_async 

595 ) 

596 

597 selected_tools = self._select_tools(tools) 

598 tool_invoker_inputs: dict[str, Any] = {"tools": selected_tools} 

599 generator_inputs: dict[str, Any] = {} 

600 if self._chat_generator_supports_tools: 

601 generator_inputs["tools"] = selected_tools 

602 if streaming_callback is not None: 

603 tool_invoker_inputs["streaming_callback"] = streaming_callback 

604 generator_inputs["streaming_callback"] = streaming_callback 

605 if generation_kwargs is not None: 

606 generator_inputs["generation_kwargs"] = generation_kwargs 

607 

608 # We add enable_streaming_callback_passthrough to the tool invoker inputs 

609 if self._tool_invoker: 

610 tool_invoker_inputs["enable_streaming_callback_passthrough"] = ( 

611 self._tool_invoker.enable_streaming_callback_passthrough 

612 ) 

613 

614 return _ExecutionContext( 

615 state=state, 

616 component_visits=dict.fromkeys(["chat_generator", "tool_invoker"], 0), 

617 chat_generator_inputs=generator_inputs, 

618 tool_invoker_inputs=tool_invoker_inputs, 

619 confirmation_strategy_context=confirmation_strategy_context, 

620 )

622 def _select_tools(self, tools: ToolsType | list[str] | None = None) -> ToolsType: 

623 """ 

624 Select tools for the current run based on the provided tools parameter. 

625 

626 :param tools: Optional list of Tool objects, a Toolset, or list of tool names to use for this run. 

627 When passing tool names, tools are selected from the Agent's originally configured tools. 

628 :returns: Selected tools for the current run. 

629 :raises ValueError: If tool names are provided but no tools were configured at initialization, 

630 or if any provided tool name is not valid. 

631 :raises TypeError: If tools is not a list of Tool objects, a Toolset, or a list of tool names (strings). 

632 """ 

633 if tools is None: 

634 return self.tools 

635 

636 if isinstance(tools, list) and all(isinstance(t, str) for t in tools): 

637 if not self.tools: 

638 raise ValueError("No tools were configured for the Agent at initialization.") 

639 available_tools = flatten_tools_or_toolsets(self.tools) 

640 selected_tool_names = cast(list[str], tools) # mypy thinks this could still be list[Tool] or Toolset 

641 valid_tool_names = {tool.name for tool in available_tools} 

642 invalid_tool_names = {name for name in selected_tool_names if name not in valid_tool_names} 

643 if invalid_tool_names: 

644 raise ValueError( 

645 f"The following tool names are not valid: {invalid_tool_names}. " 

646 f"Valid tool names are: {valid_tool_names}." 

647 ) 

648 return [tool for tool in available_tools if tool.name in selected_tool_names] 

649 

650 if isinstance(tools, Toolset): 

651 return tools 

652 

653 if isinstance(tools, list): 

654 return cast(list[Tool | Toolset], tools) # mypy can't narrow the Union type from isinstance check 

655 

656 raise TypeError( 

657 "tools must be a list of Tool and/or Toolset objects, a Toolset, or a list of tool names (strings)." 

658 )

741 def run( # noqa: PLR0915 

742 self, 

743 messages: list[ChatMessage], 

744 streaming_callback: StreamingCallbackT | None = None, 

745 *, 

746 generation_kwargs: dict[str, Any] | None = None, 

747 break_point: AgentBreakpoint | None = None, 

748 snapshot: AgentSnapshot | None = None, 

749 system_prompt: str | None = None, 

750 user_prompt: str | None = None, 

751 tools: ToolsType | list[str] | None = None, 

752 snapshot_callback: SnapshotCallback | None = None, 

753 confirmation_strategy_context: dict[str, Any] | None = None, 

754 **kwargs: Any, 

755 ) -> dict[str, Any]: 

756 """ 

757 Process messages and execute tools until an exit condition is met. 

758 

759 :param messages: List of Haystack ChatMessage objects to process. 

760 :param streaming_callback: A callback that will be invoked when a response is streamed from the LLM. 

761 The same callback can be configured to emit tool results when a tool is called. 

762 :param generation_kwargs: Additional keyword arguments for LLM. These parameters will 

763 override the parameters passed during component initialization. 

764 :param break_point: An AgentBreakpoint, can be a Breakpoint for the "chat_generator" or a ToolBreakpoint 

765 for "tool_invoker". 

766 :param snapshot: A dictionary containing a snapshot of a previously saved agent execution. The snapshot contains 

767 the relevant information to restart the Agent execution from where it left off. 

768 :param system_prompt: System prompt for the agent. If provided, it overrides the default system prompt. 

769 :param user_prompt: User prompt for the agent. If provided, it overrides the default user prompt and is 

770 appended to the messages provided at runtime. 

771 :param tools: Optional list of Tool objects, a Toolset, or list of tool names to use for this run. 

772 When passing tool names, tools are selected from the Agent's originally configured tools. 

773 :param snapshot_callback: Optional callback function that is invoked when a pipeline snapshot is created. 

774 The callback receives a `PipelineSnapshot` object and can return an optional string. 

775 If provided, the callback is used instead of the default file-saving behavior. 

776 :param confirmation_strategy_context: Optional dictionary for passing request-scoped resources 

777 to confirmation strategies. Useful in web/server environments to provide per-request 

778 objects (e.g., WebSocket connections, async queues, Redis pub/sub clients) that strategies 

779 can use for non-blocking user interaction. 

780 :param kwargs: Additional data to pass to the State schema used by the Agent. 

781 The keys must match the schema defined in the Agent's `state_schema`. 

782 :returns: 

783 A dictionary with the following keys: 

784 - "messages": List of all messages exchanged during the agent's run. 

785 - "last_message": The last message exchanged during the agent's run. 

786 - Any additional keys defined in the `state_schema`. 

787 :raises BreakpointException: If an agent breakpoint is triggered. 

788 """ 

789 agent_inputs = { 

790 "messages": messages, 

791 "streaming_callback": streaming_callback, 

792 "break_point": break_point, 

793 "snapshot": snapshot, 

794 **kwargs, 

795 } 

796 self._runtime_checks(break_point=break_point, tools=self._select_tools(tools)) 

797 

798 if snapshot: 

799 exe_context = self._initialize_from_snapshot( 

800 snapshot=snapshot, 

801 streaming_callback=streaming_callback, 

802 requires_async=False, 

803 generation_kwargs=generation_kwargs, 

804 tools=tools, 

805 confirmation_strategy_context=confirmation_strategy_context, 

806 ) 

807 else: 

808 exe_context = self._initialize_fresh_execution( 

809 messages=messages, 

810 streaming_callback=streaming_callback, 

811 requires_async=False, 

812 system_prompt=system_prompt, 

813 user_prompt=user_prompt, 

814 generation_kwargs=generation_kwargs, 

815 tools=tools, 

816 confirmation_strategy_context=confirmation_strategy_context, 

817 **kwargs, 

818 ) 

819 

820 with self._create_agent_span() as span: 

821 # agent_inputs is local and not used after this point, so we avoid deepcopying it 

822 span.set_content_tag("haystack.agent.input", agent_inputs) 

823 

824 while exe_context.counter < self.max_agent_steps: 

825 # We skip the chat generator when restarting from a snapshot from a ToolBreakpoint 

826 if exe_context.skip_chat_generator: 

827 llm_messages = exe_context.state.get("messages", [])[-1:] 

828 # Set to False so the next iteration will call the chat generator 

829 exe_context.skip_chat_generator = False 

830 else: 

831 try: 

832 result = Pipeline._run_component( 

833 component_name="chat_generator", 

834 component={"instance": self.chat_generator}, 

835 inputs={ 

836 "messages": exe_context.state.data["messages"], 

837 **exe_context.chat_generator_inputs, 

838 }, 

839 component_visits=exe_context.component_visits, 

840 parent_span=span, 

841 break_point=break_point.break_point if isinstance(break_point, AgentBreakpoint) else None, 

842 ) 

843 except PipelineRuntimeError as e: 

844 agent_name = getattr(self, "__component_name__", None)

846 agent_name=agent_name, execution_context=exe_context, break_point=None 

847 ) 

848 new_error = PipelineRuntimeError.from_exception(agent_name or "Agent", Agent, e) 

849 new_error.pipeline_snapshot = pipe_snapshot 

850 # If Agent is not in a pipeline, we save the snapshot to a file or invoke a custom callback. 

851 # Checked by __component_name__ not being set. 

852 if agent_name is None:

854 pipeline_snapshot=pipe_snapshot, snapshot_callback=snapshot_callback 

855 ) 

856 raise new_error from e 

857 except BreakpointException as e: 

858 agent_name = break_point.agent_name if break_point else None 

859 e.pipeline_snapshot = _create_pipeline_snapshot_from_chat_generator( 

860 agent_name=agent_name, execution_context=exe_context, break_point=break_point 

861 )

868 )

873 exe_context.state.set("messages", llm_messages) 

874 

875 # Check if any of the LLM responses contain a tool call or if the LLM is not using tools 

876 if not any(msg.tool_call for msg in llm_messages) or self._tool_invoker is None: 

877 exe_context.counter += 1 

878 break 

879 

880 # We only pass down the breakpoint if the tool name matches the tool call in the LLM messages 

881 break_point_to_pass = None 

882 if ( 

883 break_point 

884 and isinstance(break_point.break_point, ToolBreakpoint) 

885 and _should_trigger_tool_invoker_breakpoint( 

886 break_point=break_point.break_point, llm_messages=llm_messages 

887 ) 

888 ): 

889 break_point_to_pass = break_point.break_point 

890 

891 # Apply confirmation strategies and update State and messages sent to ToolInvoker 

892 # Run confirmation strategies to get updated tool call messages and modified chat history 

893 modified_tool_call_messages, new_chat_history = _process_confirmation_strategies( 

894 confirmation_strategies=self._confirmation_strategies, 

895 messages_with_tool_calls=llm_messages, 

896 execution_context=exe_context, 

897 ) 

898 # Replace the chat history in state with the modified one 

899 exe_context.state.set(key="messages", value=new_chat_history, handler_override=replace_values) 

900 

901 # Run ToolInvoker 

902 try: 

903 # We only send the messages from the LLM to the tool invoker 

904 tool_invoker_result = Pipeline._run_component( 

905 component_name="tool_invoker", 

906 component={"instance": self._tool_invoker}, 

907 inputs={ 

908 "messages": modified_tool_call_messages, 

909 "state": exe_context.state, 

910 **exe_context.tool_invoker_inputs, 

911 }, 

912 component_visits=exe_context.component_visits, 

913 parent_span=span, 

914 break_point=break_point_to_pass, 

915 ) 

916 except PipelineRuntimeError as e: 

917 agent_name = getattr(self, "__component_name__", None) 

918 tool_name = getattr(e.__cause__, "tool_name", None) 

919 pipe_snapshot = _create_pipeline_snapshot_from_tool_invoker( 

920 tool_name=tool_name, agent_name=agent_name, execution_context=exe_context, break_point=None 

921 ) 

922 new_error = PipelineRuntimeError.from_exception(agent_name or "Agent", Agent, e) 

923 new_error.pipeline_snapshot = pipe_snapshot 

924 # If Agent is not in a pipeline, we save the snapshot to a file or invoke a custom callback. 

925 # Checked by __component_name__ not being set. 

926 if agent_name is None: 

927 new_error.pipeline_snapshot_file_path = _save_pipeline_snapshot( 

928 pipeline_snapshot=pipe_snapshot, snapshot_callback=snapshot_callback 

929 ) 

930 raise new_error from e 

931 except BreakpointException as e: 

932 e.pipeline_snapshot = _create_pipeline_snapshot_from_tool_invoker( 

933 tool_name=e.break_point.tool_name if isinstance(e.break_point, ToolBreakpoint) else None, 

934 agent_name=break_point.agent_name if break_point else None, 

935 execution_context=exe_context, 

936 break_point=break_point, 

937 ) 

938 e._break_point = e.pipeline_snapshot.break_point 

939 # If Agent is not in a pipeline, we save the snapshot to a file or invoke a custom callback. 

940 # Checked by __component_name__ not being set. 

941 if getattr(self, "__component_name__", None) is None: 

942 e.pipeline_snapshot_file_path = _save_pipeline_snapshot( 

943 pipeline_snapshot=e.pipeline_snapshot, snapshot_callback=snapshot_callback 

944 ) 

945 raise e 

946 

947 tool_messages = tool_invoker_result["tool_messages"] 

948 exe_context.state = tool_invoker_result["state"] 

949 exe_context.state.set("messages", tool_messages) 

950 

951 # Check if any LLM message's tool call name matches an exit condition 

952 if self.exit_conditions != ["text"] and self._check_exit_conditions(llm_messages, tool_messages): 

953 exe_context.counter += 1 

954 break 

955 

956 # Increment the step counter 

957 exe_context.counter += 1 

958 

959 if exe_context.counter >= self.max_agent_steps: 

960 logger.warning( 

961 "Agent reached maximum agent steps of {max_agent_steps}, stopping.", 

962 max_agent_steps=self.max_agent_steps, 

963 ) 

964 span.set_content_tag("haystack.agent.output", exe_context.state.data) 

965 span.set_tag("haystack.agent.steps_taken", exe_context.counter) 

966 

967 result = {**exe_context.state.data} 

968 if msgs := result.get("messages"): 

969 result["last_message"] = msgs[-1] 

970 return result

1206 def _check_exit_conditions(self, llm_messages: list[ChatMessage], tool_messages: list[ChatMessage]) -> bool: 

1207 """ 

1208 Check if any of the LLM messages' tool calls match an exit condition and if there are no errors. 

1209 

1210 :param llm_messages: List of messages from the LLM 

1211 :param tool_messages: List of messages from the tool invoker 

1212 :return: True if an exit condition is met and there are no errors, False otherwise 

1213 """ 

1214 matched_exit_conditions = set() 

1215 has_errors = False 

1216 

1217 for msg in llm_messages: 

1218 if msg.tool_call and msg.tool_call.tool_name in self.exit_conditions: 

1219 matched_exit_conditions.add(msg.tool_call.tool_name) 

1220 

1221 # Check if any error is specifically from the tool matching the exit condition 

1222 tool_errors = [ 

1223 tool_msg.tool_call_result.error 

1224 for tool_msg in tool_messages 

1225 if tool_msg.tool_call_result is not None 

1226 and tool_msg.tool_call_result.origin.tool_name == msg.tool_call.tool_name 

1227 ] 

1228 if any(tool_errors): 

1229 has_errors = True 

1230 # No need to check further if we found an error 

1231 break 

1232 

1233 # Only return True if at least one exit condition was matched AND none had errors 

1234 return bool(matched_exit_conditions) and not has_errors
