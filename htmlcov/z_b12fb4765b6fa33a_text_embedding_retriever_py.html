13@component 

14class TextEmbeddingRetriever: 

15 """ 

16 A component that retrieves documents using a query with an embedding-based retriever. 

17 

18 This component takes a text query, converts it to an embedding using a text embedder, and then uses an 

19 embedding-based retriever to find relevant documents. 

20 The results are sorted by relevance score. 

21 

22 ### Usage example 

23 

24 ```python 

25 from haystack import Document 

26 from haystack.document_stores.in_memory import InMemoryDocumentStore 

27 from haystack.document_stores.types import DuplicatePolicy 

28 from haystack.components.embedders import SentenceTransformersTextEmbedder, SentenceTransformersDocumentEmbedder 

29 from haystack.components.retrievers import InMemoryEmbeddingRetriever, TextEmbeddingRetriever 

30 from haystack.components.writers import DocumentWriter 

31 

32 documents = [ 

33 Document(content="Renewable energy is energy that is collected from renewable resources."), 

34 Document(content="Solar energy is a type of green energy that is harnessed from the sun."), 

35 Document(content="Wind energy is another type of green energy that is generated by wind turbines."), 

36 Document(content="Geothermal energy is heat that comes from the sub-surface of the earth."), 

37 Document(content="Biomass energy is produced from organic materials, such as plant and animal waste."), 

38 Document(content="Fossil fuels, such as coal, oil, and natural gas, are non-renewable energy sources."), 

39 ] 

40 

41 # Populate the document store 

42 doc_store = InMemoryDocumentStore() 

43 doc_embedder = SentenceTransformersDocumentEmbedder(model="sentence-transformers/all-MiniLM-L6-v2") 

44 doc_writer = DocumentWriter(document_store=doc_store, policy=DuplicatePolicy.SKIP) 

45 documents = doc_embedder.run(documents)["documents"] 

46 doc_writer.run(documents=documents) 

47 

48 # Run the retriever 

49 in_memory_retriever = InMemoryEmbeddingRetriever(document_store=doc_store, top_k=1) 

50 text_embedder = SentenceTransformersTextEmbedder(model="sentence-transformers/all-MiniLM-L6-v2") 

51 retriever = TextEmbeddingRetriever(retriever=in_memory_retriever, text_embedder=text_embedder) 

52 result = retriever.run(query="Geothermal energy") 

53 

54 for doc in result["documents"]: 

55 print(f"Content: {doc.content}, Score: {doc.score}") 

56 # >> Content: Geothermal energy is heat that comes from the sub-surface of the earth., Score: 0.8509603046266574 

57 ``` 

58 """ 

59 

60 def __init__(self, *, retriever: EmbeddingRetriever, text_embedder: TextEmbedder) -> None: 

61 """ 

62 Initialize TextEmbeddingRetriever. 

63 

64 :param retriever: The embedding-based retriever to use for document retrieval. 

65 :param text_embedder: The text embedder to convert a text query to an embedding. 

66 """ 

67 self.retriever = retriever 

68 self.text_embedder = text_embedder 

69 self._is_warmed_up = False 

70 

71 def warm_up(self) -> None: 

72 """ 

73 Warm up the text embedder and the retriever if any has a warm_up method. 

74 """ 

75 if not self._is_warmed_up: 

76 if hasattr(self.text_embedder, "warm_up") and callable(self.text_embedder.warm_up): 

77 self.text_embedder.warm_up() 

78 if hasattr(self.retriever, "warm_up") and callable(self.retriever.warm_up): 

79 self.retriever.warm_up() 

80 self._is_warmed_up = True 

81 

82 @component.output_types(documents=list[Document]) 

83 def run( 

84 self, query: str, filters: dict[str, Any] | None = None, top_k: int | None = None 

85 ) -> dict[str, list[Document]]: 

86 """ 

87 Retrieve documents using a single query. 

88 

89 :param query: The query to retrieve documents for. 

90 :param filters: A dictionary of filters to apply when retrieving documents. 

91 :param top_k: The maximum number of documents to return. 

92 :returns: 

93 A dictionary containing: 

94 - `documents`: List of retrieved documents sorted by relevance score. 

95 """ 

96 if not self._is_warmed_up: 

97 self.warm_up() 

98 

99 embedding_result = self.text_embedder.run(text=query) 

100 result = self.retriever.run(query_embedding=embedding_result["embedding"], filters=filters, top_k=top_k) 

101 docs: list[Document] = result["documents"] 

102 

103 # sort 

104 docs.sort(key=lambda x: x.score or 0.0, reverse=True) 

105 return {"documents": docs}
