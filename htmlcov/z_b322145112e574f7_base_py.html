116 def __eq__(self, other: object) -> bool: 

117 """ 

118 Pipeline equality is defined by their type and the equality of their serialized form. 

119 

120 Pipelines of the same type share every metadata, node and edge, but they're not required to use 

121 the same node instances: this allows pipeline saved and then loaded back to be equal to themselves. 

122 """ 

123 if not isinstance(self, type(other)): 

124 return False 

125 assert isinstance(other, PipelineBase) 

126 return self.to_dict() == other.to_dict()

150 def to_dict(self) -> dict[str, Any]: 

151 """ 

152 Serializes the pipeline to a dictionary. 

153 

154 This is meant to be an intermediate representation but it can be also used to save a pipeline to file. 

155 

156 :returns: 

157 Dictionary with serialized data. 

158 """ 

159 components = {} 

160 for name, instance in self.graph.nodes(data="instance"): 

161 components[name] = component_to_dict(instance, name) 

162 

163 connections = [] 

164 for sender, receiver, edge_data in self.graph.edges.data(): 

165 sender_socket = edge_data["from_socket"].name 

166 receiver_socket = edge_data["to_socket"].name 

167 connections.append({"sender": f"{sender}.{sender_socket}", "receiver": f"{receiver}.{receiver_socket}"}) 

168 return { 

169 "metadata": self.metadata, 

170 "max_runs_per_component": self._max_runs_per_component, 

171 "components": components, 

172 "connections": connections, 

173 "connection_type_validation": self._connection_type_validation, 

174 }

177 def from_dict( 

178 cls: type[T], data: dict[str, Any], callbacks: DeserializationCallbacks | None = None, **kwargs: Any 

179 ) -> T: 

180 """ 

181 Deserializes the pipeline from a dictionary. 

182 

183 :param data: 

184 Dictionary to deserialize from. 

185 :param callbacks: 

186 Callbacks to invoke during deserialization. 

187 :param kwargs: 

188 `components`: a dictionary of `{name: instance}` to reuse instances of components instead of creating new 

189 ones. 

190 :returns: 

191 Deserialized component. 

192 """ 

193 data_copy = _deepcopy_with_exceptions(data) # to prevent modification of original data 

194 metadata = data_copy.get("metadata", {}) 

195 max_runs_per_component = data_copy.get("max_runs_per_component", 100) 

196 connection_type_validation = data_copy.get("connection_type_validation", True) 

197 pipe = cls( 

198 metadata=metadata, 

199 max_runs_per_component=max_runs_per_component, 

200 connection_type_validation=connection_type_validation, 

201 ) 

202 components_to_reuse = kwargs.get("components", {}) 

203 for name, component_data in data_copy.get("components", {}).items(): 

204 if name in components_to_reuse: 

205 # Reuse an instance 

206 instance = components_to_reuse[name] 

207 else: 

208 if "type" not in component_data: 

209 raise PipelineError(f"Missing 'type' in component '{name}'") 

210 

211 if component_data["type"] not in component.registry: 

212 try: 

213 # Import the module first... 

214 module, _ = component_data["type"].rsplit(".", 1) 

215 logger.debug("Trying to import module {module_name}", module_name=module) 

216 type_serialization.thread_safe_import(module) 

217 # ...then try again 

218 if component_data["type"] not in component.registry: 

219 raise PipelineError( # noqa: TRY301 

220 f"Successfully imported module '{module}' but couldn't find " 

221 f"'{component_data['type']}' in the component registry.\n" 

222 f"The component might be registered under a different path. " 

223 f"Here are the registered components:\n {list(component.registry.keys())}\n" 

224 ) 

225 except (ImportError, PipelineError, ValueError) as e: 

226 raise PipelineError( 

227 f"Component '{component_data['type']}' (name: '{name}') not imported. Please " 

228 f"check that the package is installed and the component path is correct." 

229 ) from e 

230 

231 # Create a new one 

232 component_class = component.registry[component_data["type"]] 

233 

234 try: 

235 instance = component_from_dict(component_class, component_data, name, callbacks) 

236 except Exception as e: 

237 # Convert to JSON with indentation, truncate if too long 

238 try: 

239 data_str = json.dumps(component_data, default=str, indent=2) 

240 except Exception: 

241 data_str = str(component_data) 

242 

243 max_len = 1000 

244 if len(data_str) > max_len: 

245 data_str = data_str[:max_len] + "\n... (truncated)" 

246 

247 msg = ( 

248 f"Couldn't deserialize component '{name}' of class '{component_class.__name__}' " 

249 f"with the following data:\n{data_str}\n\n" 

250 f"Original error: {e}" 

251 ) 

252 raise DeserializationError(msg) from e 

253 pipe.add_component(name=name, instance=instance) 

254 

255 for connection in data.get("connections", []): 

256 if "sender" not in connection: 

257 raise PipelineError(f"Missing sender in connection: {connection}") 

258 if "receiver" not in connection: 

259 raise PipelineError(f"Missing receiver in connection: {connection}") 

260 pipe.connect(sender=connection["sender"], receiver=connection["receiver"]) 

261 

262 return pipe

343 def add_component(self, name: str, instance: Component) -> None: 

344 """ 

345 Add the given component to the pipeline. 

346 

347 Components are not connected to anything by default: use `Pipeline.connect()` to connect components together. 

348 Component names must be unique, but component instances can be reused if needed. 

349 

350 :param name: 

351 The name of the component to add. 

352 :param instance: 

353 The component instance to add. 

354 

355 :raises ValueError: 

356 If a component with the same name already exists. 

357 :raises PipelineValidationError: 

358 If the given instance is not a component. 

359 """ 

360 # Component names are unique 

361 if name in self.graph.nodes: 

362 raise ValueError(f"A component named '{name}' already exists in this pipeline: choose another name.") 

363 

364 # Components can't be named `_debug` 

365 if name == "_debug": 

366 raise ValueError("'_debug' is a reserved name for debug output. Choose another name.") 

367 

368 # Component names can't have "." 

369 if "." in name: 

370 raise ValueError(f"{name} is an invalid component name, cannot contain '.' (dot) characters.") 

371 

372 # Component instances must be components 

373 if not isinstance(instance, Component): 

374 raise PipelineValidationError( 

375 f"'{type(instance)}' doesn't seem to be a component. Is this class decorated with @component?" 

376 ) 

377 

378 if getattr(instance, "__haystack_added_to_pipeline__", None): 

379 msg = ( 

380 "Component has already been added in another Pipeline. Components can't be shared between Pipelines. " 

381 "Create a new instance instead." 

382 ) 

383 raise PipelineError(msg) 

384 

385 setattr(instance, "__haystack_added_to_pipeline__", self) # noqa: B010 

386 setattr(instance, "__component_name__", name) # noqa: B010 

387 

388 # Add component to the graph, disconnected 

389 logger.debug("Adding component '{component_name}' ({component})", component_name=name, component=instance) 

390 # We're completely sure the fields exist so we ignore the type error 

391 self.graph.add_node( 

392 name, 

393 instance=instance, 

394 input_sockets=instance.__haystack_input__._sockets_dict, # type: ignore[attr-defined] 

395 output_sockets=instance.__haystack_output__._sockets_dict, # type: ignore[attr-defined] 

396 visits=0, 

397 )

402

404 All edges that connect to the component will also be deleted. 

405 

406 :param name:

434 socket.receivers = [] 

435

437 setattr(instance, "__haystack_added_to_pipeline__", None) # noqa: B010 

438 

439 return instance

441 def connect(self, sender: str, receiver: str) -> "PipelineBase": # noqa: PLR0915 PLR0912 C901 

442 """ 

443 Connects two components together. 

444 

445 All components to connect must exist in the pipeline. 

446 If connecting to a component that has several output connections, specify the inputs and output names as 

447 'component_name.connections_name'. 

448 

449 If multiple senders are connected to the same list-typed receiver socket, the socket is 

450 promoted to a lazy variadic socket so it can accept all incoming values. With `Pipeline`, 

451 the resulting list is ordered alphabetically by sender component name, not by the order in 

452 which `connect()` was called. With `AsyncPipeline`, no ordering is guaranteed, since 

453 components in different branches may run in parallel. 

454 

455 :param sender: 

456 The component that delivers the value. This can be either just a component name or can be 

457 in the format `component_name.connection_name` if the component has multiple outputs. 

458 :param receiver: 

459 The component that receives the value. This can be either just a component name or can be 

460 in the format `component_name.connection_name` if the component has multiple inputs. 

461 

462 :returns: 

463 The Pipeline instance. 

464 

465 :raises PipelineConnectError: 

466 If the two components cannot be connected (for example if one of the components is 

467 not present in the pipeline, or the connections don't match by type, and so on). 

468 """ 

469 # Edges may be named explicitly by passing 'node_name.edge_name' to connect(). 

470 sender_component_name, sender_socket_name = parse_connect_string(sender) 

471 receiver_component_name, receiver_socket_name = parse_connect_string(receiver) 

472 

473 if sender_component_name == receiver_component_name: 

474 raise PipelineConnectError("Connecting a Component to itself is not supported.") 

475 

476 # Get the nodes data. 

477 try: 

478 sender_sockets = self.graph.nodes[sender_component_name]["output_sockets"] 

479 except KeyError as exc: 

480 raise ValueError(f"Component named {sender_component_name} not found in the pipeline.") from exc 

481 try: 

482 receiver_sockets = self.graph.nodes[receiver_component_name]["input_sockets"] 

483 except KeyError as exc: 

484 raise ValueError(f"Component named {receiver_component_name} not found in the pipeline.") from exc 

485 

486 if not sender_sockets: 

487 raise PipelineConnectError( 

488 f"'{sender_component_name}' does not have any output connections. " 

489 f"Please check that the output types of '{sender_component_name}.run' are set, " 

490 f"for example by using the '@component.output_types' decorator." 

491 ) 

492 

493 # If the name of either socket is given, get the socket 

494 sender_socket: OutputSocket | None = None 

495 if sender_socket_name: 

496 sender_socket = sender_sockets.get(sender_socket_name) 

497 if not sender_socket: 

498 raise PipelineConnectError( 

499 f"'{sender}' does not exist. " 

500 f"Output connections of {sender_component_name} are: " 

501 + ", ".join([f"{name} (type {_type_name(socket.type)})" for name, socket in sender_sockets.items()]) 

502 ) 

503 

504 receiver_socket: InputSocket | None = None 

505 if receiver_socket_name: 

506 receiver_socket = receiver_sockets.get(receiver_socket_name) 

507 if not receiver_socket: 

508 raise PipelineConnectError( 

509 f"'{receiver} does not exist. " 

510 f"Input connections of {receiver_component_name} are: " 

511 + ", ".join( 

512 [f"{name} (type {_type_name(socket.type)})" for name, socket in receiver_sockets.items()] 

513 ) 

514 ) 

515 

516 # Look for a matching connection among the possible ones. 

517 # Note that if there is more than one possible connection but two sockets match by name, they're paired. 

518 sender_socket_candidates: list[OutputSocket] = ( 

519 [sender_socket] if sender_socket else list(sender_sockets.values()) 

520 ) 

521 receiver_socket_candidates: list[InputSocket] = ( 

522 [receiver_socket] if receiver_socket else list(receiver_sockets.values()) 

523 ) 

524 

525 conversion_strategy = None 

526 

527 # Find all possible connections between these two components 

528 possible_connections: list[tuple[OutputSocket, InputSocket, ConversionStrategyType]] = [] 

529 for sender_sock, receiver_sock in itertools.product(sender_socket_candidates, receiver_socket_candidates): 

530 is_compat, conversion_strategy = _types_are_compatible( 

531 sender_sock.type, receiver_sock.type, self._connection_type_validation 

532 ) 

533 if is_compat: 

534 possible_connections.append((sender_sock, receiver_sock, conversion_strategy)) 

535 

536 # If there are multiple possibilities, prioritize strict matches over convertible ones. 

537 # This ensures backward compatibility: previously, pipelines did not allow type conversion. 

538 if len(possible_connections) > 1 and self._connection_type_validation: 

539 strict_matches = [ 

540 (out_sock, in_sock, None) 

541 for out_sock, in_sock, conversion_strategy in possible_connections 

542 if conversion_strategy is None 

543 ] 

544 if strict_matches: 

545 possible_connections[:] = strict_matches 

546 

547 # We need this status for error messages, since we might need it in multiple places we calculate it here 

548 status = _connections_status( 

549 sender_node=sender_component_name, 

550 sender_sockets=sender_socket_candidates, 

551 receiver_node=receiver_component_name, 

552 receiver_sockets=receiver_socket_candidates, 

553 ) 

554 

555 if not possible_connections: 

556 # There's no possible connection between these two components 

557 if len(sender_socket_candidates) == len(receiver_socket_candidates) == 1: 

558 msg = ( 

559 f"Cannot connect '{sender_component_name}.{sender_socket_candidates[0].name}' with " 

560 f"'{receiver_component_name}.{receiver_socket_candidates[0].name}': " 

561 f"their declared input and output types do not match.\n{status}" 

562 ) 

563 else: 

564 msg = ( 

565 f"Cannot connect '{sender_component_name}' with '{receiver_component_name}': " 

566 f"no matching connections available.\n{status}" 

567 ) 

568 raise PipelineConnectError(msg) 

569 

570 if len(possible_connections) == 1: 

571 # There's only one possible connection, use it 

572 sender_socket = possible_connections[0][0] 

573 receiver_socket = possible_connections[0][1] 

574 conversion_strategy = possible_connections[0][2] 

575 

576 if len(possible_connections) > 1: 

577 # There are multiple possible connection, let's try to match them by name 

578 name_matches = [ 

579 (out_sock, in_sock, conversion_strategy_) 

580 for out_sock, in_sock, conversion_strategy_ in possible_connections 

581 if in_sock.name == out_sock.name 

582 ] 

583 if len(name_matches) != 1: 

584 # There's are either no matches or more than one, we can't pick one reliably 

585 msg = ( 

586 f"Cannot connect '{sender_component_name}' with " 

587 f"'{receiver_component_name}': more than one connection is possible " 

588 "between these components. Please specify the connection name, like: " 

589 f"pipeline.connect('{sender_component_name}.{possible_connections[0][0].name}', " 

590 f"'{receiver_component_name}.{possible_connections[0][1].name}').\n{status}" 

591 ) 

592 raise PipelineConnectError(msg) 

593 

594 # Get the only possible match 

595 sender_socket = name_matches[0][0] 

596 receiver_socket = name_matches[0][1] 

597 conversion_strategy = name_matches[0][2] 

598 

599 # Connection must be valid on both sender/receiver sides 

600 if not sender_socket or not receiver_socket or not sender_component_name or not receiver_component_name: 

601 if sender_component_name and sender_socket: 

602 sender_repr = f"{sender_component_name}.{sender_socket.name} ({_type_name(sender_socket.type)})" 

603 else: 

604 sender_repr = "input needed" 

605 

606 if receiver_component_name and receiver_socket: 

607 receiver_repr = f"({_type_name(receiver_socket.type)}) {receiver_component_name}.{receiver_socket.name}" 

608 else: 

609 receiver_repr = "output" 

610 msg = f"Connection must have both sender and receiver: {sender_repr} -> {receiver_repr}" 

611 raise PipelineConnectError(msg) 

612 

613 logger.debug(

641 to_socket=receiver_socket,

911 def validate_input(self, data: dict[str, Any]) -> None: 

912 """ 

913 Validates pipeline input data. 

914 

915 Validates that data: 

916 * Each Component name actually exists in the Pipeline 

917 * Each Component is not missing any input 

918 * Each Component has only one input per input socket, if not variadic 

919 * Each Component doesn't receive inputs that are already sent by another Component 

920 

921 :param data: 

922 A dictionary of inputs for the pipeline's components. Each key is a component name. 

923 

924 :raises ValueError: 

925 If inputs are invalid according to the above. 

926 """ 

927 for component_name, component_inputs in data.items(): 

928 # Check that the component exists 

929 if component_name not in self.graph.nodes: 

930 raise ValueError(f"Component named '{component_name}' not found in the pipeline.") 

931 # Check that no input is provided that the component can't accept 

932 instance = self.graph.nodes[component_name]["instance"] 

933 for input_name in component_inputs.keys(): 

934 if input_name not in instance.__haystack_input__._sockets_dict: 

935 raise ValueError(f"Input '{input_name}' not found in component '{component_name}'.") 

936 

937 for component_name in self.graph.nodes: 

938 instance = self.graph.nodes[component_name]["instance"] 

939 for socket_name, socket in instance.__haystack_input__._sockets_dict.items(): 

940 component_inputs = data.get(component_name, {}) 

941 # Check that no mandatory input is missing for any component in the pipeline 

942 if socket.senders == [] and socket.is_mandatory and socket_name not in component_inputs: 

943 raise ValueError(f"Missing mandatory input '{socket_name}' for component '{component_name}'.") 

944 

945 # Check if an input is provided more than once for non-variadic sockets 

946 if socket.senders and socket_name in component_inputs: 

947 self._make_socket_auto_variadic( 

948 component_name=component_name, receiver_socket=socket, error_type=ValueError

951 def _make_socket_auto_variadic( 

952 self, component_name: str, receiver_socket: InputSocket, error_type: type[Exception] 

953 ) -> InputSocket: 

954 """ 

955 Attempts to make the receiver socket lazy variadic in-place to accommodate a new sender. 

956 

957 A socket is automatically made lazy variadic when: 

958 - It already has at least one connected sender 

959 - It is not already variadic 

960 - Its type is list, Optional[list], a union of list types 

961 

962 When auto-variadicity is applied, `wrap_input_in_list` is also set to False so that sender output types match 

963 the receiver socket's declared list type directly. 

964 

965 :param component_name: 

966 Name of the component owning the receiver socket, used in error messages. 

967 :param receiver_socket: 

968 The receiver socket to inspect and potentially modify in-place. 

969 :param error_type: 

970 Exception class to raise on failure (e.g. ValueError or PipelineConnectError). 

971 :returns: 

972 The (possibly modified) receiver socket. 

973 :raises error_type: 

974 If the socket cannot accept multiple senders given its type constraints. 

975 """ 

976 # If it's already variadic, we return as-is 

977 if receiver_socket.is_variadic: 

978 return receiver_socket 

979 

980 # Get receiver origin 

981 receiver_origin = _safe_get_origin(receiver_socket.type) 

982 

983 # Handle Union types 

984 if receiver_origin == Union: 

985 # Unwrap Optional types 

986 non_none_args = [a for a in get_args(receiver_socket.type) if a is not type(None)] 

987 if len(non_none_args) == 1: 

988 receiver_origin = _safe_get_origin(non_none_args[0]) 

989 # Handle Union of list types (e.g. list[int] | list[str]) 

990 elif all(_safe_get_origin(arg) == list for arg in non_none_args): 

991 receiver_origin = list 

992 

993 # If the receiver origin is a list, we can make the socket lazy variadic 

994 if receiver_origin == list: 

995 receiver_socket.is_lazy_variadic = True 

996 receiver_socket.wrap_input_in_list = False 

997 return receiver_socket 

998 

999 raise error_type( 

1000 f"Component '{component_name}' cannot accept multiple inputs to '{receiver_socket.name}'. " 

1001 f"It is already connected to component '{receiver_socket.senders[0]}', and it can only accept " 

1002 f"inputs from multiple senders if its type is list, Optional[list], or union of list types." 

1003 )

1005 def _prepare_component_input_data(self, data: dict[str, Any]) -> dict[str, dict[str, Any]]: 

1006 """ 

1007 Prepares input data for pipeline components. 

1008 

1009 Organizes input data for pipeline components and identifies any inputs that are not matched to any 

1010 component's input slots. Deep-copies data items to avoid sharing mutables across multiple components. 

1011 

1012 This method processes a flat dictionary of input data, where each key-value pair represents an input name 

1013 and its corresponding value. It distributes these inputs to the appropriate pipeline components based on 

1014 their input requirements. Inputs that don't match any component's input slots are classified as unresolved. 

1015 

1016 :param data: 

1017 A dictionary potentially having input names as keys and input values as values. 

1018 

1019 :returns: 

1020 A dictionary mapping component names to their respective matched inputs. 

1021 """ 

1022 # check whether the data is a nested dictionary of component inputs where each key is a component name 

1023 # and each value is a dictionary of input parameters for that component 

1024 is_nested_component_input = all(isinstance(value, dict) for value in data.values()) 

1025 if not is_nested_component_input: 

1026 # flat input, a dict where keys are input names and values are the corresponding values 

1027 # we need to convert it to a nested dictionary of component inputs and then run the pipeline 

1028 # just like in the previous case 

1029 pipeline_input_data: dict[str, dict[str, Any]] = defaultdict(dict) 

1030 unresolved_kwargs = {} 

1031 

1032 # Retrieve the input slots for each component in the pipeline 

1033 available_inputs: dict[str, dict[str, Any]] = self.inputs() 

1034 

1035 # Go through all provided to distribute them to the appropriate component inputs 

1036 for input_name, input_value in data.items(): 

1037 resolved_at_least_once = False 

1038 

1039 # Check each component to see if it has a slot for the current kwarg 

1040 for component_name, component_inputs in available_inputs.items(): 

1041 if input_name in component_inputs: 

1042 # If a match is found, add the kwarg to the component's input data 

1043 pipeline_input_data[component_name][input_name] = input_value 

1044 resolved_at_least_once = True 

1045 

1046 if not resolved_at_least_once: 

1047 unresolved_kwargs[input_name] = input_value 

1048 

1049 if unresolved_kwargs: 

1050 logger.warning( 

1051 "Inputs {input_keys} were not matched to any component inputs, please check your run parameters.", 

1052 input_keys=list(unresolved_kwargs.keys()), 

1053 ) 

1054 

1055 data = dict(pipeline_input_data) 

1056 

1057 # deepcopying the inputs prevents the Pipeline run logic from being altered unexpectedly 

1058 # when the same input reference is passed to multiple components. 

1059 for component_name, component_inputs in data.items(): 

1060 data[component_name] = {k: _deepcopy_with_exceptions(v) for k, v in component_inputs.items()} 

1061 

1062 return data

1109 def _consume_component_inputs(

1111 ) -> dict[str, Any]: 

1112 """ 

1113 Extracts the inputs needed to run for the component and removes them from the global inputs state. 

1114 

1115 :param component_name: The name of a component. 

1116 :param component: Component with component metadata. 

1117 :param inputs: Global inputs state. 

1118 :returns: The inputs for the component. 

1119 """

1124 socket_inputs = component_inputs.get(socket_name, []) 

1125 socket_inputs_values = [sock["value"] for sock in socket_inputs if sock["value"] is not _NO_OUTPUT_PRODUCED] 

1126 

1127 # if we are resuming a component, the inputs are already consumed, so we just return the first input 

1128 if is_resume: 

1129 consumed_inputs[socket_name] = socket_inputs_values[0]

1131 

1132 if socket_inputs_values:

1135 # outside the pipeline. Otherwise, a greedy input from the user would trigger a pipeline to run 

1136 # indefinitely. 

1137 greedy_inputs_to_remove.add(socket_name) 

1138 consumed_inputs[socket_name] = [socket_inputs_values[0]]

1150 consumed_inputs[socket_name] = socket_inputs_values[0]

1156 ] 

1157 for socket_name, socket in component_inputs.items()

1276 def _topological_sort(self) -> dict[str, int]: 

1277 """ 

1278 Returns a topological sort of the components in the pipeline. 

1279 

1280 If the graph is a DAG, we use lexicographical topological sort to get a deterministic order of the components. 

1281 If the graph is not a DAG, we use the condensation of the graph to get a topological sort of the strongly 

1282 connected components. This way, components that are part of the same cycle will have the same priority and will 

1283 be tie-broken by their name in lexicographical order, while components that are not part of the same cycle will 

1284 be tie-broken by their topological order. 

1285 

1286 :returns: 

1287 A dictionary mapping component names to their position in the topological sort. 

1288 """ 

1289 if networkx.is_directed_acyclic_graph(self.graph): 

1290 topological_sort = networkx.lexicographical_topological_sort(self.graph) 

1291 return {node: idx for idx, node in enumerate(topological_sort)} 

1292 else: 

1293 condensed = networkx.condensation(self.graph) 

1294 condensed_sorted = {node: idx for idx, node in enumerate(networkx.topological_sort(condensed))} 

1295 return { 

1296 component_name: condensed_sorted[node] for component_name, node in condensed.graph["mapping"].items() 

1297 }

1299 def _tiebreak_waiting_components( 

1300 self, 

1301 component_name: str, 

1302 priority: ComponentPriority, 

1303 priority_queue: FIFOPriorityQueue, 

1304 topological_sort: dict[str, int] | None, 

1305 ) -> tuple[str, dict[str, int] | None]: 

1306 """ 

1307 Decides which component to run when multiple components are waiting for inputs with the same priority. 

1308 

1309 NOTE: This was designed to only tie-break for priorities DEFER and DEFER_LAST. Since this function also removes 

1310 these components from the priority queue we rely on _is_queue_stale to then refill the priority queue. 

1311 And _is_queue_stale only triggers when all remaining components have BLOCKED priority. 

1312 

1313 :param component_name: The name of the component. 

1314 :param priority: Priority of the component. 

1315 :param priority_queue: Priority queue of component names. 

1316 :param topological_sort: Cached topological sort of all components in the pipeline. 

1317 

1318 :returns: 

1319 The name of the component to run and the cached topological sort of all components in the pipeline. 

1320 """ 

1321 # Create a list of all components that have the same priority as the current component, including the 

1322 # current component itself and remove them from the priority queue. 

1323 has_deferred_priority = priority in [ComponentPriority.DEFER, ComponentPriority.DEFER_LAST] 

1324 components_with_same_priority = [component_name] 

1325 while len(priority_queue) > 0: 

1326 next_priority, next_component_name = priority_queue.peek() 

1327 # For tiebreaking purposes we treat DEFER and DEFER_LAST as the same priority. 

1328 if ( 

1329 has_deferred_priority 

1330 and next_priority in [ComponentPriority.DEFER, ComponentPriority.DEFER_LAST] 

1331 or next_priority == priority 

1332 ): 

1333 priority_queue.pop() # actually remove the component 

1334 components_with_same_priority.append(next_component_name) 

1335 else: 

1336 break 

1337 

1338 # If there are multiple components with the same priority, we tiebreak them to decide which one to run first. 

1339 if len(components_with_same_priority) > 1: 

1340 topological_sort = topological_sort or self._topological_sort() 

1341 components_with_same_priority = sorted( 

1342 components_with_same_priority, key=lambda comp_name: (topological_sort[comp_name], comp_name.lower()) 

1343 ) 

1344 component_name = components_with_same_priority[0] 

1345 

1346 return component_name, topological_sort

1348 def _find_components_blocking_pipeline( 

1349 self, priority_queue: FIFOPriorityQueue, component_visits: dict[str, int], inputs: InputsType 

1350 ) -> tuple[list[str], list[str]]: 

1351 """ 

1352 Finds the components that are most likely blocking the pipeline execution. 

1353 

1354 :returns: 

1355 The list of component names that are most likely blocking the pipeline execution and their corresponding 

1356 component types. 

1357 """ 

1358 # 1. Go through all components in priority queue (should all be blocked at this point) 

1359 comps_in_queue: list[str] = [comp_name for _, _, comp_name in priority_queue._queue] 

1360 

1361 # 2. Check which components have entries in inputs. 

1362 comps_with_inputs = [] 

1363 for comp_name in comps_in_queue: 

1364 # If component has non-empty inputs and is blocked it means that the component is waiting for more inputs 

1365 # to run, so it could be blocking the pipeline. 

1366 if inputs.get(comp_name): 

1367 comps_with_inputs.append(comp_name) 

1368 

1369 # If there are no components with any inputs we fallback to checking all components in the queue. 

1370 # This isn't always ideal since already executed components can also be in the queue at this point also 

1371 # with blocked priority. 

1372 if not comps_with_inputs: 

1373 comps_with_inputs = comps_in_queue 

1374 

1375 # 3. Only keep components with the lowest number of visits. Mostly needed to handle the fallback case if no 

1376 # components with inputs are found. 

1377 ordered_comps_with_inputs = sorted(comps_with_inputs, key=lambda x: component_visits[x]) 

1378 lowest_component_visit = component_visits[ordered_comps_with_inputs[0]] 

1379 possible_blocking_comps = [ 

1380 comp for comp in ordered_comps_with_inputs if component_visits[comp] == lowest_component_visit 

1381 ] 

1382 

1383 # If there is only one component with the lowest visits, return it as the most likely blocking component. 

1384 if len(possible_blocking_comps) == 1: 

1385 blocking_comp_types = [self.graph.nodes[possible_blocking_comps[0]]["instance"].__class__.__name__] 

1386 self._log_warning_for_blocking_components( 

1387 blocking_comp_names=possible_blocking_comps, blocking_comp_types=blocking_comp_types 

1388 ) 

1389 return possible_blocking_comps, blocking_comp_types 

1390 

1391 # 4. Then for all components with the same lowest component visits we sort topologically before returning. 

1392 topological_sort = self._topological_sort() 

1393 possible_blocking_comps = sorted( 

1394 possible_blocking_comps, key=lambda comp_name: (topological_sort[comp_name], comp_name.lower()) 

1395 ) 

1396 possible_blocking_comp_types = [ 

1397 self.graph.nodes[comp_name]["instance"].__class__.__name__ for comp_name in possible_blocking_comps 

1398 ] 

1399 

1400 self._log_warning_for_blocking_components( 

1401 blocking_comp_names=possible_blocking_comps, blocking_comp_types=possible_blocking_comp_types 

1402 ) 

1403 return possible_blocking_comps, possible_blocking_comp_types

1429 def _write_component_outputs( 

1430 self, 

1431 *, 

1432 component_name: str, 

1433 component_outputs: Mapping[str, Any], 

1434 inputs: InputsType, 

1435 receivers: Sequence[tuple[str, OutputSocket, InputSocket, ConversionStrategyType]], 

1436 include_outputs_from: set[str], 

1437 ) -> Mapping[str, Any]: 

1438 """ 

1439 Distributes the outputs of a component to the input sockets that it is connected to. 

1440 

1441 :param component_name: The name of the component. 

1442 :param component_outputs: The outputs of the component. 

1443 :param inputs: The current global input state. 

1444 :param receivers: A sequence of tuples containing: 

1445 - receiver component name, 

1446 - output socket of the sender, 

1447 - input socket of the receiver, 

1448 - ConversionStrategy to be used to convert the value if required, otherwise None. 

1449 :param include_outputs_from: Set of component names that should always return an output from the pipeline. 

1450 """ 

1451 for receiver_name, sender_socket, receiver_socket, conversion_strategy in receivers: 

1452 # We either get the value that was produced by the actor or we use the _NO_OUTPUT_PRODUCED class to indicate 

1453 # that the sender did not produce an output for this socket. 

1454 # This allows us to track if a predecessor already ran but did not produce an output. 

1455 value = component_outputs.get(sender_socket.name, _NO_OUTPUT_PRODUCED) 

1456 

1457 if value is not _NO_OUTPUT_PRODUCED and conversion_strategy: 

1458 try: 

1459 value = _convert_value(value=value, conversion_strategy=conversion_strategy) 

1460 except Exception as e: 

1461 sender_node = self.graph.nodes.get(component_name) 

1462 sender_instance = sender_node.get("instance") if sender_node else None 

1463 sender_type_name = type(sender_instance).__name__ if sender_instance else "unknown" 

1464 

1465 receiver_node = self.graph.nodes.get(receiver_name) 

1466 receiver_instance = receiver_node.get("instance") if receiver_node else None 

1467 receiver_type_name = type(receiver_instance).__name__ if receiver_instance else "unknown" 

1468 

1469 msg = ( 

1470 f"Failed to perform conversion between components:\n" 

1471 f"Sender component: '{component_name}' (type: '{sender_type_name}')\n" 

1472 f"Sender socket: '{sender_socket.name}'\n" 

1473 f"Receiver component: '{receiver_name}' (type: '{receiver_type_name}')\n" 

1474 f"Receiver socket: '{receiver_socket.name}'\n" 

1475 f"Error: {e}" 

1476 ) 

1477 raise PipelineRuntimeError(component_name=None, component_type=None, message=msg) from e 

1478 

1479 if receiver_name not in inputs: 

1480 inputs[receiver_name] = {} 

1481 

1482 if receiver_socket.is_lazy_variadic: 

1483 # If the receiver socket is lazy variadic, we append the new input. 

1484 # Lazy variadic sockets can collect multiple inputs. 

1485 _write_to_lazy_variadic_socket( 

1486 inputs=inputs, 

1487 receiver_name=receiver_name, 

1488 receiver_socket_name=receiver_socket.name, 

1489 component_name=component_name, 

1490 value=value, 

1491 ) 

1492 else: 

1493 # If the receiver socket is not lazy variadic, it is greedy variadic or non-variadic. 

1494 # We overwrite with the new input if it's not _NO_OUTPUT_PRODUCED or if the current value is None. 

1495 _write_to_standard_socket( 

1496 inputs=inputs, 

1497 receiver_name=receiver_name, 

1498 receiver_socket_name=receiver_socket.name, 

1499 component_name=component_name, 

1500 value=value, 

1501 ) 

1502 

1503 # If we want to include all outputs from this actor in the final outputs, we don't need to prune any consumed 

1504 # outputs 

1505 if component_name in include_outputs_from: 

1506 return component_outputs 

1507 

1508 # We prune outputs that were consumed by any receiving sockets. 

1509 # All remaining outputs will be added to the final outputs of the pipeline. 

1510 consumed_outputs = {sender_socket.name for _, sender_socket, __, ___ in receivers} 

1511 return {key: value for key, value in component_outputs.items() if key not in consumed_outputs}

1514 def _is_queue_stale(priority_queue: FIFOPriorityQueue) -> bool: 

1515 """ 

1516 Checks if the priority queue needs to be recomputed because the priorities might have changed. 

1517 

1518 The queue is considered stale if it is empty or if the highest priority component is not READY or HIGHEST. 

1519 

1520 For example, if the next component in a queue has the priority READY then the equality becomes 

1521 ComponentPriority.READY > ComponentPriority.READY which is false. 

1522 However, if the next component has priority DEFER (or BLOCKED) then the equality becomes 

1523 ComponentPriority.DEFER > ComponentPriority.READY which is true, indicating that the queue is stale. 

1524 

1525 :param priority_queue: Priority queue of component names. 

1526 """ 

1527 return len(priority_queue) == 0 or priority_queue.peek()[0] > ComponentPriority.READY

1561 def _merge_super_component_pipelines(self) -> tuple[networkx.MultiDiGraph, dict[str, str]]: 

1562 """ 

1563 Merge the internal pipelines of SuperComponents into the main pipeline graph structure. 

1564 

1565 This creates a new networkx.MultiDiGraph containing all the components from both the main pipeline 

1566 and all the internal SuperComponents' pipelines. The SuperComponents are removed and their internal 

1567 components are connected to corresponding input and output sockets of the main pipeline. 

1568 

1569 :returns: 

1570 A tuple containing: 

1571 - A networkx.MultiDiGraph with the expanded structure of the main pipeline and all it's SuperComponents 

1572 - A dictionary mapping component names to boolean indicating that this component was part of a 

1573 SuperComponent 

1574 - A dictionary mapping component names to their SuperComponent name 

1575 """ 

1576 merged_graph = self.graph.copy() 

1577 super_component_mapping: dict[str, str] = {} 

1578 

1579 for super_name, super_component in self._find_super_components(): 

1580 internal_pipeline = super_component.pipeline # type: ignore 

1581 internal_graph = internal_pipeline.graph.copy() 

1582 

1583 # Mark all components in the internal pipeline as being part of a SuperComponent 

1584 for node in internal_graph.nodes(): 

1585 super_component_mapping[node] = super_name 

1586 

1587 # edges connected to the super component 

1588 incoming_edges = list(merged_graph.in_edges(super_name, data=True)) 

1589 outgoing_edges = list(merged_graph.out_edges(super_name, data=True)) 

1590 

1591 # merge the SuperComponent graph into the main graph and remove the super component node 

1592 # since its components are now part of the main graph 

1593 merged_graph = networkx.compose(merged_graph, internal_graph) 

1594 merged_graph.remove_node(super_name) 

1595 

1596 # get the entry and exit points of the SuperComponent internal pipeline 

1597 entry_points = [n for n in internal_graph.nodes() if internal_graph.in_degree(n) == 0] 

1598 exit_points = [n for n in internal_graph.nodes() if internal_graph.out_degree(n) == 0] 

1599 

1600 # connect the incoming edges to entry points 

1601 for sender, _, edge_data in incoming_edges: 

1602 sender_socket = edge_data["from_socket"] 

1603 for entry_point in entry_points: 

1604 # find a matching input socket in the entry point 

1605 entry_point_sockets = internal_graph.nodes[entry_point]["input_sockets"] 

1606 for socket_name, socket in entry_point_sockets.items(): 

1607 if _types_are_compatible(sender_socket.type, socket.type, self._connection_type_validation)[0]: 

1608 merged_graph.add_edge( 

1609 sender, 

1610 entry_point, 

1611 key=f"{sender_socket.name}/{socket_name}", 

1612 conn_type=_type_name(sender_socket.type), 

1613 from_socket=sender_socket, 

1614 to_socket=socket, 

1615 mandatory=socket.is_mandatory, 

1616 ) 

1617 

1618 # connect outgoing edges from exit points 

1619 for _, receiver, edge_data in outgoing_edges: 

1620 receiver_socket = edge_data["to_socket"] 

1621 for exit_point in exit_points: 

1622 # find a matching output socket in the exit point 

1623 exit_point_sockets = internal_graph.nodes[exit_point]["output_sockets"] 

1624 for socket_name, socket in exit_point_sockets.items(): 

1625 if _types_are_compatible(socket.type, receiver_socket.type, self._connection_type_validation)[ 

1626 0 

1627 ]: 

1628 merged_graph.add_edge( 

1629 exit_point, 

1630 receiver, 

1631 key=f"{socket_name}/{receiver_socket.name}", 

1632 conn_type=_type_name(socket.type), 

1633 from_socket=socket, 

1634 to_socket=receiver_socket, 

1635 mandatory=receiver_socket.is_mandatory, 

1636 ) 

1637 

1638 return merged_graph, super_component_mapping

1640 def _is_pipeline_possibly_blocked(self, current_pipeline_outputs: dict[str, Any]) -> bool: 

1641 """ 

1642 Heuristically determines whether the pipeline is possibly blocked based on its current outputs. 

1643 

1644 This method checks if the pipeline has produced any of the expected outputs. 

1645 - If no outputs are expected (i.e., `self.outputs()` returns an empty list), the method assumes the pipeline 

1646 is not blocked. 

1647 - If at least one expected output is present in `current_pipeline_outputs`, the pipeline is also assumed to not 

1648 be blocked. 

1649 - If none of the expected outputs are present, the pipeline is considered to be possibly blocked. 

1650 

1651 Note: This check is not definitive—it is intended as a best-effort guess to detect a stalled or misconfigured 

1652 pipeline when there are no more runnable components. 

1653 

1654 :param current_pipeline_outputs: A dictionary of outputs currently produced by the pipeline. 

1655 :returns: 

1656 bool: True if the pipeline is possibly blocked (i.e., expected outputs are missing), False otherwise. 

1657 """ 

1658 expected_outputs = self.outputs() 

1659 return bool(expected_outputs) and not any(k in current_pipeline_outputs for k in expected_outputs)
