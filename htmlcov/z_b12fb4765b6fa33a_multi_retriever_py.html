18@_experimental 

19@component 

20class MultiRetriever: 

21 """ 

22 A component that accepts text retrievers and runs them in parallel, combining their results. 

23 

24 > **Note:** This component is experimental and may change or be removed in future releases without prior 

25 deprecation notice. 

26 

27 All retrievers must implement the `TextRetriever` protocol. Use `TextEmbeddingRetriever` to wrap an 

28 embedding-based retriever before passing it to this component. 

29 

30 Each retriever is queried concurrently using a thread pool. 

31 The results are deduplicated and returned as a single list of documents. 

32 

33 ### Usage example 

34 

35 ```python 

36 from haystack import Document 

37 from haystack.document_stores.in_memory import InMemoryDocumentStore 

38 from haystack.document_stores.types import DuplicatePolicy 

39 from haystack.components.retrievers import InMemoryBM25Retriever, InMemoryEmbeddingRetriever 

40 from haystack.components.retrievers import TextEmbeddingRetriever, MultiRetriever 

41 from haystack.components.embedders import SentenceTransformersTextEmbedder, SentenceTransformersDocumentEmbedder 

42 from haystack.components.writers import DocumentWriter 

43 

44 documents = [ 

45 Document(content="Renewable energy is energy that is collected from renewable resources."), 

46 Document(content="Solar energy is a type of green energy that is harnessed from the sun."), 

47 Document(content="Wind energy is another type of green energy that is generated by wind turbines."), 

48 ] 

49 

50 # Populate the document store 

51 doc_store = InMemoryDocumentStore() 

52 doc_embedder = SentenceTransformersDocumentEmbedder(model="sentence-transformers/all-MiniLM-L6-v2") 

53 doc_writer = DocumentWriter(document_store=doc_store, policy=DuplicatePolicy.SKIP) 

54 doc_writer.run(documents=doc_embedder.run(documents)["documents"]) 

55 

56 # Run the multi-retriever with all retrievers 

57 retriever = MultiRetriever( 

58 retrievers={ 

59 "bm25": InMemoryBM25Retriever(document_store=doc_store), 

60 "embedding": TextEmbeddingRetriever( 

61 retriever=InMemoryEmbeddingRetriever(document_store=doc_store), 

62 text_embedder=SentenceTransformersTextEmbedder(model="sentence-transformers/all-MiniLM-L6-v2"), 

63 ), 

64 }, 

65 top_k=3, 

66 ) 

67 

68 # Run all retrievers 

69 result = retriever.run(query="green energy sources") 

70 

71 # Run only the BM25 retriever 

72 result = retriever.run(query="green energy sources", active_retrievers=["bm25"]) 

73 

74 for doc in result["documents"]: 

75 print(doc.content) 

76 ``` 

77 """

79 def __init__( 

80 self, 

81 *, 

82 retrievers: dict[str, TextRetriever], 

83 filters: dict[str, Any] | None = None, 

84 top_k: int = 10, 

85 max_workers: int = 4, 

86 join_mode: Literal["concatenate", "reciprocal_rank_fusion"] = "reciprocal_rank_fusion", 

87 ) -> None: 

88 """ 

89 Create the MultiRetriever component. 

90 

91 :param retrievers: 

92 A dictionary mapping names to text retrievers (implementing the `TextRetriever` protocol) to run in 

93 parallel. 

94 :param filters: 

95 A dictionary of filters to apply when retrieving documents. 

96 :param top_k: 

97 The maximum number of documents to return per retriever. 

98 :param max_workers: 

99 The maximum number of threads to use for parallel retrieval. 

100 :param join_mode: 

101 How to merge results from multiple retrievers. Available modes: 

102 - `concatenate`: Combines all results into a single list and deduplicates. 

103 - `reciprocal_rank_fusion`: Deduplicates and assigns scores based on reciprocal rank fusion. 

104 """ 

105 self.retrievers = retrievers 

106 self.filters = filters 

107 self.top_k = top_k 

108 self.max_workers = max_workers 

109 self.join_mode = join_mode 

110 self._is_warmed_up = False 

111 

112 def _merge_results(self, document_lists: list[list[Document]]) -> list[Document]: 

113 """ 

114 Merge per-retriever result lists according to `join_mode`. 

115 

116 In `concatenate` mode, all lists are flattened and deduplicated. In `reciprocal_rank_fusion` mode, results 

117 are deduplicated and re-scored using RRF, then returned in descending score order. 

118 """ 

119 if self.join_mode == "reciprocal_rank_fusion": 

120 documents = _reciprocal_rank_fusion(document_lists) 

121 return sorted(documents, key=lambda d: d.score if d.score is not None else -inf, reverse=True) 

122 return _deduplicate_documents([doc for docs in document_lists for doc in docs]) 

123 

124 def _resolve_retrievers(self, active_retrievers: list[str] | None) -> dict[str, TextRetriever]: 

125 """ 

126 Returns the subset of retrievers to run based on the active_retrievers list. 

127 

128 :param active_retrievers: 

129 A list of retriever names to run. If None, all retrievers are returned. 

130 

131 :returns: 

132 A dictionary of retriever names to retriever instances. 

133 

134 :raises ValueError: 

135 If any name in `active_retrievers` does not match a retriever name. 

136 """ 

137 if active_retrievers is None: 

138 return self.retrievers 

139 unknown = set(active_retrievers) - self.retrievers.keys() 

140 if unknown: 

141 raise ValueError( 

142 f"Unknown retriever name(s): {sorted(unknown)}. Available retrievers: {sorted(self.retrievers.keys())}" 

143 ) 

144 return {name: self.retrievers[name] for name in active_retrievers} 

145 

146 def warm_up(self) -> None: 

147 """ 

148 Warm up the retrievers if any has a warm_up method. 

149 """ 

150 if self._is_warmed_up: 

151 return 

152 for retriever in self.retrievers.values(): 

153 if hasattr(retriever, "warm_up") and callable(retriever.warm_up): 

154 retriever.warm_up() 

155 self._is_warmed_up = True 

156 

157 @component.output_types(documents=list[Document]) 

158 def run( 

159 self, 

160 query: str, 

161 filters: dict[str, Any] | None = None, 

162 top_k: int | None = None, 

163 *, 

164 active_retrievers: list[str] | None = None, 

165 ) -> dict[str, list[Document]]: 

166 """ 

167 Runs retrievers in parallel on the given query and returns deduplicated results. 

168 

169 :param query: 

170 The query to run the retrievers on. 

171 :param filters: 

172 Filters to apply. Defaults to the value set at initialization. 

173 :param top_k: 

174 Maximum documents to return per retriever. Defaults to the value set at initialization. 

175 :param active_retrievers: 

176 Names of retrievers to run. Defaults to all. Must match keys in the `retrievers` dictionary. 

177 

178 :returns: 

179 A dictionary with the keys: 

180 - "documents": A deduplicated list of retrieved documents. 

181 

182 :raises ValueError: 

183 If any name in `active_retrievers` does not match a retriever name. 

184 """ 

185 if not self._is_warmed_up: 

186 self.warm_up() 

187 

188 resolved_top_k = top_k if top_k is not None else self.top_k 

189 resolved_filters = filters if filters is not None else self.filters 

190 

191 retrievers_to_run = self._resolve_retrievers(active_retrievers) 

192 

193 results_by_name: dict[str, list[Document]] = {} 

194 with ThreadPoolExecutor(max_workers=self.max_workers) as executor: 

195 future_to_name = { 

196 executor.submit(retriever.run, query=query, filters=resolved_filters, top_k=resolved_top_k): name 

197 for name, retriever in retrievers_to_run.items() 

198 } 

199 for future in as_completed(future_to_name): 

200 name = future_to_name[future] 

201 try: 

202 results_by_name[name] = future.result().get("documents", []) 

203 except Exception as e: 

204 raise RuntimeError(f"Retriever '{name}' failed: {e}") from e 

205 

206 document_lists = [results_by_name[name] for name in retrievers_to_run] 

207 return {"documents": self._merge_results(document_lists)}

209 @component.output_types(documents=list[Document]) 

210 async def run_async( 

211 self, 

212 query: str, 

213 filters: dict[str, Any] | None = None, 

214 top_k: int | None = None, 

215 *, 

216 active_retrievers: list[str] | None = None, 

217 ) -> dict[str, list[Document]]: 

218 """ 

219 Runs retrievers concurrently on the given query and returns deduplicated results. 

220 

221 Uses each retriever's `run_async` method if available, otherwise runs `run` in a thread executor. 

222 

223 :param query: 

224 The query to run the retrievers on. 

225 :param filters: 

226 Filters to apply. Defaults to the value set at initialization. 

227 :param top_k: 

228 Maximum documents to return per retriever. Defaults to the value set at initialization. 

229 :param active_retrievers: 

230 Names of retrievers to run. Defaults to all. Must match keys in the `retrievers` dictionary. 

231 

232 :returns: 

233 A dictionary with the keys: 

234 - "documents": A deduplicated list of retrieved documents. 

235 

236 :raises ValueError: 

237 If any name in `active_retrievers` does not match a retriever name. 

238 """ 

239 if not self._is_warmed_up: 

240 self.warm_up() 

241 

242 resolved_top_k = top_k if top_k is not None else self.top_k 

243 resolved_filters = filters if filters is not None else self.filters 

244 

245 retrievers_to_run = self._resolve_retrievers(active_retrievers) 

246 

247 loop = asyncio.get_running_loop() 

248 

249 async def _run_one(name: str, retriever: TextRetriever) -> list[Document]: 

250 try: 

251 if hasattr(retriever, "run_async") and callable(retriever.run_async): 

252 result = await retriever.run_async(query=query, filters=resolved_filters, top_k=resolved_top_k) 

253 else: 

254 result = await loop.run_in_executor( 

255 None, lambda: retriever.run(query=query, filters=resolved_filters, top_k=resolved_top_k) 

256 ) 

257 return result.get("documents", []) 

258 except Exception as e: 

259 raise RuntimeError(f"Retriever '{name}' failed: {e}") from e 

260 

261 document_lists = list(await asyncio.gather(*[_run_one(name, r) for name, r in retrievers_to_run.items()])) 

262 return {"documents": self._merge_results(document_lists)}

265 """

267

271 return default_to_dict(

273 retrievers={name: component_to_dict(obj=r, name=name) for name, r in self.retrievers.items()}, 

274 filters=self.filters, 

275 top_k=self.top_k, 

276 max_workers=self.max_workers, 

277 join_mode=self.join_mode, 

278 )

280 @classmethod 

281 def from_dict(cls, data: dict[str, Any]) -> "MultiRetriever": 

282 """ 

283 Creates an instance of the component from a dictionary. 

284 

285 :param data: 

286 Dictionary with the data to create the component. 

287 """ 

288 retrievers_data = data.get("init_parameters", {}).get("retrievers", {}) 

289 if retrievers_data: 

290 retrievers = {} 

291 for name, retriever_data in retrievers_data.items(): 

292 try: 

293 imported_class = import_class_by_name(retriever_data["type"]) 

294 except ImportError as e: 

295 raise ImportError( 

296 f"Could not import class {retriever_data['type']} for retriever '{name}'. Error: {str(e)}" 

297 ) from e 

298 retrievers[name] = component_from_dict(cls=imported_class, data=retriever_data, name=name) 

299 data["init_parameters"]["retrievers"] = retrievers 

300 return default_from_dict(cls, data)
