5from pathlib import Path 

6from typing import Any

8from haystack.core.serialization import default_from_dict, default_to_dict 

9from haystack.hooks.tool_result_offloading.types import ToolResultStore 

10 

11 

12class FileSystemToolResultStore(ToolResultStore): 

13 """ 

14 A `ToolResultStore` that writes offloaded tool results to files under a root directory on the local file system. 

15 

16 ```python 

17 from haystack.hooks.tool_result_offloading import FileSystemToolResultStore 

18 

19 store = FileSystemToolResultStore(root="tool_results") 

20 reference = store.write(key="search_1.txt", content="...") 

21 store.read(reference) 

22 ``` 

23 """ 

24 

25 def __init__(self, root: str | Path) -> None: 

26 """ 

27 Initialize the store with the root directory results are written under. 

28 

29 :param root: Directory under which result files are written. Created on first write if it does not exist. 

30 """ 

31 self.root = Path(root) 

32 

33 def write(self, *, key: str, content: str) -> str: 

34 """ 

35 Write `content` to `<root>/<key>`, creating parent directories, and return the file path. 

36 

37 The resolved target must stay within the root directory: a `key` that escapes it (e.g. containing `../` or an 

38 absolute path) is rejected, so a tool-provided key cannot write outside the store. 

39 

40 :param key: Relative file name for the result within the store root. 

41 :param content: The tool result to persist. 

42 :returns: The absolute path the content was written to, as a string, for use with `read`. 

43 :raises ValueError: If `key` resolves to a location outside the store root. 

44 """ 

45 root = self.root.resolve() 

46 path = (root / key).resolve() 

47 if not path.is_relative_to(root): 

48 raise ValueError(f"Result key '{key}' resolves outside the store root '{root}'.") 

49 path.parent.mkdir(parents=True, exist_ok=True) 

50 path.write_text(content, encoding="utf-8") 

51 return str(path) 

52 

53 def read(self, reference: str) -> str: 

54 """ 

55 Read back the content previously written to `reference`. 

56 

57 :param reference: A path returned by `write`. 

58 :returns: The stored content. 

59 """ 

60 return Path(reference).read_text(encoding="utf-8") 

61 

62 def to_dict(self) -> dict[str, Any]: 

63 """ 

64 Serialize the store, storing its root directory as a string. 

65 

66 :returns: A dictionary representation of the store. 

67 """ 

68 return default_to_dict(self, root=str(self.root)) 

69 

70 @classmethod 

71 def from_dict(cls, data: dict[str, Any]) -> "FileSystemToolResultStore": 

72 """ 

73 Deserialize the store from a dictionary. 

74 

75 :param data: A dictionary representation produced by `to_dict`. 

76 :returns: The deserialized `FileSystemToolResultStore`. 

77 """ 

78 return default_from_dict(cls, data)
